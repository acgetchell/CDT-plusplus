107 SCENARIO("Function_ref operations", "[function_ref]")
108 {
109  GIVEN("A simple lambda stored in a function_ref.")
110  {
111  auto increment_lambda = [](int a) { return ++a; };
112  function_ref<int(int)> lambda_ref(increment_lambda);
113  WHEN("Function_ref is called with 0.")
114  {
115  THEN("We should get 1.") { REQUIRE(lambda_ref(1) == 2); }
116  }
117  }
118  GIVEN("A complex lambda storing a move stored in a function_ref.")
119  {
120  constexpr auto simplices = static_cast<std::int_fast32_t>(6400);
121  constexpr auto timeslices = static_cast<std::int_fast32_t>(7);
122  SimplicialManifold universe(simplices, timeslices);
123  Move_tracker attempted_moves{{0, 0, 0, 0, 0}};
124  // Verify triangulation
125  CHECK(universe.triangulation);
126  CHECK(universe.geometry->number_of_cells() ==
127  universe.triangulation->number_of_finite_cells());
128  CHECK(universe.geometry->number_of_edges() ==
129  universe.triangulation->number_of_finite_edges());
130  CHECK(universe.geometry->N0() ==
131  universe.triangulation->number_of_vertices());
132  CHECK(universe.triangulation->dimension() == 3);
133  CHECK(fix_timeslices(universe.triangulation));
134  CHECK(universe.triangulation->is_valid());
135  CHECK(universe.triangulation->tds().is_valid());
136 
137  VolumePerTimeslice(universe);
138 
139  CHECK(universe.geometry->max_timevalue().get() == timeslices);
140  CHECK(universe.geometry->min_timevalue().get() == 1);
141  // Previous state
142  auto N3_31_pre_move = universe.geometry->N3_31();
143  auto N3_22_pre_move = universe.geometry->N3_22();
144  auto N3_13_pre_move = universe.geometry->N3_13();
145  auto N1_TL_pre_move = universe.geometry->N1_TL();
146  auto N1_SL_pre_move = universe.geometry->N1_SL();
147  auto N0_pre_move = universe.geometry->N0();
148  // No moves recorded
149  CHECK(attempted_moves[0] == 0);
150 
151  auto move_23_lambda =
152  [](SimplicialManifold manifold,
153  Move_tracker& attempted_moves) -> SimplicialManifold {
154  return make_23_move(std::move(manifold), attempted_moves);
155  };
156  function_ref<SimplicialManifold(SimplicialManifold, Move_tracker&)>
157  complex_ref(move_23_lambda);
158  WHEN("The function_ref is invoked.")
159  {
160  REQUIRE_NOTHROW(universe = complex_ref(universe, attempted_moves));
161  THEN(
162  "The move from the function_ref is correct and the triangulation "
163  "invariants are maintained.")
164  {
165  // The triangulation is still valid
166  CHECK(universe.triangulation->tds().is_valid(true));
167  CHECK(universe.triangulation->dimension() == 3);
168  CHECK(fix_timeslices(universe.triangulation));
169  // The move is correct
170  CHECK(universe.geometry->N3_31() == N3_31_pre_move);
171  CHECK(universe.geometry->N3_22() == N3_22_pre_move + 1);
172  CHECK(universe.geometry->N3_13() == N3_13_pre_move);
173  CHECK(universe.geometry->N1_TL() == N1_TL_pre_move + 1);
174  CHECK(universe.geometry->N1_SL() == N1_SL_pre_move);
175  CHECK(universe.geometry->N0() == N0_pre_move);
176  // Move attempts were recorded
177  CHECK_FALSE(attempted_moves[0] == 0);
178  std::cout << "There were " << attempted_moves[0]
179  << " attempted (2,3) moves.\n";
180  }
181  }
