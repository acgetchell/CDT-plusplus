339  template <int dimension>
340  [[nodiscard]] auto expected_cell_type(Cell_handle_t<dimension> const& t_cell)
341  {
342 #ifndef NDEBUG
343  spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
344 #endif
345  std::vector<int> vertex_timevalues;
346  // There are d+1 vertices in a d-dimensional simplex
347  for (auto i = 0; i < dimension + 1; ++i)
348  {
349  // Obtain timevalue of vertex
350  vertex_timevalues.emplace_back(t_cell->vertex(i)->info());
351  }
352  auto const maxtime_ref =
353  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
354  auto const mintime_ref =
355  std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
356  auto maxtime = *maxtime_ref;
357  auto mintime = *mintime_ref;
358  // A properly foliated simplex should have a timevalue difference of 1
359  if (maxtime - mintime != 1 || maxtime == mintime)
360  {
361 #ifndef NDEBUG
362  spdlog::trace("This simplex is acausal:\n");
363  spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
364  mintime);
365  spdlog::trace("--\n");
366 #endif
367  return Cell_type::ACAUSAL;
368  }
369  std::multiset<int> const timevalues{vertex_timevalues.begin(),
370  vertex_timevalues.end()};
371  auto max_vertices = timevalues.count(maxtime);
372  auto min_vertices = timevalues.count(mintime);
373 
374  // 3D simplices
375  if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
376  if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
377  if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
378 
379  // 4D simplices
380  if (max_vertices == 4 && min_vertices == 1) { return Cell_type::ONE_FOUR; }
381  if (max_vertices == 3 && min_vertices == 2) { return Cell_type::TWO_THREE; }
382  if (max_vertices == 2 && min_vertices == 3) { return Cell_type::THREE_TWO; }
383  if (max_vertices == 1 && min_vertices == 4) { return Cell_type::FOUR_ONE; }
384 
385  // If we got here, there's some kind of error
386 #ifndef NDEBUG
387  spdlog::trace("This simplex has an error:\n");
388  spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
389  mintime);
390  spdlog::trace(
391  "There are {} vertices with the max timevalue and {} vertices with "
392  "the min timevalue.\n",
393  max_vertices, min_vertices);
394  spdlog::trace("--\n");
395 #endif
396  return Cell_type::UNCLASSIFIED;
397  } // expected_cell_type

614  template <int dimension>
615  [[nodiscard]] auto find_bad_vertex(Cell_handle_t<dimension> const& cell)
616  -> Vertex_handle_t<dimension>
617  {
618 #ifndef NDEBUG
619  spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
620  spdlog::debug("===Invalid Cell===\n");
621  std::vector<Cell_handle_t<dimension>> bad_cell{cell};
622  debug_print_cells<dimension>(bad_cell);
623 #endif
624  std::multimap<Int_precision, Vertex_handle_t<dimension>> vertices;
625  for (int i = 0; i < dimension + 1; ++i)
626  {
627  vertices.emplace(
628  std::make_pair(cell->vertex(i)->info(), cell->vertex(i)));
629  }
630  // Now it's sorted in the multimap
631  auto const minvalue = vertices.cbegin()->first;
632  auto const maxvalue = vertices.crbegin()->first;
633  auto const minvalue_count = vertices.count(minvalue);
634  auto const maxvalue_count = vertices.count(maxvalue);
635  // Return the vertex with the highest value if there are equal or more
636  // vertices with lower values. Note that we preferentially return higher
637  // timeslice vertices because there are typically more cells at higher
638  // timeslices (see expected_points_per_timeslice())
639  if (minvalue_count >= maxvalue_count) { return vertices.rbegin()->second; }
640  return vertices.begin()->second;
641  } // find_bad_vertex

651  -> bool

666  t_triangulation.remove(vertices_to_remove.begin(),

669  Ensures(t_triangulation.is_valid());
670  return vertices_to_remove.empty();
671  }

685  template <int dimension>
686  [[nodiscard]] auto make_foliated_ball(
687  Int_precision const t_simplices, Int_precision const t_timeslices,
688  double const initial_radius = INITIAL_RADIUS,
689  double const foliation_spacing = FOLIATION_SPACING)
690  {
691  Causal_vertices_t<dimension> causal_vertices;
692  causal_vertices.reserve(static_cast<std::size_t>(t_simplices));
693  const auto points_per_timeslice = utilities::expected_points_per_timeslice(
694  dimension, t_simplices, t_timeslices);
695  Expects(points_per_timeslice >= 2);
696

698  {

719  template <int dimension>
720  [[nodiscard]] auto make_triangulation(
721  Int_precision const t_simplices, Int_precision const t_timeslices,
722  double const initial_radius = INITIAL_RADIUS,
723  double const foliation_spacing = FOLIATION_SPACING)
724  -> Delaunay_t<dimension>
725  {
726 #ifndef NDEBUG
727  spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
728 #endif
729  fmt::print("\nGenerating universe ...\n");
730 #ifdef CGAL_LINKED_WITH_TBB
731  // Construct the locking data-structure
732  // using the bounding-box of the points
733  auto bounding_box_size = static_cast<double>(t_timeslices + 1);
734  typename Delaunay_t<dimension>::Lock_data_structure locking_ds{
735  CGAL::Bbox_3{-bounding_box_size, -bounding_box_size, -bounding_box_size,
736  bounding_box_size, bounding_box_size, bounding_box_size},
737  50};
738  Delaunay_t<dimension> triangulation =
739  Delaunay_t<dimension>{TriangulationTraits<3>::Kernel{}, &locking_ds};
740 #else
741  Delaunay_t<dimension> triangulation = Delaunay_t<dimension>{};
742 #endif
743 
744  // Make initial triangulation
745  auto causal_vertices = make_foliated_ball<dimension>(
746  t_simplices, t_timeslices, initial_radius, foliation_spacing);
747  triangulation.insert(causal_vertices.begin(), causal_vertices.end());
748 
749  // Fix vertices
750  while (!check_vertices<dimension>(triangulation, initial_radius,
751  foliation_spacing))
752  {
753 #ifndef NDEBUG
754  fmt::print("Deleting incorrect vertices ....\n");
755 #endif
756  fix_vertices<dimension>(triangulation, initial_radius, foliation_spacing);
757  }
758 
759  // Fix timeslices
760  int passes = 1;
