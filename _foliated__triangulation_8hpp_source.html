445  template <typename Triangulation>
446  [[nodiscard]] auto check_timeslices(Triangulation&& triangulation) const
447  -> std::optional<std::vector<Vertex_handle>>
448  {
449  std::vector<Vertex_handle> invalid_vertices;
450  // Iterate over all cells in the triangulation
451  for (Delaunay3::Finite_cells_iterator cit =
452  triangulation.finite_cells_begin();
453  cit != triangulation.finite_cells_end(); ++cit)
454  {
455  Ensures(cit->is_valid());
456  std::multimap<int, Vertex_handle> this_cell;
457  // Collect a map of timevalues and vertices in each cell
458  for (int i = 0; i < 4; ++i)
459  {
460  this_cell.emplace(
461  std::make_pair(cit->vertex(i)->info(), cit->vertex(i)));
462  }
463  // Now it's sorted in the multimap
464  auto minvalue = this_cell.cbegin()->first;
465  auto maxvalue = this_cell.crbegin()->first;
466 
467 #ifdef DETAILED_DEBUGGING
468  auto min_vertex = this_cell.cbegin()->second;
469  auto max_vertex = this_cell.crbegin()->second;
470  fmt::print("Smallest timevalue in this cell is: {}\n", minvalue);
471  fmt::print("Largest timevalue in this cell is: {}\n", maxvalue);
472  fmt::print("Min vertex info() {}\n", min_vertex->info());
473  fmt::print("Max vertex info() {}\n", max_vertex->info());
474 #endif
475  // There must be a timevalue delta of 1 for a validly foliated simplex
476  if (maxvalue - minvalue == 1)
477  {
478 #ifdef DETAILED_DEBUGGING
479  fmt::print("This cell is valid.\n");
480 #endif
481  }
482  else
483  {
484  auto minvalue_count = this_cell.count(minvalue);
485  auto maxvalue_count = this_cell.count(maxvalue);
486 #ifdef DETAILED_DEBUGGING
487  fmt::print("This cell is invalid.\n");
488  fmt::print("There are {} vertices with the minvalue.\n",
489  minvalue_count);
490  fmt::print("There are {} vertices with the maxvalue.\n",
491  maxvalue_count);
492  fmt::print("So we should remove ");
493 #endif
494 
495  if (minvalue_count > maxvalue_count)
496  {
497  invalid_vertices.emplace_back(this_cell.rbegin()->second);
498 #ifdef DETAILED_DEBUGGING
499  fmt::print("maxvalue.\n");
500 #endif
501  }
502  else
503  {
504  invalid_vertices.emplace_back(this_cell.begin()->second);
505 #ifdef DETAILED_DEBUGGING
506  fmt::print("minvalue.\n");
507 #endif
508  }
509  }
510  }
511  if (invalid_vertices.empty()) { return std::nullopt; }
512  else
513  {
514 #ifdef DETAILED_DEBUGGING
515  fmt::print("Removing ...\n");
516  for (auto& v : invalid_vertices)
517  {
518  fmt::print("Vertex {} with timevalue {}\n", v->point(), v->info());
519  }
520 #endif
521  return invalid_vertices;
522  }
523  } // check_timeslices

572  [[nodiscard]] Causal_vertices make_foliated_sphere(
573  std::int_fast64_t const simplices, std::int_fast64_t const timeslices,
574  double const initial_radius = INITIAL_RADIUS,
575  double const radial_factor = RADIAL_FACTOR) const
576  {
577  Causal_vertices causal_vertices;
578  causal_vertices.reserve(static_cast<std::size_t>(simplices));
579  const auto points_per_timeslice =
580  expected_points_per_timeslice(3, simplices, timeslices);
581  Expects(points_per_timeslice >= 2);
582

585  {

661  [[nodiscard]] auto classify_cells(std::vector<Cell_handle> const& cells,
662  bool debugging = false) const
663  -> std::vector<Cell_handle>
664  {
665  Expects(cells.size() == number_of_finite_cells());
666  std::vector<Vertex_handle> cell_vertices;
667  cell_vertices.reserve(4);
668  std::vector<int> vertex_timevalues;
669  vertex_timevalues.reserve(4);
670  for (auto const& c : cells)
671  {
672  if (debugging)
673  {
674  fmt::print("Cell info was {}\n", c->info());
675  }
676 
677  for (int j = 0; j < 4; ++j)
678  {
679  cell_vertices.emplace_back(c->vertex(j));
680  vertex_timevalues.emplace_back(c->vertex(j)->info());
681  if (debugging)
682  {
683  fmt::print("Cell vertex {} has timevalue {}\n", j,
684  c->vertex(j)->info());
685  }
686  }
687 
688  // This is simply not sufficient. Need to check *both* maxtime and
689  // min_time, and that there are exactly 1 and 3, 2 and 2, or 3 and 1.
690  // Anything else means we have an invalid simplex which we should
691  // also return.
692  // We also need to check that maxtime - min_time = 1, else we have
693  // a mis-classified vertex (probably)
694  auto maxtime =
695  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
696  auto maxtime_vertices = std::count_if(
697  cell_vertices.begin(), cell_vertices.end(),
698  [maxtime](auto const& vertex) { return vertex->info() == *maxtime; });
699  // Check maxtime - min_time here
700  switch (maxtime_vertices)
701  {
702  case 1:
703  c->info() = static_cast<int>(Cell_type::THREE_ONE);
704  break;
705  case 2:
706  c->info() = static_cast<int>(Cell_type::TWO_TWO);
707  break;
708  case 3:
709  c->info() = static_cast<int>(Cell_type::ONE_THREE);
710  break;
711  default:
712  throw std::logic_error("Mis-classified cell.");
713  }
714  if (debugging)
715  {
716  fmt::print("Max timevalue is {}\n", *maxtime);
717  fmt::print("There are {} vertices with max timeslice in the cell.\n",
718  maxtime_vertices);
719  fmt::print("Cell info is now {}\n", c->info());
720  fmt::print("--\n");
721  }
722  cell_vertices.clear();
723  vertex_timevalues.clear();
724  }
725  return cells;
726  } // classify_cells
