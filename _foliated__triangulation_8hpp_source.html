189  [[nodiscard]] inline auto expected_cell_type(Cell_handle_3 const& t_cell,
190  bool t_debug_flag = false)
191  {
192  std::vector<int> vertex_timevalues;
193  for (int i = 0; i < 4; ++i)
194  {
195  // Obtain timevalue of vertex
196  vertex_timevalues.emplace_back(t_cell->vertex(i)->info());
197  }
198  auto maxtime_ref =
199  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
200  auto mintime_ref =
201  std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
202  auto maxtime = *maxtime_ref;
203  auto mintime = *mintime_ref;
204  // A properly foliated simplex should have a timevalue difference of 1
205  if (maxtime - mintime != 1) { return Cell_type::ERROR; }
206  std::multiset<int> timevalues{vertex_timevalues.begin(),
207  vertex_timevalues.end()};
208  auto max_vertices = timevalues.count(maxtime);
209  auto min_vertices = timevalues.count(mintime);
210 
211  if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
212  if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
213  if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
214 
215  // If we got here, there's some kind of error
216  if (t_debug_flag)
217  {
218  fmt::print("This simplex has an error.\n");
219  fmt::print("Max timevalue is {}\n", maxtime);
220  fmt::print("There are {} vertices with the max timevalue.\n",
221  max_vertices);
222  fmt::print("Min timevalue is {}\n", mintime);
223  fmt::print("There are {} vertices with the min timevalue.\n",
224  min_vertices);
225  fmt::print("--\n");
226  }
227  return Cell_type::ERROR;
228  } // expected_cell_type

290  [[nodiscard]] inline auto make_foliated_sphere(
291  Int_precision const t_simplices, Int_precision const t_timeslices,
292  double const initial_radius = INITIAL_RADIUS,
293  double const foliation_spacing = FOLIATION_SPACING) -> Causal_vertices
294  {
295  Causal_vertices causal_vertices;
296  causal_vertices.reserve(static_cast<std::size_t>(t_simplices));
297  const auto points_per_timeslice =
298  expected_points_per_timeslice(3, t_simplices, t_timeslices);
299  Expects(points_per_timeslice >= 2);
300 
301  using Spherical_points_generator = CGAL::Random_points_on_sphere_3<Point_3>;
302  for (gsl::index i = 0; i < t_timeslices; ++i)
303  {
304  auto radius = initial_radius + static_cast<double>(i) * foliation_spacing;
305  Spherical_points_generator gen{static_cast<double>(radius)};
306  // Generate random points at the radius
307  for (gsl::index j = 0;
308  j < static_cast<Int_precision>(points_per_timeslice * radius); ++j)
309  {
310  causal_vertices.emplace_back(std::make_pair(*gen++, i + 1));
311  } // j
312  } // i
313  return causal_vertices;
314  } // make_foliated_sphere

322  [[nodiscard]] inline auto volume_per_timeslice(
323  std::vector<Face_handle_3> const& t_facets, bool t_debug_flag = false)
324  -> std::multimap<Int_precision, Facet_3>
325  {
326  std::multimap<Int_precision, Facet_3> space_faces;
327  for (auto const& face : t_facets)
328  {
329  Cell_handle_3 ch = face.first;
330  auto index_of_facet = face.second;
331  if (t_debug_flag) { fmt::print("Facet index is {}\n", index_of_facet); }
332  std::set<Int_precision> facet_timevalues;
333  for (int i = 0; i < 4; ++i)
334  {
335  if (i != index_of_facet)
336  {
337  if (t_debug_flag)
338  {
339  fmt::print("Vertex[{}] has timevalue {}\n", i,
340  ch->vertex(i)->info());
341  }
342  facet_timevalues.insert(ch->vertex(i)->info());
343  }
344  }
345  // If we have a 1-element set then all timevalues on that facet are
346  // equal
347  if (facet_timevalues.size() == 1)
348  {
349  if (t_debug_flag)
350  {
351  fmt::print("Facet is spacelike on timevalue {}.\n",
352  *facet_timevalues.begin());
353  }
354  space_faces.insert({*facet_timevalues.begin(), face});
355  }
356  else
357  {
358  if (t_debug_flag) { fmt::print("Facet is timelike.\n"); }
359  }
360  }
361  return space_faces;
362  } // volume_per_timeslice

953  template <typename Triangulation>
954  [[nodiscard]] auto check_timeslices(Triangulation&& t_triangulation) const
955  -> std::optional<std::vector<Vertex_handle_3>>
956  {
957  std::vector<Vertex_handle_3> invalid_vertices;
958  // Iterate over all cells in the triangulation
959  for (Delaunay3::Finite_cells_iterator cit =
960  t_triangulation.finite_cells_begin();
961  cit != t_triangulation.finite_cells_end(); ++cit)
962  {
963  Ensures(cit->is_valid());
964  std::multimap<int, Vertex_handle_3> this_cell;
965  // Collect a map of timevalues and vertices in each cell
966  for (int i = 0; i < 4; ++i)
967  {
968  this_cell.emplace(
969  std::make_pair(cit->vertex(i)->info(), cit->vertex(i)));
970  }
971  // Now it's sorted in the multimap
972  auto minvalue = this_cell.cbegin()->first;
973  auto maxvalue = this_cell.crbegin()->first;
974 
975 #ifdef DETAILED_DEBUGGING
976  auto min_vertex = this_cell.cbegin()->second;
977  auto max_vertex = this_cell.crbegin()->second;
978  fmt::print("Smallest timevalue in this cell is: {}\n", minvalue);
979  fmt::print("Largest timevalue in this cell is: {}\n", maxvalue);
980  fmt::print("Min vertex info() {}\n", min_vertex->info());
981  fmt::print("Max vertex info() {}\n", max_vertex->info());
982 #endif
983  // There must be a timevalue delta of 1 for a validly foliated simplex
984  if (maxvalue - minvalue == 1)
985  {
986 #ifdef DETAILED_DEBUGGING
987  fmt::print("This cell is valid.\n");
988 #endif
989  }
990  else
991  {
992  auto minvalue_count = this_cell.count(minvalue);
993  auto maxvalue_count = this_cell.count(maxvalue);
994 #ifdef DETAILED_DEBUGGING
995  fmt::print("This cell is invalid.\n");
996  fmt::print("There are {} vertices with the minvalue.\n",
997  minvalue_count);
998  fmt::print("There are {} vertices with the maxvalue.\n",
999  maxvalue_count);
1000  fmt::print("So we should remove ");
1001 #endif
1002 
1003  if (minvalue_count > maxvalue_count)
1004  {
1005  invalid_vertices.emplace_back(this_cell.rbegin()->second);
1006 #ifdef DETAILED_DEBUGGING
1007  fmt::print("maxvalue.\n");
1008 #endif
1009  }
1010  else
1011  {
1012  invalid_vertices.emplace_back(this_cell.begin()->second);
1013 #ifdef DETAILED_DEBUGGING
1014  fmt::print("minvalue.\n");
1015 #endif
1016  }
1017  }
1018  }
1019  if (invalid_vertices.empty()) { return std::nullopt; }
1020 
1021 #ifdef DETAILED_DEBUGGING
1022  fmt::print("Removing ...\n");
1023  for (auto& v : invalid_vertices)
1024  {
1025  fmt::print("Vertex {} with timevalue {}\n", v->point(), v->info());
1026  }
1027 #endif
1028  return invalid_vertices;
1029 
1030  } // check_timeslices
