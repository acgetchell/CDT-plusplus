460  template <typename Triangulation>
461  [[nodiscard]] auto check_timeslices(Triangulation&& t_triangulation) const
462  -> std::optional<std::vector<Vertex_handle>>
463  {
464  std::vector<Vertex_handle> invalid_vertices;
465  // Iterate over all cells in the triangulation
466  for (Delaunay3::Finite_cells_iterator cit =
467  t_triangulation.finite_cells_begin();
468  cit != t_triangulation.finite_cells_end(); ++cit)
469  {
470  Ensures(cit->is_valid());
471  std::multimap<int, Vertex_handle> this_cell;
472  // Collect a map of timevalues and vertices in each cell
473  for (int i = 0; i < 4; ++i)
474  {
475  this_cell.emplace(
476  std::make_pair(cit->vertex(i)->info(), cit->vertex(i)));
477  }
478  // Now it's sorted in the multimap
479  auto minvalue = this_cell.cbegin()->first;
480  auto maxvalue = this_cell.crbegin()->first;
481 
482 #ifdef DETAILED_DEBUGGING
483  auto min_vertex = this_cell.cbegin()->second;
484  auto max_vertex = this_cell.crbegin()->second;
485  fmt::print("Smallest timevalue in this cell is: {}\n", minvalue);
486  fmt::print("Largest timevalue in this cell is: {}\n", maxvalue);
487  fmt::print("Min vertex info() {}\n", min_vertex->info());
488  fmt::print("Max vertex info() {}\n", max_vertex->info());
489 #endif
490  // There must be a timevalue delta of 1 for a validly foliated simplex
491  if (maxvalue - minvalue == 1)
492  {
493 #ifdef DETAILED_DEBUGGING
494  fmt::print("This cell is valid.\n");
495 #endif
496  }
497  else
498  {
499  auto minvalue_count = this_cell.count(minvalue);
500  auto maxvalue_count = this_cell.count(maxvalue);
501 #ifdef DETAILED_DEBUGGING
502  fmt::print("This cell is invalid.\n");
503  fmt::print("There are {} vertices with the minvalue.\n",
504  minvalue_count);
505  fmt::print("There are {} vertices with the maxvalue.\n",
506  maxvalue_count);
507  fmt::print("So we should remove ");
508 #endif
509 
510  if (minvalue_count > maxvalue_count)
511  {
512  invalid_vertices.emplace_back(this_cell.rbegin()->second);
513 #ifdef DETAILED_DEBUGGING
514  fmt::print("maxvalue.\n");
515 #endif
516  }
517  else
518  {
519  invalid_vertices.emplace_back(this_cell.begin()->second);
520 #ifdef DETAILED_DEBUGGING
521  fmt::print("minvalue.\n");
522 #endif
523  }
524  }
525  }
526  if (invalid_vertices.empty()) { return std::nullopt; }
527  else
528  {
529 #ifdef DETAILED_DEBUGGING
530  fmt::print("Removing ...\n");
531  for (auto& v : invalid_vertices)
532  { fmt::print("Vertex {} with timevalue {}\n", v->point(), v->info()); }
533 #endif
534  return invalid_vertices;
535  }
536  } // check_timeslices

585  [[nodiscard]] Causal_vertices make_foliated_sphere(
586  Int_precision const t_simplices, Int_precision const t_timeslices,
587  long double const initial_radius = INITIAL_RADIUS,
588  long double const radial_factor = RADIAL_FACTOR) const
589  {

679  [[nodiscard]] auto classify_cells(std::vector<Cell_handle> const& cells,
680  bool t_debug_flag = false) const
681  -> std::vector<Cell_handle>
682  {
683  Expects(cells.size() == number_of_finite_cells());
684  std::vector<Vertex_handle> cell_vertices;
685  cell_vertices.reserve(4);
686  std::vector<int> vertex_timevalues;
687  vertex_timevalues.reserve(4);
688  for (auto const& c : cells)
689  {
690  if (t_debug_flag) { fmt::print("Cell info was {}\n", c->info()); }
691 
692  for (int j = 0; j < 4; ++j)
693  {
694  cell_vertices.emplace_back(c->vertex(j));
695  vertex_timevalues.emplace_back(c->vertex(j)->info());
696  if (t_debug_flag)
697  {
698  fmt::print("Cell vertex {} has timevalue {}\n", j,
699  c->vertex(j)->info());
700  }
701  }
702 
703  // This is simply not sufficient. Need to check *both* maxtime and
704  // min_time, and that there are exactly 1 and 3, 2 and 2, or 3 and 1.
705  // Anything else means we have an invalid simplex which we should
706  // also return.
707  // We also need to check that maxtime - min_time = 1, else we have
708  // a mis-classified vertex (probably)
709  auto maxtime =
710  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
711  auto maxtime_vertices = std::count_if(
712  cell_vertices.begin(), cell_vertices.end(),
713  [maxtime](auto const& vertex) { return vertex->info() == *maxtime; });
714  // Check maxtime - min_time here
715  switch (maxtime_vertices)
716  {
717  case 1:
718  c->info() = static_cast<int>(Cell_type::THREE_ONE);
719  break;
720  case 2:
721  c->info() = static_cast<int>(Cell_type::TWO_TWO);
722  break;
723  case 3:
724  c->info() = static_cast<int>(Cell_type::ONE_THREE);
725  break;
726  default:
727  throw std::logic_error("Mis-classified cell.");
728  }
729  if (t_debug_flag)
730  {
731  fmt::print("Max timevalue is {}\n", *maxtime);
732  fmt::print("There are {} vertices with max timeslice in the cell.\n",
733  maxtime_vertices);
734  fmt::print("Cell info is now {}\n", c->info());
735  fmt::print("--\n");
736  }
737  cell_vertices.clear();
738  vertex_timevalues.clear();
739  }
740  return cells;
741  } // classify_cells

787  facet_timevalues.insert(ch->vertex(i)->info());
