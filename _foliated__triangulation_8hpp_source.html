481  template <int dimension>
482  [[nodiscard]] auto expected_cell_type(Cell_handle_t<dimension> const& t_cell)
483  {
484 #ifndef NDEBUG
485  spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
486 #endif
487  std::array<int, static_cast<std::size_t>(dimension) + 1>
488  vertex_timevalues{};
489  // There are d+1 vertices in a d-dimensional simplex
490  for (auto i = 0; i < dimension + 1; ++i)
491  {
492  // Obtain timevalue of vertex
493  vertex_timevalues.at(static_cast<std::size_t>(i)) =
494  t_cell->vertex(i)->info();
495  }
496  auto const maxtime_ref =
497  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
498  auto const mintime_ref =
499  std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
500  auto maxtime = *maxtime_ref;
501  auto mintime = *mintime_ref;
502  // A properly foliated simplex should have a timevalue difference of 1
503  if (maxtime - mintime != 1 || maxtime == mintime)
504  {
505 #ifndef NDEBUG
506  spdlog::trace("This simplex is acausal:\n");
507  spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
508  mintime);
509  spdlog::trace("--\n");
510 #endif
511  return Cell_type::ACAUSAL;
512  }
513  std::multiset<int> const timevalues{vertex_timevalues.begin(),
514  vertex_timevalues.end()};
515  auto max_vertices = timevalues.count(maxtime);
516  auto min_vertices = timevalues.count(mintime);
517 
518  // 3D simplices
519  if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
520  if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
521  if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
522 
523  // 4D simplices
524  if (max_vertices == 4 && min_vertices == 1) { return Cell_type::ONE_FOUR; }
525  if (max_vertices == 3 && min_vertices == 2) { return Cell_type::TWO_THREE; }
526  if (max_vertices == 2 && min_vertices == 3) { return Cell_type::THREE_TWO; }
527  if (max_vertices == 1 && min_vertices == 4) { return Cell_type::FOUR_ONE; }
528 
529  // If we got here, there's some kind of error
530 #ifndef NDEBUG
531  spdlog::trace("This simplex has an error:\n");
532  spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
533  mintime);
534  spdlog::trace(
535  "There are {} vertices with the max timevalue and {} vertices with "
536  "the min timevalue.\n",
537  max_vertices, min_vertices);
538  spdlog::trace("--\n");
539 #endif
540  return Cell_type::UNCLASSIFIED;
541  } // expected_cell_type

789  // timeslice vertices because there are typically more cells at higher
790  // timeslices (see expected_points_per_timeslice())
791  return (minvalue_count >= maxvalue_count) ? vertices.rbegin()->second

837  [[nodiscard]] auto make_foliated_ball(
838  Int_precision t_simplices, Int_precision t_timeslices,
839  double initial_radius = INITIAL_RADIUS,
840  double foliation_spacing = FOLIATION_SPACING)
841  {
842  Causal_vertices_t<dimension> causal_vertices;
843  causal_vertices.reserve(static_cast<std::size_t>(t_simplices));
844  auto const points_per_timeslice = utilities::expected_points_per_timeslice(
845  dimension, t_simplices, t_timeslices);
846  Expects(points_per_timeslice >= 2);
847

849  {

851  initial_radius + static_cast<double>(i) * foliation_spacing;
852  Spherical_points_generator_t<dimension> gen{radius};
853  // Generate random points at the radius
854  for (gsl::index j = 0;
855  j < static_cast<Int_precision>(points_per_timeslice * radius); ++j)

872  Int_precision t_simplices, Int_precision t_timeslices,
873  double initial_radius = INITIAL_RADIUS,
874  double foliation_spacing = FOLIATION_SPACING) -> Delaunay_t<dimension>
875  {
876 #ifndef NDEBUG

878 #endif
879  fmt::print("\nGenerating universe ...\n");
880 #ifdef CGAL_LINKED_WITH_TBB

882  // using the bounding-box of the points
883  auto bounding_box_size = static_cast<double>(t_timeslices + 1);

893 #endif

898  triangulation.insert(causal_vertices.begin(), causal_vertices.end());

904  foliation_spacing))
905  {
906  break;
907  }
908 #ifndef NDEBUG
909  spdlog::warn("Deleting incorrect vertices pass #{}\n", passes);
910 #endif
911  }
912 
913  // Fix timeslices
914  for (auto passes = 1; passes < MAX_FIX_PASSES + 1; ++passes)

916  if (!fix_timevalues<dimension>(triangulation)) { break; }
917 #ifndef NDEBUG
918  spdlog::warn("Fixing timeslices pass #{}\n", passes);
919 #endif
920  }

928 #endif
929  }

933  return triangulation;
