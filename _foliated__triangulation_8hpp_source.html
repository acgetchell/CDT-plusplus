98 [[nodiscard]] inline auto pad_locking_box(CGAL::Bbox_3 const& box)
99 -> CGAL::Bbox_3
100 {
101 auto const padding = std::max({1.0, (box.xmax() - box.xmin()) * 0.01,
102 (box.ymax() - box.ymin()) * 0.01,
103 (box.zmax() - box.zmin()) * 0.01});
104 return {box.xmin() - padding, box.ymin() - padding, box.zmin() - padding,
105 box.xmax() + padding, box.ymax() + padding, box.zmax() + padding};
106 }
107
108 [[nodiscard]] inline auto default_locking_box() -> CGAL::Bbox_3
109 {
110 auto const extent = static_cast<double>(GV_BOUNDING_BOX_SIZE);
111 return {-extent, -extent, -extent, extent, extent, extent};
112 }
113
114 template <typename Iterator, typename Point_projection>
115 [[nodiscard]] auto locking_box(Iterator first, Iterator last,
116 Point_projection point_for) -> CGAL::Bbox_3
117 {
118 if (first == last) { return default_locking_box(); }
119
120 auto const box = std::accumulate(
121 std::next(first), last, std::invoke(point_for, *first).bbox(),
122 [&point_for](CGAL::Bbox_3 accumulated, auto const& value) {
123 return accumulated + std::invoke(point_for, value).bbox();
124 });
125 return pad_locking_box(box);
126 }

132 return locking_box(causal_vertices.begin(), causal_vertices.end(),
133 [](auto const& causal_vertex) -> auto const& {

172 static_assert(std::is_nothrow_swappable_v<Lock_owner>,
173 "Delaunay_state swap requires a non-throwing lock owner.");
174 static_assert(std::is_nothrow_move_constructible_v<Lock_owner> &&
175 std::is_nothrow_move_constructible_v<Delaunay>,
176 "Delaunay_state move construction requires non-throwing "

217 [[nodiscard]] static auto make_adopted_state(Delaunay source)
218 -> Pending_state
219 {

221 CDT_ENABLE_PARALLEL_TRIANGULATION

223 locking_box<dimension>(source), LOCK_GRID_RESOLUTION);
224 source.set_lock_data_structure(lock.get());
225 return {std::move(lock), std::move(source)};
226#else
227 source.set_lock_data_structure(nullptr);

229#endif

346 template <int dimension>
347 auto make_causal_vertices(std::span<Point_t<dimension> const> vertices,
348 std::span<size_t const> timevalues)
349 -> Causal_vertices_t<dimension>
350 {
351 if (vertices.size() != timevalues.size())
352 {
353 throw std::length_error("Vertices and timevalues must be the same size.");
354 }
355 Causal_vertices_t<dimension> causal_vertices;
356 causal_vertices.reserve(vertices.size());
357 std::ranges::transform(
358 vertices, timevalues, std::back_inserter(causal_vertices),
359 [](Point_t<dimension> point, size_t time) {
360 if (!std::in_range<Int_precision>(time))
361 {
362 throw std::out_of_range("Timevalue does not fit Int_precision.");
363 }
364 return std::pair{point, static_cast<Int_precision>(time)};
365 });
366 return causal_vertices;
367 }

752 template <int dimension>
753 [[nodiscard]] auto expected_cell_type(Cell_handle_t<dimension> const& t_cell)
754 {
755#ifndef NDEBUG
756 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
757#endif
758 std::array<int, static_cast<std::size_t>(dimension) + 1>
759 vertex_timevalues{};
760 // There are d+1 vertices in a d-dimensional simplex
761 for (auto i = 0; i < dimension + 1; ++i)
762 {
763 // Obtain timevalue of vertex
764 vertex_timevalues.at(static_cast<std::size_t>(i)) =
765 t_cell->vertex(i)->info();
766 }
767 auto const maxtime_ref =
768 std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
769 auto const mintime_ref =
770 std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
771 auto maxtime = *maxtime_ref;
772 auto mintime = *mintime_ref;
773 // A properly foliated simplex should have a timevalue difference of 1
774 if (maxtime - mintime != 1 || maxtime == mintime)
775 {
776#ifndef NDEBUG
777 spdlog::trace("This simplex is acausal:\n");
778 spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
779 mintime);
780 spdlog::trace("--\n");
781#endif
782 return Cell_type::ACAUSAL;
783 }
784 std::multiset<int> const timevalues{vertex_timevalues.begin(),
785 vertex_timevalues.end()};
786 auto max_vertices = timevalues.count(maxtime);
787 auto min_vertices = timevalues.count(mintime);
788
789 // 3D simplices
790 if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
791 if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
792 if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
793
794 // If we got here, there's some kind of error
795#ifndef NDEBUG
796 spdlog::trace("This simplex has an error:\n");

960 [[nodiscard]] auto collect_spacelike_facets(Container const& t_facets)
961 -> std::vector<std::pair<Int_precision, Facet_t<3>>>
962 {
963#ifndef NDEBUG
964 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
965#endif
966 using Volume_entry = std::pair<Int_precision, Facet_t<3>>;
967 std::vector<Volume_entry> space_faces;
968 if constexpr (std::ranges::sized_range<Container>)
969 {
970 space_faces.reserve(std::ranges::size(t_facets));
971 }
972 for (auto const& face : t_facets)
973 {
974 Cell_handle_t<dimension> const cell = face.first;

976#ifndef NDEBUG
977 spdlog::trace("Facet index is {}\n", index_of_facet);
978#endif
979 std::set<Int_precision> facet_timevalues;
980 // There are d+1 vertices in a d-dimensional simplex
981 for (int i = 0; i < dimension + 1; ++i)
982 {
983 if (i != index_of_facet)
984 {
985#ifndef NDEBUG
986 spdlog::trace("Vertex[{}] has timevalue {}\n", i,
987 cell->vertex(i)->info());
988#endif
989 facet_timevalues.insert(cell->vertex(i)->info());
990 }
991 }
992 // If we have a 1-element set then all timevalues on that facet are
993 // equal
994 if (facet_timevalues.size() == 1)
995 {
996#ifndef NDEBUG
997 spdlog::trace("Facet is spacelike on timevalue {}.\n",
998 *facet_timevalues.begin());
999#endif
1000 space_faces.emplace_back(*facet_timevalues.begin(), face);
1001 }
1002 else
1003 {
1004#ifndef NDEBUG
1005 spdlog::trace("Facet is timelike.\n");
1006#endif
1007 }
1008 }
1009 std::ranges::stable_sort(
1010 space_faces, std::ranges::less{},
1011 [](Volume_entry const& entry) noexcept { return entry.first; });
1012 return space_faces;

1065 template <int dimension>
1066 [[nodiscard]] auto find_bad_vertex(Cell_handle_t<dimension> const& cell)
1067 -> Vertex_handle_t<dimension>
1068 {
1069#ifndef NDEBUG
1070 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
1071 spdlog::debug("===Invalid Cell===\n");
1072 std::vector<Cell_handle_t<dimension>> bad_cell{cell};
1073 debug_print_cells<dimension>(std::span{bad_cell});
1074#endif
1075 std::multimap<Int_precision, Vertex_handle_t<dimension>> vertices;
1076 for (int i = 0; i < dimension + 1; ++i)
1077 {
1078 vertices.emplace(
1079 std::make_pair(cell->vertex(i)->info(), cell->vertex(i)));
1080 }
1081 // Now it's sorted in the multimap
1082 auto const minvalue = vertices.cbegin()->first;
1083 auto const maxvalue = vertices.crbegin()->first;
1084 auto const minvalue_count = vertices.count(minvalue);
1085 auto const maxvalue_count = vertices.count(maxvalue);
1086 // Return the vertex with the highest value if there are equal or more
1087 // vertices with lower values. Note that we preferentially return higher
1088 // timeslice vertices because there are typically more cells at higher
1089 // timeslices (see expected_points_per_timeslice())
1090 return minvalue_count >= maxvalue_count ? vertices.rbegin()->second
1091 : vertices.begin()->second;
1092 } // find_bad_vertex

1100 [[nodiscard]] auto fix_timevalues(Delaunay_t<dimension>& t_triangulation)
1101 -> bool
1102 {
1103 // Obtain a container of cells that are incorrectly foliated
1104 if (auto invalid_cells = check_timevalues<dimension>(t_triangulation);

1120 vertices_to_remove.end());
1121 assert(t_triangulation.tds().is_valid());
1122 assert(t_triangulation.is_valid());

1124 }
1125 return false;

1141 template <int dimension, std::uniform_random_bit_generator Generator>
1142 [[nodiscard]] auto make_foliated_ball(Int_precision const t_simplices,
1143 Int_precision const t_timeslices,
1144 double const initial_radius,
1145 double const foliation_spacing,
1146 Generator& generator)
1147 {
1148 if (t_simplices < 2 || t_timeslices < 2)
1149 {
1150 throw std::invalid_argument(
1151 "Simplices and timeslices must each be at least 2.");
1152 }
1153 if (!std::isfinite(initial_radius) || initial_radius <= 0.0)
1154 {
1155 throw std::invalid_argument(
1156 "Initial radius must be finite and positive.");
1157 }
1158 if (!std::isfinite(foliation_spacing) || foliation_spacing <= 0.0)
1159 {
1160 throw std::invalid_argument(
1161 "Foliation spacing must be finite and positive.");
1162 }
1163
1164 auto const population = utilities::generated_population_bounds(
1165 dimension, t_simplices, t_timeslices, initial_radius,
1166 foliation_spacing);
1167 if (population.points_per_timeslice < 2)
1168 {
1169 throw std::invalid_argument(
1170 "Simplices and timeslices would create an empty triangulation.");
1171 }
1172 if (!std::isfinite(population.last_layer_points) ||
1173 population.last_layer_points >
1174 static_cast<long double>(std::numeric_limits<Int_precision>::max()))
1175 {
1176 throw std::out_of_range(
1177 "Foliation parameters generate too many points per timeslice.");
1178 }
1179
1180 Causal_vertices_t<dimension> causal_vertices;
1181 causal_vertices.reserve(static_cast<std::size_t>(t_simplices));
1182 std::uniform_int_distribution<unsigned int> seed_distribution;
1183 CGAL::Random cgal_random{seed_distribution(generator)};
1184
1185 for (gsl::index i = 0; i < t_timeslices; ++i)
1186 {
1187 auto const radius =
1188 initial_radius + static_cast<double>(i) * foliation_spacing;
1189 auto const generated_points =
1190 static_cast<long double>(population.points_per_timeslice) * radius;
1191 if (!std::isfinite(radius) || generated_points < 2.0L)
1192 {
1193 throw std::invalid_argument(
1194 "Foliation parameters do not populate every timeslice.");
1195 }
1196 if (generated_points >
1197 static_cast<long double>(std::numeric_limits<Int_precision>::max()))
1198 {
1199 throw std::out_of_range(
1200 "Foliation parameters generate too many points per timeslice.");
1201 }
1202 Spherical_points_generator_t<dimension> gen{radius, cgal_random};
1203 // Generate random points at the radius
1204 for (gsl::index j = 0; j < static_cast<Int_precision>(generated_points);
1205 ++j)
1206 {
1207 causal_vertices.emplace_back(*gen++, i + 1);
1208 } // j
1209 } // i
1210 if (causal_vertices.size() < static_cast<std::size_t>(dimension + 1))
1211 {
1212 throw std::invalid_argument("Parameters create an empty triangulation.");
1213 }
1214 return causal_vertices;
1215 } // make_foliated_ball

1235 [[nodiscard]] auto make_triangulation(Int_precision const t_simplices,
1236 Int_precision t_timeslices,
1237 double const initial_radius,
1238 double const foliation_spacing,
1239 Generator& generator)
1240 -> Delaunay_t<dimension>
1241 {
1242#ifndef NDEBUG
1243 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
1244#endif
1245 fmt::print("\nGenerating universe ...\n");
1246 // Make initial triangulation

1248 make_foliated_ball<dimension>(t_simplices, t_timeslices, initial_radius,
