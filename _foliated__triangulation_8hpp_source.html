557  // t_triangulation) const -> std::optional<std::vector<Vertex_handle>>

639  [[nodiscard]] auto make_triangulation(
640  Int_precision const t_simplices, Int_precision const t_timeslices,
641  long double const initial_radius = INITIAL_RADIUS,
642  long double const radial_factor = RADIAL_FACTOR) -> Delaunay3
643  {
644  fmt::print("Generating universe ...\n");
645 #ifdef CGAL_LINKED_WITH_TBB
646  // Construct the locking data-structure
647  // using the bounding-box of the points
648  auto bounding_box_size = static_cast<double>(t_timeslices + 1);
649  Delaunay3::Lock_data_structure locking_ds{
650  CGAL::Bbox_3{-bounding_box_size, -bounding_box_size, -bounding_box_size,
651  bounding_box_size, bounding_box_size, bounding_box_size},
652  50};
653  Delaunay3 triangulation = Delaunay3{Kernel{}, &locking_ds};
654 #else
655  Delaunay3 triangulation = Delaunay3{};
656 #endif
657 
658  auto causal_vertices = make_foliated_sphere(t_simplices, t_timeslices,
659  initial_radius, radial_factor);
660  triangulation.insert(causal_vertices.begin(), causal_vertices.end());
661  int passes = 1;
662  while (!fix_timeslices(triangulation))
663  {
664 #ifndef NDEBUG
665  fmt::print("Fix pass #{}\n", passes);
666 #endif
667  ++passes;
668  }
669  print_triangulation(triangulation);
670  Ensures(!check_timeslices(triangulation));
671  return triangulation;
672  }
679  [[nodiscard]] static auto make_foliated_sphere(
680  Int_precision const t_simplices, Int_precision const t_timeslices,
681  long double const initial_radius = INITIAL_RADIUS,
682  long double const radial_factor = RADIAL_FACTOR) -> Causal_vertices
683  {
684  Causal_vertices causal_vertices;
685  causal_vertices.reserve(static_cast<std::size_t>(t_simplices));
686  const auto points_per_timeslice =
687  expected_points_per_timeslice(3, t_simplices, t_timeslices);
688  Expects(points_per_timeslice >= 2);
689 
690  using Spherical_points_generator = CGAL::Random_points_on_sphere_3<Point>;
691  for (gsl::index i = 0; i < t_timeslices; ++i)
692  {
693  auto radius =
694  initial_radius + static_cast<long double>(i) * radial_factor;
695  Spherical_points_generator gen{static_cast<double>(radius)};
696  // Generate random points at the radius
697  for (gsl::index j = 0;
698  j < static_cast<Int_precision>(points_per_timeslice * radius); ++j)
699  { causal_vertices.emplace_back(std::make_pair(*gen++, i + 1)); } // j
700  } // i
701  return causal_vertices;
702  } // make_foliated_sphere

722  template <typename DelaunayTriangulation>
723  [[nodiscard]] auto fix_timeslices(DelaunayTriangulation&& t_dt) -> bool
724  {
725  auto vertices_to_delete = check_timeslices(t_dt);
726  std::set<Vertex_handle> deleted_vertices;
727  // Remove duplicates
728  if (vertices_to_delete)
729  {
730  for (auto& v : vertices_to_delete.value())
731  { deleted_vertices.emplace(v); }
732  }
733  auto invalid = deleted_vertices.size();
734 
735  // Delete invalid vertices
736  t_dt.remove(deleted_vertices.begin(), deleted_vertices.end());
737  // Check that the triangulation is still valid
738  // Turned off by -DCGAL_TRIANGULATION_NO_POSTCONDITIONS
739  // CGAL_triangulation_expensive_postcondition(universe_ptr->is_valid());
740  // if (!_delaunay.tds().is_valid())
741  // throw std::logic_error("Delaunay tds invalid!");
742  Ensures(t_dt.tds().is_valid());
743  Ensures(t_dt.is_valid());
744 
745 #ifndef NDEBUG
746  fmt::print("There are {} invalid simplices.\n", invalid);
747 #endif
748  return invalid == 0;

773  [[nodiscard]] auto classify_cells(std::vector<Cell_handle> const& cells,
774  bool t_debug_flag = false) const
775  -> std::vector<Cell_handle>
776  {
777  Expects(cells.size() == number_of_finite_cells());
778  std::vector<Vertex_handle> cell_vertices;
779  cell_vertices.reserve(4);
780  std::vector<int> vertex_timevalues;
781  vertex_timevalues.reserve(4);
782  for (auto const& c : cells)
783  {
784  if (t_debug_flag) { fmt::print("Cell info was {}\n", c->info()); }
785 
786  for (int j = 0; j < 4; ++j)
787  {
788  cell_vertices.emplace_back(c->vertex(j));
789  vertex_timevalues.emplace_back(c->vertex(j)->info());
790  if (t_debug_flag)
791  {
792  fmt::print("Cell vertex {} has timevalue {}\n", j,
793  c->vertex(j)->info());
794  }
795  }
796 
797  // This is simply not sufficient. Need to check *both* maxtime and
798  // min_time, and that there are exactly 1 and 3, 2 and 2, or 3 and 1.
799  // Anything else means we have an invalid simplex which we should
800  // also return.
801  // We also need to check that maxtime - min_time = 1, else we have
802  // a mis-classified vertex (probably)
803  auto maxtime =
804  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
805  auto maxtime_vertices = std::count_if(
806  cell_vertices.begin(), cell_vertices.end(),
807  [maxtime](auto const& vertex) { return vertex->info() == *maxtime; });
808  // Check maxtime - min_time here
809  switch (maxtime_vertices)
810  {
811  case 1:
812  c->info() = static_cast<int>(Cell_type::THREE_ONE);
813  break;
814  case 2:
815  c->info() = static_cast<int>(Cell_type::TWO_TWO);
816  break;
817  case 3:
818  c->info() = static_cast<int>(Cell_type::ONE_THREE);
819  break;
820  default:
821  throw std::logic_error("Mis-classified cell.");
822  }
823  if (t_debug_flag)
824  {
825  fmt::print("Max timevalue is {}\n", *maxtime);
826  fmt::print("There are {} vertices with max timeslice in the cell.\n",
827  maxtime_vertices);
828  fmt::print("Cell info is now {}\n", c->info());
829  fmt::print("--\n");
830  }
831  cell_vertices.clear();
832  vertex_timevalues.clear();
833  }
834  return cells;
835  } // classify_cells
