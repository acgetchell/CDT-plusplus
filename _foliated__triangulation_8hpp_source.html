91 [[nodiscard]] inline auto pad_locking_box(CGAL::Bbox_3 const& box)
92 -> CGAL::Bbox_3
93 {
94 auto const padding = std::max({1.0, (box.xmax() - box.xmin()) * 0.01,
95 (box.ymax() - box.ymin()) * 0.01,
96 (box.zmax() - box.zmin()) * 0.01});
97 return {box.xmin() - padding, box.ymin() - padding, box.zmin() - padding,
98 box.xmax() + padding, box.ymax() + padding, box.zmax() + padding};
99 }
100
101 [[nodiscard]] inline auto default_locking_box() -> CGAL::Bbox_3
102 {
103 auto const extent = static_cast<double>(GV_BOUNDING_BOX_SIZE);
104 return {-extent, -extent, -extent, extent, extent, extent};
105 }
106
107 template <typename Iterator, typename Point_projection>
108 [[nodiscard]] auto locking_box(Iterator first, Iterator last,
109 Point_projection point_for) -> CGAL::Bbox_3
110 {
111 if (first == last) { return default_locking_box(); }
112
113 auto const box = std::accumulate(
114 std::next(first), last, std::invoke(point_for, *first).bbox(),
115 [&point_for](CGAL::Bbox_3 accumulated, auto const& value) {
116 return accumulated + std::invoke(point_for, value).bbox();
117 });
118 return pad_locking_box(box);
119 }

126 [](auto const& causal_vertex) -> auto const& {

146 template <int dimension>
147 class Delaunay_state
148 {
149 public:
150 using Delaunay = Delaunay_t<dimension>;
151 using Kernel = typename TriangulationTraits<dimension>::Kernel;
152
153#ifdef CGAL_LINKED_WITH_TBB
154 using Lock_data_structure = typename Delaunay::Lock_data_structure;
155 using Lock_owner = std::unique_ptr<Lock_data_structure>;
156#else
157 struct Lock_owner
158 {};
159#endif
160
161 static_assert(
162 noexcept(std::declval<Delaunay&>().swap(std::declval<Delaunay&>())),
163 "Delaunay_state swap requires CGAL's swap to be noexcept.");
164 static_assert(std::is_nothrow_swappable_v<Lock_owner>,
165 "Delaunay_state swap requires a non-throwing lock owner.");
166 static_assert(std::is_nothrow_move_constructible_v<Lock_owner> &&
167 std::is_nothrow_move_constructible_v<Delaunay>,
168 "Delaunay_state move construction requires non-throwing "
169 "members.");
170
171 private:
172 Lock_owner m_lock_data_structure;
173 Delaunay m_triangulation;
174
175 struct Pending_state
176 {
177 Lock_owner lock_data_structure;
178 Delaunay triangulation;
179 };
180
181 [[nodiscard]] static auto make_empty_state() -> Pending_state
182 {
183#ifdef CGAL_LINKED_WITH_TBB
184 auto lock = std::make_unique<Lock_data_structure>(
185 locking_box<dimension>(Delaunay{}), LOCK_GRID_RESOLUTION);
186 Delaunay triangulation{Kernel{}, lock.get()};
187 return {std::move(lock), std::move(triangulation)};
188#else
189 return {Lock_owner{}, Delaunay{}};
190#endif
191 }
192
193 [[nodiscard]] static auto make_insertion_state(
194 Causal_vertices_t<dimension> const& causal_vertices) -> Pending_state
195 {
196#ifdef CGAL_LINKED_WITH_TBB
197 auto lock = std::make_unique<Lock_data_structure>(
198 locking_box<dimension>(causal_vertices), LOCK_GRID_RESOLUTION);
199 Delaunay triangulation{Kernel{}, lock.get()};
200 return {std::move(lock), std::move(triangulation)};
201#else
202 static_cast<void>(causal_vertices);
203 return {Lock_owner{}, Delaunay{}};
204#endif
205 }
206
207 [[nodiscard]] static auto make_adopted_state(Delaunay source)
208 -> Pending_state
209 {
210#ifdef CGAL_LINKED_WITH_TBB
211 auto lock = std::make_unique<Lock_data_structure>(
212 locking_box<dimension>(source), LOCK_GRID_RESOLUTION);
213 source.set_lock_data_structure(lock.get());
214 return {std::move(lock), std::move(source)};
215#else
216 source.set_lock_data_structure(nullptr);
217 return {Lock_owner{}, std::move(source)};
218#endif
219 }
220
221 explicit Delaunay_state(Pending_state state) noexcept
222 : m_lock_data_structure{std::move(state.lock_data_structure)}
223 , m_triangulation{std::move(state.triangulation)}
224 { state.triangulation.set_lock_data_structure(nullptr); }
225
226 public:
227 Delaunay_state() : Delaunay_state{make_empty_state()} {}
228
229 explicit Delaunay_state(
230 Causal_vertices_t<dimension> const& causal_vertices)
231 : Delaunay_state{make_insertion_state(causal_vertices)}
232 {
233 m_triangulation.insert(causal_vertices.begin(), causal_vertices.end());
234 }
235
236 explicit Delaunay_state(Delaunay source)
237 : Delaunay_state{make_adopted_state(std::move(source))}
238 {}

240 Delaunay_state(Delaunay_state const& other)
241 : Delaunay_state{Delaunay{other.m_triangulation}}
242 {}
243
244 Delaunay_state(Delaunay_state&& other) noexcept
245 : m_lock_data_structure{std::move(other.m_lock_data_structure)}
246 , m_triangulation{std::move(other.m_triangulation)}
247 { other.m_triangulation.set_lock_data_structure(nullptr); }
248
249 friend void swap(Delaunay_state& lhs, Delaunay_state& rhs) noexcept
250 {
251 lhs.m_triangulation.swap(rhs.m_triangulation);
252 using std::swap;
253 swap(lhs.m_lock_data_structure, rhs.m_lock_data_structure);
254 }
255
256 auto operator=(Delaunay_state const& other) -> Delaunay_state&
257 {
258 if (this != &other)
259 {
260 Delaunay_state copy{other};
261 swap(*this, copy);
262 }
263 return *this;
264 }
265
266 auto operator=(Delaunay_state&& other) noexcept -> Delaunay_state&
267 {
268 if (this != &other) { swap(*this, other); }
269 return *this;
270 }

302#endif

328 auto make_causal_vertices(std::span<Point_t<dimension> const> vertices,
329 std::span<size_t const> timevalues)
330 -> Causal_vertices_t<dimension>
331 {
332 if (vertices.size() != timevalues.size())
333 {
334 throw std::length_error("Vertices and timevalues must be the same size.");

337 causal_vertices.reserve(vertices.size());

340 [](Point_t<dimension> point, size_t time) {

342 });
343 return causal_vertices;

352 template <int dimension>
353 [[nodiscard]] auto collect_edges(Delaunay_t<dimension> const& delaunay)
354 {
355 assert(delaunay.is_valid());
356 std::vector<Edge_handle_t<dimension>> init_edges;
357 init_edges.reserve(delaunay.number_of_finite_edges());
358 for (auto eit = delaunay.finite_edges_begin();
359 eit != delaunay.finite_edges_end(); ++eit)
360 {
361 Cell_handle_t<3> const cell = eit->first;
362 Edge_handle_t<3> thisEdge{cell, cell->index(cell->vertex(eit->second)),
363 cell->index(cell->vertex(eit->third))};
364 // Each edge is valid in the triangulation
365 assert(delaunay.tds().is_valid(thisEdge.first, thisEdge.second,
366 thisEdge.third));
367 init_edges.emplace_back(thisEdge);
368 }
369 assert(init_edges.size() == delaunay.number_of_finite_edges());
370 return init_edges;
371 } // collect_edges

430 template <int dimension, detail::ContainerType Container>
431 [[nodiscard]] auto find_max_timevalue(Container&& t_vertices) -> Int_precision
432 {
433 auto vertices = std::forward<Container>(t_vertices);
434 if (vertices.empty())
435 {
436 throw std::invalid_argument("Cannot classify an empty triangulation.");
437 }
438 auto max_element = std::max_element(vertices.begin(), vertices.end(),
439 compare_v_info<dimension>);
440 auto result_index = std::distance(vertices.begin(), max_element);
441 // std::distance may be negative if random-access iterators are used and
442 // first is reachable from last
443 assert(result_index >= 0);
444 auto const index = static_cast<std::size_t>(std::abs(result_index));
445 return vertices[index]->info();
446 } // find_max_timevalue

452 [[nodiscard]] auto find_min_timevalue(Container&& t_vertices) -> Int_precision
453 {
454 auto vertices = std::forward<Container>(t_vertices);
455 if (vertices.empty())
456 {
457 throw std::invalid_argument("Cannot classify an empty triangulation.");
458 }
459 auto min_element = std::min_element(vertices.begin(), vertices.end(),
460 compare_v_info<dimension>);
461 auto result_index = std::distance(vertices.begin(), min_element);
462 assert(result_index >= 0);
463 auto const index = static_cast<std::size_t>(std::abs(result_index));
464 return vertices[index]->info();
465 } // find_min_timevalue

749 template <int dimension>
750 [[nodiscard]] auto expected_cell_type(Cell_handle_t<dimension> const& t_cell)
751 {
752#ifndef NDEBUG
753 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
754#endif
755 std::array<int, static_cast<std::size_t>(dimension) + 1>
756 vertex_timevalues{};
757 // There are d+1 vertices in a d-dimensional simplex
758 for (auto i = 0; i < dimension + 1; ++i)
759 {
760 // Obtain timevalue of vertex
761 vertex_timevalues.at(static_cast<std::size_t>(i)) =
762 t_cell->vertex(i)->info();
763 }
764 auto const maxtime_ref =
765 std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
766 auto const mintime_ref =
767 std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
768 auto maxtime = *maxtime_ref;
769 auto mintime = *mintime_ref;
770 // A properly foliated simplex should have a timevalue difference of 1
771 if (maxtime - mintime != 1 || maxtime == mintime)
772 {
773#ifndef NDEBUG
774 spdlog::trace("This simplex is acausal:\n");
775 spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
776 mintime);
777 spdlog::trace("--\n");
778#endif
779 return Cell_type::ACAUSAL;
780 }
781 std::multiset<int> const timevalues{vertex_timevalues.begin(),
782 vertex_timevalues.end()};
783 auto max_vertices = timevalues.count(maxtime);
784 auto min_vertices = timevalues.count(mintime);
785
786 // 3D simplices
787 if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
788 if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
789 if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
790
791 // If we got here, there's some kind of error
792#ifndef NDEBUG
793 spdlog::trace("This simplex has an error:\n");
794 spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
795 mintime);
796 spdlog::trace(
797 "There are {} vertices with the max timevalue and {} vertices with "

965 template <int dimension, detail::ContainerType Container>
966 [[nodiscard]] auto collect_spacelike_facets(Container&& t_facets)
967 -> std::vector<std::pair<Int_precision, Facet_t<3>>>
968 {
969#ifndef NDEBUG
970 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
971#endif
972 using Volume_entry = std::pair<Int_precision, Facet_t<3>>;
973 std::vector<Volume_entry> space_faces;
974 if constexpr (std::ranges::sized_range<Container>)
975 {
976 space_faces.reserve(std::ranges::size(t_facets));
977 }
978 for (auto const& face : t_facets)
979 {
980 Cell_handle_t<dimension> const cell = face.first;
981 auto index_of_facet = face.second;
982#ifndef NDEBUG
983 spdlog::trace("Facet index is {}\n", index_of_facet);
984#endif
985 std::set<Int_precision> facet_timevalues;
986 // There are d+1 vertices in a d-dimensional simplex
987 for (int i = 0; i < dimension + 1; ++i)
988 {
989 if (i != index_of_facet)
990 {
991#ifndef NDEBUG
992 spdlog::trace("Vertex[{}] has timevalue {}\n", i,
993 cell->vertex(i)->info());
994#endif
995 facet_timevalues.insert(cell->vertex(i)->info());
996 }
997 }
998 // If we have a 1-element set then all timevalues on that facet are
999 // equal
1000 if (facet_timevalues.size() == 1)
1001 {
1002#ifndef NDEBUG
1003 spdlog::trace("Facet is spacelike on timevalue {}.\n",
1004 *facet_timevalues.begin());
1005#endif
1006 space_faces.emplace_back(*facet_timevalues.begin(), face);
1007 }
1008 else
1009 {
1010#ifndef NDEBUG
1011 spdlog::trace("Facet is timelike.\n");
1012#endif
1013 }
1014 }
1015 std::ranges::stable_sort(
1016 space_faces, std::ranges::less{},
1017 [](Volume_entry const& entry) noexcept { return entry.first; });
1018 return space_faces;
1019 } // collect_spacelike_facets

1071 template <int dimension>
1072 [[nodiscard]] auto find_bad_vertex(Cell_handle_t<dimension> const& cell)
1073 -> Vertex_handle_t<dimension>
1074 {
1075#ifndef NDEBUG
1076 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
1077 spdlog::debug("===Invalid Cell===\n");
1078 std::vector<Cell_handle_t<dimension>> bad_cell{cell};
1079 debug_print_cells<dimension>(std::span{bad_cell});
1080#endif
1081 std::multimap<Int_precision, Vertex_handle_t<dimension>> vertices;
1082 for (int i = 0; i < dimension + 1; ++i)
1083 {
1084 vertices.emplace(
1085 std::make_pair(cell->vertex(i)->info(), cell->vertex(i)));
1086 }
1087 // Now it's sorted in the multimap
1088 auto const minvalue = vertices.cbegin()->first;
1089 auto const maxvalue = vertices.crbegin()->first;
1090 auto const minvalue_count = vertices.count(minvalue);
1091 auto const maxvalue_count = vertices.count(maxvalue);
1092 // Return the vertex with the highest value if there are equal or more
1093 // vertices with lower values. Note that we preferentially return higher
1094 // timeslice vertices because there are typically more cells at higher
1095 // timeslices (see expected_points_per_timeslice())
1096 return minvalue_count >= maxvalue_count ? vertices.rbegin()->second
1097 : vertices.begin()->second;
1098 } // find_bad_vertex

1106 [[nodiscard]] auto fix_timevalues(Delaunay_t<dimension>& t_triangulation)
1107 -> bool
1108 {
1109 // Obtain a container of cells that are incorrectly foliated
1110 if (auto invalid_cells = check_timevalues<dimension>(t_triangulation);
1111 invalid_cells)
1112 {
1113 std::set<Vertex_handle_t<dimension>> vertices_to_remove;

1117 invalid_cells->begin(), invalid_cells->end(),
1118 std::inserter(vertices_to_remove, vertices_to_remove.begin()),
1119 find_bad_vertex<dimension>);
1120 // Remove the vertices
1121#ifndef NDEBUG
1122 spdlog::warn("There are {} invalid vertices.\n",
1123 vertices_to_remove.size());
1124#endif
1125 t_triangulation.remove(vertices_to_remove.begin(),
1126 vertices_to_remove.end());
1127 assert(t_triangulation.tds().is_valid());
1128 assert(t_triangulation.is_valid());
1129 return true;
1130 }
1131 return false;

1147 template <int dimension, std::uniform_random_bit_generator Generator>
1148 [[nodiscard]] auto make_foliated_ball(Int_precision const t_simplices,
1149 Int_precision const t_timeslices,
1150 double const initial_radius,
1151 double const foliation_spacing,
1152 Generator& generator)
1153 {
1154 if (t_simplices < 2 || t_timeslices < 2)
1155 {
1156 throw std::invalid_argument(
1157 "Simplices and timeslices must each be at least 2.");
1158 }
1159 if (!std::isfinite(initial_radius) || initial_radius <= 0.0)
1160 {
1161 throw std::invalid_argument(
1162 "Initial radius must be finite and positive.");
1163 }
1164 if (!std::isfinite(foliation_spacing) || foliation_spacing <= 0.0)
1165 {
1166 throw std::invalid_argument(
1167 "Foliation spacing must be finite and positive.");
1168 }
1169
1170 auto const population = utilities::generated_population_bounds(
1171 dimension, t_simplices, t_timeslices, initial_radius,
1172 foliation_spacing);
1173 if (population.points_per_timeslice < 2)
1174 {
1175 throw std::invalid_argument(
1176 "Simplices and timeslices would create an empty triangulation.");
1177 }
1178 if (!std::isfinite(population.last_layer_points) ||
1179 population.last_layer_points >
1180 static_cast<long double>(std::numeric_limits<Int_precision>::max()))
1181 {
1182 throw std::out_of_range(
1183 "Foliation parameters generate too many points per timeslice.");
1184 }
1185
1186 Causal_vertices_t<dimension> causal_vertices;
1187 causal_vertices.reserve(static_cast<std::size_t>(t_simplices));
1188 std::uniform_int_distribution<unsigned int> seed_distribution;
1189 CGAL::Random cgal_random{seed_distribution(generator)};
1190
1191 for (gsl::index i = 0; i < t_timeslices; ++i)
1192 {
1193 auto const radius =
1194 initial_radius + static_cast<double>(i) * foliation_spacing;
1195 auto const generated_points =
1196 static_cast<long double>(population.points_per_timeslice) * radius;
1197 if (!std::isfinite(radius) || generated_points < 2.0L)
1198 {
1199 throw std::invalid_argument(
1200 "Foliation parameters do not populate every timeslice.");
1201 }
1202 if (generated_points >
1203 static_cast<long double>(std::numeric_limits<Int_precision>::max()))
1204 {
1205 throw std::out_of_range(
1206 "Foliation parameters generate too many points per timeslice.");
1207 }
1208 Spherical_points_generator_t<dimension> gen{radius, cgal_random};
1209 // Generate random points at the radius
1210 for (gsl::index j = 0; j < static_cast<Int_precision>(generated_points);
1211 ++j)
1212 {
1213 causal_vertices.emplace_back(*gen++, i + 1);
1214 } // j
1215 } // i
1216 if (causal_vertices.size() < static_cast<std::size_t>(dimension + 1))
1217 {
1218 throw std::invalid_argument("Parameters create an empty triangulation.");
1219 }
1220 return causal_vertices;
1221 } // make_foliated_ball

1241 [[nodiscard]] auto make_triangulation(Int_precision const t_simplices,
1242 Int_precision t_timeslices,
1243 double const initial_radius,
1244 double const foliation_spacing,
1245 Generator& generator)
1246 -> Delaunay_t<dimension>
1247 {

1250#endif

1252 // Make initial triangulation
