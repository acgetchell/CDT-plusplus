105 template <int dimension>
106 auto make_causal_vertices(std::span<Point_t<dimension> const> vertices,
107 std::span<size_t const> timevalues)
108 -> Causal_vertices_t<dimension>
109 {
110 if (vertices.size() != timevalues.size())
111 {
112 throw std::length_error("Vertices and timevalues must be the same size.");
113 }
114 Causal_vertices_t<dimension> causal_vertices;
115 causal_vertices.reserve(vertices.size());
116 std::ranges::transform(vertices, timevalues,
117 std::back_inserter(causal_vertices),
118 [](Point_t<dimension> point, size_t time) {
119 return std::make_pair(point, time);
120 });
121 return causal_vertices;
122 }

130 template <int dimension>
131 [[nodiscard]] auto collect_edges(Delaunay_t<dimension> const& delaunay)
132 {
133 assert(delaunay.is_valid());
134 std::vector<Edge_handle_t<dimension>> init_edges;
135 init_edges.reserve(delaunay.number_of_finite_edges());
136 for (auto eit = delaunay.finite_edges_begin();
137 eit != delaunay.finite_edges_end(); ++eit)
138 {
139 Cell_handle_t<3> const cell = eit->first;
140 Edge_handle_t<3> thisEdge{cell, cell->index(cell->vertex(eit->second)),
141 cell->index(cell->vertex(eit->third))};

143 assert(delaunay.tds().is_valid(thisEdge.first, thisEdge.second,
144 thisEdge.third));
145 init_edges.emplace_back(thisEdge);
146 }
147 assert(init_edges.size() == delaunay.number_of_finite_edges());
148 return init_edges;
149 } // collect_edges

208 template <int dimension, detail::ContainerType Container>
209 [[nodiscard]] auto find_max_timevalue(Container&& t_vertices) -> Int_precision
210 {
211 auto vertices = std::forward<Container>(t_vertices);
212 if (vertices.empty())
213 {
214 throw std::invalid_argument("Cannot classify an empty triangulation.");
215 }
216 auto max_element = std::max_element(vertices.begin(), vertices.end(),
217 compare_v_info<dimension>);
218 auto result_index = std::distance(vertices.begin(), max_element);
219 // std::distance may be negative if random-access iterators are used and
220 // first is reachable from last
221 assert(result_index >= 0);
222 auto const index = static_cast<std::size_t>(std::abs(result_index));
223 return vertices[index]->info();
224 } // find_max_timevalue

527 template <int dimension>
528 [[nodiscard]] auto expected_cell_type(Cell_handle_t<dimension> const& t_cell)
529 {
530#ifndef NDEBUG
531 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
532#endif
533 std::array<int, static_cast<std::size_t>(dimension) + 1>
534 vertex_timevalues{};
535 // There are d+1 vertices in a d-dimensional simplex
536 for (auto i = 0; i < dimension + 1; ++i)
537 {
538 // Obtain timevalue of vertex
539 vertex_timevalues.at(static_cast<std::size_t>(i)) =
540 t_cell->vertex(i)->info();
541 }
542 auto const maxtime_ref =
543 std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
544 auto const mintime_ref =
545 std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
546 auto maxtime = *maxtime_ref;
547 auto mintime = *mintime_ref;
548 // A properly foliated simplex should have a timevalue difference of 1
549 if (maxtime - mintime != 1 || maxtime == mintime)
550 {
551#ifndef NDEBUG
552 spdlog::trace("This simplex is acausal:\n");
553 spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
554 mintime);
555 spdlog::trace("--\n");
556#endif
557 return Cell_type::ACAUSAL;
558 }
559 std::multiset<int> const timevalues{vertex_timevalues.begin(),
560 vertex_timevalues.end()};
561 auto max_vertices = timevalues.count(maxtime);
562 auto min_vertices = timevalues.count(mintime);
563
564 // 3D simplices
565 if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
566 if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
567 if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
568
569 // If we got here, there's some kind of error
570#ifndef NDEBUG
571 spdlog::trace("This simplex has an error:\n");
572 spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
573 mintime);
574 spdlog::trace(
575 "There are {} vertices with the max timevalue and {} vertices with "
576 "the min timevalue.\n",
577 max_vertices, min_vertices);
578 spdlog::trace("--\n");
579#endif
580 return Cell_type::UNCLASSIFIED;
581 } // expected_cell_type

743 template <int dimension, detail::ContainerType Container>
744 [[nodiscard]] auto collect_spacelike_facets(Container&& t_facets)
745 -> std::vector<std::pair<Int_precision, Facet_t<3>>>
746 {
747#ifndef NDEBUG
748 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
749#endif
750 using Volume_entry = std::pair<Int_precision, Facet_t<3>>;
751 std::vector<Volume_entry> space_faces;
752 if constexpr (std::ranges::sized_range<Container>)
753 {
754 space_faces.reserve(std::ranges::size(t_facets));
755 }
756 for (auto const& face : t_facets)
757 {
758 Cell_handle_t<dimension> const cell = face.first;
759 auto index_of_facet = face.second;
760#ifndef NDEBUG
761 spdlog::trace("Facet index is {}\n", index_of_facet);
762#endif
763 std::set<Int_precision> facet_timevalues;
764 // There are d+1 vertices in a d-dimensional simplex
765 for (int i = 0; i < dimension + 1; ++i)
766 {
767 if (i != index_of_facet)
768 {
769#ifndef NDEBUG
770 spdlog::trace("Vertex[{}] has timevalue {}\n", i,
771 cell->vertex(i)->info());
772#endif
773 facet_timevalues.insert(cell->vertex(i)->info());
774 }
775 }
776 // If we have a 1-element set then all timevalues on that facet are
777 // equal
778 if (facet_timevalues.size() == 1)
779 {
780#ifndef NDEBUG
781 spdlog::trace("Facet is spacelike on timevalue {}.\n",
782 *facet_timevalues.begin());
783#endif
784 space_faces.emplace_back(*facet_timevalues.begin(), face);
785 }
786 else
787 {
788#ifndef NDEBUG
789 spdlog::trace("Facet is timelike.\n");
790#endif
791 }
792 }
793 std::ranges::stable_sort(
794 space_faces, std::ranges::less{},
795 [](Volume_entry const& entry) noexcept { return entry.first; });
796 return space_faces;
797 } // collect_spacelike_facets

859 std::multimap<Int_precision, Vertex_handle_t<dimension>> vertices;
860 for (int i = 0; i < dimension + 1; ++i)
861 {
862 vertices.emplace(
863 std::make_pair(cell->vertex(i)->info(), cell->vertex(i)));
864 }
865 // Now it's sorted in the multimap
866 auto const minvalue = vertices.cbegin()->first;
867 auto const maxvalue = vertices.crbegin()->first;
868 auto const minvalue_count = vertices.count(minvalue);
869 auto const maxvalue_count = vertices.count(maxvalue);
870 // Return the vertex with the highest value if there are equal or more
871 // vertices with lower values. Note that we preferentially return higher
872 // timeslice vertices because there are typically more cells at higher
873 // timeslices (see expected_points_per_timeslice())
874 return minvalue_count >= maxvalue_count ? vertices.rbegin()->second
875 : vertices.begin()->second;
876 } // find_bad_vertex

883 template <int dimension>
884 [[nodiscard]] auto fix_timevalues(Delaunay_t<dimension>& t_triangulation)
885 -> bool
886 {
887 // Obtain a container of cells that are incorrectly foliated
888 if (auto invalid_cells = check_timevalues<dimension>(t_triangulation);
889 invalid_cells)
890 {
891 std::set<Vertex_handle_t<dimension>> vertices_to_remove;
892 // Transform the invalid cells into a set of vertices to remove
893 // Reduction to unique vertices happens via the set container
894 std::transform(
895 invalid_cells->begin(), invalid_cells->end(),
896 std::inserter(vertices_to_remove, vertices_to_remove.begin()),
897 find_bad_vertex<dimension>);
898 // Remove the vertices
899#ifndef NDEBUG
900 spdlog::warn("There are {} invalid vertices.\n",
901 vertices_to_remove.size());
902#endif
903 t_triangulation.remove(vertices_to_remove.begin(),
904 vertices_to_remove.end());
905 assert(t_triangulation.tds().is_valid());
906 assert(t_triangulation.is_valid());
907 return true;
908 }
909 return false;
910 } // fix_timevalues

925 template <int dimension, std::uniform_random_bit_generator Generator>
926 [[nodiscard]] auto make_foliated_ball(Int_precision const t_simplices,
927 Int_precision const t_timeslices,
928 double const initial_radius,
929 double const foliation_spacing,
930 Generator& generator)
931 {
932 if (t_simplices < 2 || t_timeslices < 2)
933 {
934 throw std::invalid_argument(
935 "Simplices and timeslices must each be at least 2.");
936 }
937 if (!std::isfinite(initial_radius) || initial_radius <= 0.0)
938 {
939 throw std::invalid_argument(
940 "Initial radius must be finite and positive.");
941 }
942 if (!std::isfinite(foliation_spacing) || foliation_spacing <= 0.0)
943 {
944 throw std::invalid_argument(
945 "Foliation spacing must be finite and positive.");
946 }
947
948 auto const population = utilities::generated_population_bounds(
949 dimension, t_simplices, t_timeslices, initial_radius,
950 foliation_spacing);
951 if (population.points_per_timeslice < 2)
952 {
953 throw std::invalid_argument(
954 "Simplices and timeslices would create an empty triangulation.");
955 }
956 if (!std::isfinite(population.last_layer_points) ||
957 population.last_layer_points >
958 static_cast<long double>(std::numeric_limits<Int_precision>::max()))
959 {
960 throw std::out_of_range(
961 "Foliation parameters generate too many points per timeslice.");
962 }
963
964 Causal_vertices_t<dimension> causal_vertices;
965 causal_vertices.reserve(static_cast<std::size_t>(t_simplices));
966 std::uniform_int_distribution<unsigned int> seed_distribution;
967 CGAL::Random cgal_random{seed_distribution(generator)};
968
969 for (gsl::index i = 0; i < t_timeslices; ++i)
970 {
971 auto const radius =
972 initial_radius + static_cast<double>(i) * foliation_spacing;
973 auto const generated_points =
974 static_cast<long double>(population.points_per_timeslice) * radius;
975 if (!std::isfinite(radius) || generated_points < 2.0L)
976 {
977 throw std::invalid_argument(
978 "Foliation parameters do not populate every timeslice.");
979 }
980 if (generated_points >
981 static_cast<long double>(std::numeric_limits<Int_precision>::max()))
982 {
983 throw std::out_of_range(
984 "Foliation parameters generate too many points per timeslice.");
985 }
986 Spherical_points_generator_t<dimension> gen{radius, cgal_random};
987 // Generate random points at the radius
988 for (gsl::index j = 0; j < static_cast<Int_precision>(generated_points);
989 ++j)
990 {
991 causal_vertices.emplace_back(*gen++, i + 1);
992 } // j
993 } // i
994 if (causal_vertices.size() < static_cast<std::size_t>(dimension + 1))
995 {
996 throw std::invalid_argument("Parameters create an empty triangulation.");
997 }
998 return causal_vertices;
999 } // make_foliated_ball

1013 [[nodiscard]] auto make_triangulation(Int_precision const t_simplices,
1014 Int_precision t_timeslices,
1015 double const initial_radius,
1016 double const foliation_spacing,
1017 Generator& generator)
1018 -> Delaunay_t<dimension>
1019 {
1020#ifndef NDEBUG
1021 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
1022#endif
1023 fmt::print("\nGenerating universe ...\n");
1024#ifdef CGAL_LINKED_WITH_TBB
1025 // Construct the locking data-structure
1026 // using the bounding-box of the points
1027 auto bounding_box_size = static_cast<double>(t_timeslices + 1);
1028 typename Delaunay_t<dimension>::Lock_data_structure locking_ds{
1029 CGAL::Bbox_3{-bounding_box_size, -bounding_box_size, -bounding_box_size,

1037#endif

1040 auto causal_vertices =

1050 {
1051 break;
1052 }
1053#ifndef NDEBUG
1054 spdlog::warn("Deleting incorrect vertices pass #{}\n", passes);
1055#endif
1056 }
1057
1058 // Fix timeslices
1059 for (auto passes = 1; passes < detail::MAX_FIX_PASSES + 1; ++passes)
1060 {
1061 if (!fix_timevalues<dimension>(triangulation)) { break; }
1062#ifndef NDEBUG
1063 spdlog::warn("Fixing timeslices pass #{}\n", passes);

1065 }
1066
1067 // Fix cells
1068 for (auto i = 1; i < detail::MAX_FIX_PASSES + 1; ++i)
1069 {
1070 if (!fix_cells<dimension>(triangulation)) { break; }
1071#ifndef NDEBUG
1072 spdlog::warn("Fixing incorrect cells pass #{}\n", i);
1073#endif
1074 }
1075
1076 utilities::print_delaunay(triangulation);
1077 assert(!check_timevalues<dimension>(triangulation));
1078 return triangulation;
