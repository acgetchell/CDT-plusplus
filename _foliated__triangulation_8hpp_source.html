680  [[nodiscard]] static auto expected_cell_type(Cell_handle_3 const& t_cell,
681  bool t_debug_flag = false)
682  {
683  std::vector<int> vertex_timevalues;
684  for (int i = 0; i < 4; ++i)
685  {
686  // Obtain timevalue of vertex
687  vertex_timevalues.emplace_back(t_cell->vertex(i)->info());
688  }
689  auto maxtime_ref =
690  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
691  auto mintime_ref =
692  std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
693  auto maxtime = *maxtime_ref;
694  auto mintime = *mintime_ref;
695  // A properly foliated simplex should have a timevalue difference of 1
696  if (maxtime - mintime != 1) { return Cell_type::ERROR; }
697  std::multiset<int> timevalues{vertex_timevalues.begin(),
698  vertex_timevalues.end()};
699  auto max_vertices = timevalues.count(maxtime);
700  auto min_vertices = timevalues.count(mintime);
701 
702  if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
703  if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
704  if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
705 
706  // If we got here, there's some kind of error
707  if (t_debug_flag)
708  {
709  fmt::print("This simplex has an error.\n");
710  fmt::print("Max timevalue is {}\n", maxtime);
711  fmt::print("There are {} vertices with the max timevalue.\n",
712  max_vertices);
713  fmt::print("Min timevalue is {}\n", mintime);
714  fmt::print("There are {} vertices with the min timevalue.\n",
715  min_vertices);
716  fmt::print("--\n");
717  }
718  return Cell_type::ERROR;
719  } // expected_cell_type
