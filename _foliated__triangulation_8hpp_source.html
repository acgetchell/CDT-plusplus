94 template <int dimension>
95 auto make_causal_vertices(std::span<Point_t<dimension> const> vertices,
96 std::span<size_t const> timevalues)
97 -> Causal_vertices_t<dimension>
98 {
99 if (vertices.size() != timevalues.size())
100 {
101 throw std::length_error("Vertices and timevalues must be the same size.");
102 }
103 Causal_vertices_t<dimension> causal_vertices;
104 causal_vertices.reserve(vertices.size());

107 [](Point_t<dimension> point, size_t time) {
108 return std::make_pair(point, time);
109 });
110 return causal_vertices;

120 [[nodiscard]] auto collect_edges(Delaunay_t<dimension> const& delaunay)
121 {
122 assert(delaunay.is_valid());
123 std::vector<Edge_handle_t<dimension>> init_edges;

125 for (auto eit = delaunay.finite_edges_begin();

127 {
128 Cell_handle_t<3> const cell = eit->first;
129 Edge_handle_t<3> thisEdge{cell, cell->index(cell->vertex(eit->second)),

133 thisEdge.third));
134 init_edges.emplace_back(thisEdge);
135 }
136 assert(init_edges.size() == delaunay.number_of_finite_edges());
137 return init_edges;

197 template <int dimension, ContainerType Container>
198 [[nodiscard]] auto find_max_timevalue(Container&& t_vertices) -> Int_precision
199 {
200 auto vertices = std::forward<Container>(t_vertices);
201 if (vertices.empty())
202 {
203 throw std::invalid_argument("Cannot classify an empty triangulation.");
204 }
205 auto max_element = std::max_element(vertices.begin(), vertices.end(),
206 compare_v_info<dimension>);
207 auto result_index = std::distance(vertices.begin(), max_element);
208 // std::distance may be negative if random-access iterators are used and
209 // first is reachable from last
210 assert(result_index >= 0);
211 auto const index = static_cast<std::size_t>(std::abs(result_index));
212 return vertices[index]->info();
213 } // find_max_timevalue

514 template <int dimension>
515 [[nodiscard]] auto expected_cell_type(Cell_handle_t<dimension> const& t_cell)
516 {
517#ifndef NDEBUG
518 spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
519#endif
520 std::array<int, static_cast<std::size_t>(dimension) + 1>
521 vertex_timevalues{};
522 // There are d+1 vertices in a d-dimensional simplex
523 for (auto i = 0; i < dimension + 1; ++i)
524 {
525 // Obtain timevalue of vertex
526 vertex_timevalues.at(static_cast<std::size_t>(i)) =
527 t_cell->vertex(i)->info();
528 }
529 auto const maxtime_ref =
530 std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
531 auto const mintime_ref =
532 std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
533 auto maxtime = *maxtime_ref;
534 auto mintime = *mintime_ref;
535 // A properly foliated simplex should have a timevalue difference of 1
536 if (maxtime - mintime != 1 || maxtime == mintime)
537 {
538#ifndef NDEBUG
539 spdlog::trace("This simplex is acausal:\n");
540 spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
541 mintime);
542 spdlog::trace("--\n");
543#endif
544 return Cell_type::ACAUSAL;
545 }
546 std::multiset<int> const timevalues{vertex_timevalues.begin(),
547 vertex_timevalues.end()};
548 auto max_vertices = timevalues.count(maxtime);
549 auto min_vertices = timevalues.count(mintime);
550
551 // 3D simplices
552 if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
553 if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
554 if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
555
556 // If we got here, there's some kind of error
557#ifndef NDEBUG
558 spdlog::trace("This simplex has an error:\n");
559 spdlog::trace("Max timevalue is {} and min timevalue is {}.\n", maxtime,
560 mintime);
561 spdlog::trace(
562 "There are {} vertices with the max timevalue and {} vertices with "
563 "the min timevalue.\n",
564 max_vertices, min_vertices);
565 spdlog::trace("--\n");
566#endif
567 return Cell_type::UNCLASSIFIED;
568 } // expected_cell_type

730 template <int dimension, ContainerType Container>
731 [[nodiscard]] auto collect_spacelike_facets(Container&& t_facets)
732 -> std::vector<std::pair<Int_precision, Facet_t<3>>>
733 {
734#ifndef NDEBUG
735 spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
736#endif
737 using Volume_entry = std::pair<Int_precision, Facet_t<3>>;
738 std::vector<Volume_entry> space_faces;
739 if constexpr (std::ranges::sized_range<Container>)
740 {
741 space_faces.reserve(std::ranges::size(t_facets));
742 }
743 for (auto const& face : t_facets)
744 {
745 Cell_handle_t<dimension> const cell = face.first;
746 auto index_of_facet = face.second;
747#ifndef NDEBUG
748 spdlog::trace("Facet index is {}\n", index_of_facet);
749#endif
750 std::set<Int_precision> facet_timevalues;
751 // There are d+1 vertices in a d-dimensional simplex
752 for (int i = 0; i < dimension + 1; ++i)
753 {
754 if (i != index_of_facet)
755 {
756#ifndef NDEBUG
757 spdlog::trace("Vertex[{}] has timevalue {}\n", i,
758 cell->vertex(i)->info());
759#endif
760 facet_timevalues.insert(cell->vertex(i)->info());
761 }
762 }
763 // If we have a 1-element set then all timevalues on that facet are
764 // equal
765 if (facet_timevalues.size() == 1)
766 {
767#ifndef NDEBUG
768 spdlog::trace("Facet is spacelike on timevalue {}.\n",
769 *facet_timevalues.begin());
770#endif
771 space_faces.emplace_back(*facet_timevalues.begin(), face);
772 }
773 else
774 {
775#ifndef NDEBUG
776 spdlog::trace("Facet is timelike.\n");
777#endif
778 }
779 }
780 std::ranges::stable_sort(
781 space_faces, std::ranges::less{},
782 [](Volume_entry const& entry) noexcept { return entry.first; });
783 return space_faces;
784 } // collect_spacelike_facets

837 [[nodiscard]] auto find_bad_vertex(Cell_handle_t<dimension> const& cell)
838 -> Vertex_handle_t<dimension>

840#ifndef NDEBUG

850 std::make_pair(cell->vertex(i)->info(), cell->vertex(i)));
851 }
852 // Now it's sorted in the multimap
853 auto const minvalue = vertices.cbegin()->first;
854 auto const maxvalue = vertices.crbegin()->first;
855 auto const minvalue_count = vertices.count(minvalue);
856 auto const maxvalue_count = vertices.count(maxvalue);
857 // Return the vertex with the highest value if there are equal or more
858 // vertices with lower values. Note that we preferentially return higher
859 // timeslice vertices because there are typically more cells at higher
860 // timeslices (see expected_points_per_timeslice())
861 return minvalue_count >= maxvalue_count ? vertices.rbegin()->second
862 : vertices.begin()->second;
863 } // find_bad_vertex

870 template <int dimension>
871 [[nodiscard]] auto fix_timevalues(Delaunay_t<dimension>& t_triangulation)
872 -> bool
873 {
874 // Obtain a container of cells that are incorrectly foliated
875 if (auto invalid_cells = check_timevalues<dimension>(t_triangulation);
876 invalid_cells)
877 {
878 std::set<Vertex_handle_t<dimension>> vertices_to_remove;
879 // Transform the invalid cells into a set of vertices to remove
880 // Reduction to unique vertices happens via the set container
881 std::transform(
882 invalid_cells->begin(), invalid_cells->end(),
883 std::inserter(vertices_to_remove, vertices_to_remove.begin()),
884 find_bad_vertex<dimension>);
885 // Remove the vertices
886#ifndef NDEBUG
887 spdlog::warn("There are {} invalid vertices.\n",
888 vertices_to_remove.size());
889#endif
890 t_triangulation.remove(vertices_to_remove.begin(),
891 vertices_to_remove.end());
892 assert(t_triangulation.tds().is_valid());
893 assert(t_triangulation.is_valid());
894 return true;
895 }
896 return false;
897 } // fix_timevalues

912 template <int dimension, std::uniform_random_bit_generator Generator>
913 [[nodiscard]] auto make_foliated_ball(Int_precision const t_simplices,
914 Int_precision const t_timeslices,
915 double const initial_radius,
916 double const foliation_spacing,
917 Generator& generator)
918 {
919 if (t_simplices < 2 || t_timeslices < 2)
920 {
921 throw std::invalid_argument(
922 "Simplices and timeslices must each be at least 2.");
923 }
924 if (!std::isfinite(initial_radius) || initial_radius <= 0.0)
925 {
926 throw std::invalid_argument(
927 "Initial radius must be finite and positive.");
928 }
929 if (!std::isfinite(foliation_spacing) || foliation_spacing <= 0.0)
930 {
931 throw std::invalid_argument(
932 "Foliation spacing must be finite and positive.");
933 }
934
935 auto const population = utilities::generated_population_bounds(
936 dimension, t_simplices, t_timeslices, initial_radius,
937 foliation_spacing);
938 if (population.points_per_timeslice < 2)
939 {
940 throw std::invalid_argument(
941 "Simplices and timeslices would create an empty triangulation.");
942 }
943 if (!std::isfinite(population.last_layer_points) ||
944 population.last_layer_points >
945 static_cast<long double>(std::numeric_limits<Int_precision>::max()))
946 {
947 throw std::out_of_range(
948 "Foliation parameters generate too many points per timeslice.");
949 }
950
951 Causal_vertices_t<dimension> causal_vertices;
952 causal_vertices.reserve(static_cast<std::size_t>(t_simplices));
953 std::uniform_int_distribution<unsigned int> seed_distribution;
954 CGAL::Random cgal_random{seed_distribution(generator)};
955
956 for (gsl::index i = 0; i < t_timeslices; ++i)
957 {
958 auto const radius =
959 initial_radius + static_cast<double>(i) * foliation_spacing;
960 auto const generated_points =
961 static_cast<long double>(population.points_per_timeslice) * radius;
962 if (!std::isfinite(radius) || generated_points < 2.0L)
963 {
964 throw std::invalid_argument(
965 "Foliation parameters do not populate every timeslice.");
966 }
967 if (generated_points >
968 static_cast<long double>(std::numeric_limits<Int_precision>::max()))
969 {
970 throw std::out_of_range(
971 "Foliation parameters generate too many points per timeslice.");
972 }
973 Spherical_points_generator_t<dimension> gen{radius, cgal_random};
974 // Generate random points at the radius
975 for (gsl::index j = 0; j < static_cast<Int_precision>(generated_points);
976 ++j)
977 {
978 causal_vertices.emplace_back(*gen++, i + 1);
979 } // j
980 } // i
981 if (causal_vertices.size() < static_cast<std::size_t>(dimension + 1))
982 {
983 throw std::invalid_argument("Parameters create an empty triangulation.");
984 }
985 return causal_vertices;
986 } // make_foliated_ball

999 template <int dimension, std::uniform_random_bit_generator Generator>
1000 [[nodiscard]] auto make_triangulation(Int_precision const t_simplices,
1001 Int_precision t_timeslices,
1002 double const initial_radius,
1003 double const foliation_spacing,
1004 Generator& generator)
1005 -> Delaunay_t<dimension>
1006 {
1007#ifndef NDEBUG
1008 spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
1009#endif
1010 fmt::print("\nGenerating universe ...\n");
1011#ifdef CGAL_LINKED_WITH_TBB
1012 // Construct the locking data-structure
1013 // using the bounding-box of the points
1014 auto bounding_box_size = static_cast<double>(t_timeslices + 1);
1015 typename Delaunay_t<dimension>::Lock_data_structure locking_ds{
1016 CGAL::Bbox_3{-bounding_box_size, -bounding_box_size, -bounding_box_size,
1017 bounding_box_size, bounding_box_size, bounding_box_size},
1018 50
1019 };
1020 Delaunay_t<dimension> triangulation =
1021 Delaunay_t<dimension>{TriangulationTraits<3>::Kernel{}, &locking_ds};
1022#else
1023 Delaunay_t<dimension> triangulation = Delaunay_t<dimension>{};
1024#endif
1025
1026 // Make initial triangulation
1027 auto causal_vertices =
1028 make_foliated_ball<dimension>(t_simplices, t_timeslices, initial_radius,
1029 foliation_spacing, generator);
1030 triangulation.insert(causal_vertices.begin(), causal_vertices.end());
1031
1032 // Fix vertices
1033 for (auto passes = 1; passes < MAX_FIX_PASSES + 1; ++passes)
1034 {
1035 if (!fix_vertices<dimension>(triangulation, initial_radius,
1036 foliation_spacing))
1037 {
1038 break;
1039 }
1040#ifndef NDEBUG
1041 spdlog::warn("Deleting incorrect vertices pass #{}\n", passes);
1042#endif
1043 }
1044
1045 // Fix timeslices
1046 for (auto passes = 1; passes < MAX_FIX_PASSES + 1; ++passes)
1047 {
1048 if (!fix_timevalues<dimension>(triangulation)) { break; }
1049#ifndef NDEBUG
1050 spdlog::warn("Fixing timeslices pass #{}\n", passes);
1051#endif
1052 }
1053
1054 // Fix cells
1055 for (auto i = 1; i < MAX_FIX_PASSES + 1; ++i)
1056 {
1057 if (!fix_cells<dimension>(triangulation)) { break; }
1058#ifndef NDEBUG
1059 spdlog::warn("Fixing incorrect cells pass #{}\n", i);
1060#endif
1061 }
1062
1063 utilities::print_delaunay(triangulation);
1064 assert(!check_timevalues<dimension>(triangulation));
1065 return triangulation;
1066 } // make_triangulation
