83  template <int dimension>
84  [[nodiscard]] inline auto find_max_timevalue(
85  std::vector<Vertex_handle_t<dimension>> const& t_vertices)
86  -> Int_precision
87  {
88  Expects(!t_vertices.empty());
89  auto it = std::max_element(t_vertices.begin(), t_vertices.end(),
90  compare_v_info<dimension>);
91  auto result_index = std::distance(t_vertices.begin(), it);
92  // std::distance may be negative if random-access iterators are used and
93  // first is reachable from last
94  Ensures(result_index >= 0);
95  auto index = static_cast<std::size_t>(std::abs(result_index));
96  return t_vertices[index]->info();
97  } // find_max_timevalue

191  template <int dimension>
192  [[nodiscard]] inline auto expected_cell_type(
193  Cell_handle_t<dimension> const& t_cell, bool t_debug_flag = false)
194  {
195  std::vector<int> vertex_timevalues;
196  for (auto i = 0; i < dimension + 1; ++i)
197  {
198  // Obtain timevalue of vertex
199  vertex_timevalues.emplace_back(t_cell->vertex(i)->info());
200  }
201  auto maxtime_ref =
202  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
203  auto mintime_ref =
204  std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
205  auto maxtime = *maxtime_ref;
206  auto mintime = *mintime_ref;
207  // A properly foliated simplex should have a timevalue difference of 1
208  if (maxtime - mintime != 1) { return Cell_type::ERROR; }
209  std::multiset<int> timevalues{vertex_timevalues.begin(),
210  vertex_timevalues.end()};
211  auto max_vertices = timevalues.count(maxtime);
212  auto min_vertices = timevalues.count(mintime);
213 
214  // 3D simplices
215  if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
216  if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
217  if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
218 
219  // 4D simplices
220  if (max_vertices == 4 && min_vertices == 1) { return Cell_type::ONE_FOUR; }
221  if (max_vertices == 3 && min_vertices == 2) { return Cell_type::TWO_THREE; }
222  if (max_vertices == 2 && min_vertices == 3) { return Cell_type::THREE_TWO; }
223  if (max_vertices == 1 && min_vertices == 4) { return Cell_type::FOUR_ONE; }
224 
225  // If we got here, there's some kind of error
226  if (t_debug_flag)
227  {
228  fmt::print("This simplex has an error.\n");
229  fmt::print("Max timevalue is {}\n", maxtime);
230  fmt::print("There are {} vertices with the max timevalue.\n",
231  max_vertices);
232  fmt::print("Min timevalue is {}\n", mintime);
233  fmt::print("There are {} vertices with the min timevalue.\n",
234  min_vertices);
235  fmt::print("--\n");
236  }
237  return Cell_type::ERROR;
238  } // expected_cell_type

305  template <int dimension>
306  [[nodiscard]] inline auto make_foliated_sphere(
307  Int_precision const t_simplices, Int_precision const t_timeslices,
308  double const initial_radius = INITIAL_RADIUS,
309  double const foliation_spacing = FOLIATION_SPACING)
310  {
311  Causal_vertices_t<dimension> causal_vertices;
312  causal_vertices.reserve(static_cast<std::size_t>(t_simplices));
313  const auto points_per_timeslice =
314  expected_points_per_timeslice(dimension, t_simplices, t_timeslices);
315  Expects(points_per_timeslice >= 2);
316 
317  for (gsl::index i = 0; i < t_timeslices; ++i)
318  {
319  auto radius = initial_radius + static_cast<double>(i) * foliation_spacing;
320  typename triangulation_traits<dimension>::Spherical_points_generator gen{
321  static_cast<double>(radius)};
322  // Generate random points at the radius
323  for (gsl::index j = 0;
324  j < static_cast<Int_precision>(points_per_timeslice * radius); ++j)
325  {
326  causal_vertices.emplace_back(std::make_pair(*gen++, i + 1));
327  } // j
328  } // i
329  return causal_vertices;
330  } // make_foliated_sphere

400  template <int dimension>
401  [[nodiscard]] auto check_timeslices(Delaunay_t<3> const& t_triangulation)
402  -> std::optional<std::vector<Vertex_handle_t<3>>>
403  {
404  std::vector<Vertex_handle_t<3>> invalid_vertices;
405 
406  // Iterate over all cells in the triangulation
407  for (Delaunay_t<3>::Finite_cells_iterator cit =
408  t_triangulation.finite_cells_begin();
409  cit != t_triangulation.finite_cells_end(); ++cit)
410  {
411  Ensures(cit->is_valid());
412  std::multimap<int, Vertex_handle_t<3>> this_cell;
413  // Collect a map of timevalues and vertices in each cell
414  for (int i = 0; i < 4; ++i)
415  {
416  this_cell.emplace(
417  std::make_pair(cit->vertex(i)->info(), cit->vertex(i)));
418  }
419  // Now it's sorted in the multimap
420  auto minvalue = this_cell.cbegin()->first;
421  auto maxvalue = this_cell.crbegin()->first;
422 
423 #ifdef DETAILED_DEBUGGING
424  auto min_vertex = this_cell.cbegin()->second;
425  auto max_vertex = this_cell.crbegin()->second;
426  fmt::print("Smallest timevalue in this cell is: {}\n", minvalue);
427  fmt::print("Largest timevalue in this cell is: {}\n", maxvalue);
428  fmt::print("Min vertex info() {}\n", min_vertex->info());
429  fmt::print("Max vertex info() {}\n", max_vertex->info());
430 #endif
431  // There must be a timevalue delta of 1 for a validly foliated simplex
432  if (maxvalue - minvalue == 1)
433  {
434 #ifdef DETAILED_DEBUGGING
435  fmt::print("This cell is valid.\n");
436 #endif
437  }
438  else
439  {
440  auto minvalue_count = this_cell.count(minvalue);
441  auto maxvalue_count = this_cell.count(maxvalue);
442 #ifdef DETAILED_DEBUGGING
443  fmt::print("This cell is invalid.\n");
444  fmt::print("There are {} vertices with the minvalue.\n",
445  minvalue_count);
446  fmt::print("There are {} vertices with the maxvalue.\n",
447  maxvalue_count);
448  fmt::print("So we should remove ");
449 #endif
450 
451  if (minvalue_count > maxvalue_count)
452  {
453  invalid_vertices.emplace_back(this_cell.rbegin()->second);
454 #ifdef DETAILED_DEBUGGING
455  fmt::print("maxvalue.\n");
456 #endif
457  }
458  else
459  {
460  invalid_vertices.emplace_back(this_cell.begin()->second);
461 #ifdef DETAILED_DEBUGGING
462  fmt::print("minvalue.\n");
463 #endif
464  }
465  }
466  }
467  if (invalid_vertices.empty()) { return std::nullopt; }
468 
469 #ifdef DETAILED_DEBUGGING
470  fmt::print("Removing ...\n");
471  for (auto& v : invalid_vertices)
472  {
473  fmt::print("Vertex {} with timevalue {}\n", v->point(), v->info());
474  }
475 #endif
476  return invalid_vertices;
477 
478  } // check_timeslices
