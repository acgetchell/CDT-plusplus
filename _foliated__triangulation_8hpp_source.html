90  template <typename VertexType>
91  [[nodiscard]] inline auto find_max_timevalue(
92  std::vector<VertexType> const& t_vertices) -> Int_precision
93  {
94  Expects(!t_vertices.empty());
95  auto it = std::max_element(t_vertices.begin(), t_vertices.end(),
96  compare_v_info<VertexType>);
97  auto result_index = std::distance(t_vertices.begin(), it);
98  // std::distance may be negative if random-access iterators are used and
99  // first is reachable from last
100  Ensures(result_index >= 0);
101  auto index = static_cast<std::size_t>(std::abs(result_index));
102  return t_vertices[index]->info();
103  } // find_max_timevalue

201  template <typename CellType, int dimension>
202  [[nodiscard]] inline auto expected_cell_type(CellType const& t_cell,
203  bool t_debug_flag = false)
204  {
205  std::vector<int> vertex_timevalues;
206  for (int i = 0; i < dimension + 1; ++i)
207  {
208  // Obtain timevalue of vertex
209  vertex_timevalues.emplace_back(t_cell->vertex(i)->info());
210  }
211  auto maxtime_ref =
212  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
213  auto mintime_ref =
214  std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
215  auto maxtime = *maxtime_ref;
216  auto mintime = *mintime_ref;
217  // A properly foliated simplex should have a timevalue difference of 1
218  if (maxtime - mintime != 1) { return Cell_type::ERROR; }
219  std::multiset<int> timevalues{vertex_timevalues.begin(),
220  vertex_timevalues.end()};
221  auto max_vertices = timevalues.count(maxtime);
222  auto min_vertices = timevalues.count(mintime);
223 
224  // 3D simplices
225  if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
226  if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
227  if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
228 
229  // 4D simplices
230  if (max_vertices == 4 && min_vertices == 1) { return Cell_type::ONE_FOUR; }
231  if (max_vertices == 3 && min_vertices == 2) { return Cell_type::TWO_THREE; }
232  if (max_vertices == 2 && min_vertices == 3) { return Cell_type::THREE_TWO; }
233  if (max_vertices == 1 && min_vertices == 4) { return Cell_type::FOUR_ONE; }
234 
235  // If we got here, there's some kind of error
236  if (t_debug_flag)
237  {
238  fmt::print("This simplex has an error.\n");
239  fmt::print("Max timevalue is {}\n", maxtime);
240  fmt::print("There are {} vertices with the max timevalue.\n",
241  max_vertices);
242  fmt::print("Min timevalue is {}\n", mintime);
243  fmt::print("There are {} vertices with the min timevalue.\n",
244  min_vertices);
245  fmt::print("--\n");
246  }
247  return Cell_type::ERROR;
248  } // expected_cell_type

315  [[nodiscard]] inline auto make_foliated_sphere(
316  Int_precision const t_simplices, Int_precision const t_timeslices,
317  double const initial_radius = INITIAL_RADIUS,
318  double const foliation_spacing = FOLIATION_SPACING) -> Causal_vertices
319  {
320  Causal_vertices causal_vertices;
321  causal_vertices.reserve(static_cast<std::size_t>(t_simplices));
322  const auto points_per_timeslice =
323  expected_points_per_timeslice(3, t_simplices, t_timeslices);
324  Expects(points_per_timeslice >= 2);
325 
326  using Spherical_points_generator = CGAL::Random_points_on_sphere_3<Point_3>;
327  for (gsl::index i = 0; i < t_timeslices; ++i)
328  {
329  auto radius = initial_radius + static_cast<double>(i) * foliation_spacing;
330  Spherical_points_generator gen{static_cast<double>(radius)};
331  // Generate random points at the radius
332  for (gsl::index j = 0;
333  j < static_cast<Int_precision>(points_per_timeslice * radius); ++j)
334  {
335  causal_vertices.emplace_back(std::make_pair(*gen++, i + 1));
336  } // j
337  } // i
338  return causal_vertices;
339  } // make_foliated_sphere

347  [[nodiscard]] inline auto volume_per_timeslice(
348  std::vector<Face_handle_3> const& t_facets, bool t_debug_flag = false)
349  -> std::multimap<Int_precision, Facet_3>
350  {
351  std::multimap<Int_precision, Facet_3> space_faces;
352  for (auto const& face : t_facets)
353  {
354  Cell_handle_3 ch = face.first;
355  auto index_of_facet = face.second;
356  if (t_debug_flag) { fmt::print("Facet index is {}\n", index_of_facet); }
357  std::set<Int_precision> facet_timevalues;
358  for (int i = 0; i < 4; ++i)
359  {
360  if (i != index_of_facet)
361  {
362  if (t_debug_flag)
363  {
364  fmt::print("Vertex[{}] has timevalue {}\n", i,
365  ch->vertex(i)->info());
366  }
367  facet_timevalues.insert(ch->vertex(i)->info());
368  }
369  }
370  // If we have a 1-element set then all timevalues on that facet are
371  // equal
372  if (facet_timevalues.size() == 1)
373  {
374  if (t_debug_flag)
375  {
376  fmt::print("Facet is spacelike on timevalue {}.\n",
377  *facet_timevalues.begin());
378  }
379  space_faces.insert({*facet_timevalues.begin(), face});
380  }
381  else
382  {
383  if (t_debug_flag) { fmt::print("Facet is timelike.\n"); }
384  }
385  }
386  return space_faces;

401  template <typename Triangulation>
402  [[nodiscard]] auto check_timeslices(Triangulation&& t_triangulation)
403  -> std::optional<std::vector<Vertex_handle_3>>
404  {
405  std::vector<Vertex_handle_3> invalid_vertices;
406  // Iterate over all cells in the triangulation
407  for (Delaunay3::Finite_cells_iterator cit =
408  t_triangulation.finite_cells_begin();
409  cit != t_triangulation.finite_cells_end(); ++cit)
410  {
411  Ensures(cit->is_valid());
412  std::multimap<int, Vertex_handle_3> this_cell;
413  // Collect a map of timevalues and vertices in each cell
414  for (int i = 0; i < 4; ++i)
415  {
416  this_cell.emplace(
417  std::make_pair(cit->vertex(i)->info(), cit->vertex(i)));
418  }
419  // Now it's sorted in the multimap
420  auto minvalue = this_cell.cbegin()->first;
421  auto maxvalue = this_cell.crbegin()->first;
422 
423 #ifdef DETAILED_DEBUGGING
424  auto min_vertex = this_cell.cbegin()->second;
425  auto max_vertex = this_cell.crbegin()->second;
426  fmt::print("Smallest timevalue in this cell is: {}\n", minvalue);
427  fmt::print("Largest timevalue in this cell is: {}\n", maxvalue);
428  fmt::print("Min vertex info() {}\n", min_vertex->info());
429  fmt::print("Max vertex info() {}\n", max_vertex->info());
430 #endif
431  // There must be a timevalue delta of 1 for a validly foliated simplex
432  if (maxvalue - minvalue == 1)
433  {
434 #ifdef DETAILED_DEBUGGING
435  fmt::print("This cell is valid.\n");
436 #endif
437  }
438  else
439  {
440  auto minvalue_count = this_cell.count(minvalue);
441  auto maxvalue_count = this_cell.count(maxvalue);
442 #ifdef DETAILED_DEBUGGING
443  fmt::print("This cell is invalid.\n");
444  fmt::print("There are {} vertices with the minvalue.\n",
445  minvalue_count);
446  fmt::print("There are {} vertices with the maxvalue.\n",
447  maxvalue_count);
448  fmt::print("So we should remove ");
449 #endif
450 
451  if (minvalue_count > maxvalue_count)
452  {
453  invalid_vertices.emplace_back(this_cell.rbegin()->second);
454 #ifdef DETAILED_DEBUGGING
455  fmt::print("maxvalue.\n");
456 #endif
457  }
458  else
459  {
460  invalid_vertices.emplace_back(this_cell.begin()->second);
461 #ifdef DETAILED_DEBUGGING
462  fmt::print("minvalue.\n");
463 #endif
464  }
465  }
466  }
467  if (invalid_vertices.empty()) { return std::nullopt; }
468 
469 #ifdef DETAILED_DEBUGGING
470  fmt::print("Removing ...\n");
471  for (auto& v : invalid_vertices)
472  {
473  fmt::print("Vertex {} with timevalue {}\n", v->point(), v->info());
474  }
475 #endif
476  return invalid_vertices;
477 
478  } // check_timeslices
