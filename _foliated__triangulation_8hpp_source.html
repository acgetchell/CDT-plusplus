710  template <typename Triangulation>
711  [[nodiscard]] auto check_timeslices(Triangulation&& t_triangulation) const
712  -> std::optional<std::vector<Vertex_handle>>
713  {
714  std::vector<Vertex_handle> invalid_vertices;
715  // Iterate over all cells in the triangulation
716  for (Delaunay3::Finite_cells_iterator cit =
717  t_triangulation.finite_cells_begin();
718  cit != t_triangulation.finite_cells_end(); ++cit)
719  {
720  Ensures(cit->is_valid());
721  std::multimap<int, Vertex_handle> this_cell;
722  // Collect a map of timevalues and vertices in each cell
723  for (int i = 0; i < 4; ++i)
724  {
725  this_cell.emplace(
726  std::make_pair(cit->vertex(i)->info(), cit->vertex(i)));
727  }
728  // Now it's sorted in the multimap
729  auto minvalue = this_cell.cbegin()->first;
730  auto maxvalue = this_cell.crbegin()->first;
731 
732 #ifdef DETAILED_DEBUGGING
733  auto min_vertex = this_cell.cbegin()->second;
734  auto max_vertex = this_cell.crbegin()->second;
735  fmt::print("Smallest timevalue in this cell is: {}\n", minvalue);
736  fmt::print("Largest timevalue in this cell is: {}\n", maxvalue);
737  fmt::print("Min vertex info() {}\n", min_vertex->info());
738  fmt::print("Max vertex info() {}\n", max_vertex->info());
739 #endif
740  // There must be a timevalue delta of 1 for a validly foliated simplex
741  if (maxvalue - minvalue == 1)
742  {
743 #ifdef DETAILED_DEBUGGING
744  fmt::print("This cell is valid.\n");
745 #endif
746  }
747  else
748  {
749  auto minvalue_count = this_cell.count(minvalue);
750  auto maxvalue_count = this_cell.count(maxvalue);
751 #ifdef DETAILED_DEBUGGING
752  fmt::print("This cell is invalid.\n");
753  fmt::print("There are {} vertices with the minvalue.\n",
754  minvalue_count);
755  fmt::print("There are {} vertices with the maxvalue.\n",
756  maxvalue_count);
757  fmt::print("So we should remove ");
758 #endif
759 
760  if (minvalue_count > maxvalue_count)
761  {
762  invalid_vertices.emplace_back(this_cell.rbegin()->second);
763 #ifdef DETAILED_DEBUGGING
764  fmt::print("maxvalue.\n");
765 #endif
766  }

933  [[nodiscard]] auto classify_cells(std::vector<Cell_handle> const& cells,
934  bool t_debug_flag = false) const
935  -> std::vector<Cell_handle>
936  {
937  Expects(cells.size() == number_of_finite_cells());
938  std::vector<Vertex_handle> cell_vertices;
939  cell_vertices.reserve(4);
940  std::vector<int> vertex_timevalues;
941  vertex_timevalues.reserve(4);
942  for (auto const& c : cells)
943  {
944  if (t_debug_flag) { fmt::print("Cell info was {}\n", c->info()); }
945 
946  for (int j = 0; j < 4; ++j)
947  {
948  cell_vertices.emplace_back(c->vertex(j));
949  vertex_timevalues.emplace_back(c->vertex(j)->info());
950  if (t_debug_flag)
951  {
952  fmt::print("Cell vertex {} has timevalue {}\n", j,
953  c->vertex(j)->info());
954  }
955  }
956 
957  // This is simply not sufficient. Need to check *both* maxtime and
958  // min_time, and that there are exactly 1 and 3, 2 and 2, or 3 and 1.
959  // Anything else means we have an invalid simplex which we should
960  // also return.
961  // We also need to check that maxtime - min_time = 1, else we have
962  // a mis-classified vertex (probably)
963  auto maxtime =
964  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
965  auto maxtime_vertices = std::count_if(
966  cell_vertices.begin(), cell_vertices.end(),
967  [maxtime](auto const& vertex) { return vertex->info() == *maxtime; });
968  // Check maxtime - min_time here
969  switch (maxtime_vertices)
970  {
971  case 1:
972  c->info() = static_cast<int>(Cell_type::THREE_ONE);
973  break;
974  case 2:
975  c->info() = static_cast<int>(Cell_type::TWO_TWO);
976  break;
977  case 3:
978  c->info() = static_cast<int>(Cell_type::ONE_THREE);
979  break;
980  default:
981  throw std::logic_error("Mis-classified cell.");
982  }
983  if (t_debug_flag)
984  {
985  fmt::print("Max timevalue is {}\n", *maxtime);
986  fmt::print("There are {} vertices with max timeslice in the cell.\n",
987  maxtime_vertices);
988  fmt::print("Cell info is now {}\n", c->info());
989  fmt::print("--\n");
990  }
991  cell_vertices.clear();
992  vertex_timevalues.clear();
993  }
994  return cells;
995  } // classify_cells
