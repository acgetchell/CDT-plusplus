666  [[nodiscard]] static auto expected_cell_type(Cell_handle const& t_cell,
667  bool t_debug_flag = false)
668  {
669  std::vector<int> vertex_timevalues;
670  for (int i = 0; i < 4; ++i)
671  {
672  // Obtain timevalue of vertex
673  vertex_timevalues.emplace_back(t_cell->vertex(i)->info());
674  }
675  auto maxtime_ref =
676  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
677  auto mintime_ref =
678  std::min_element(vertex_timevalues.begin(), vertex_timevalues.end());
679  auto maxtime = *maxtime_ref;
680  auto mintime = *mintime_ref;
681  // A properly foliated simplex should have a timevalue difference of 1
682  if (maxtime - mintime != 1) { return Cell_type::ERROR; }
683  std::multiset<int> timevalues{vertex_timevalues.begin(),
684  vertex_timevalues.end()};
685  auto max_vertices = timevalues.count(maxtime);
686  auto min_vertices = timevalues.count(mintime);
687 
688  if (max_vertices == 3 && min_vertices == 1) { return Cell_type::ONE_THREE; }
689  if (max_vertices == 2 && min_vertices == 2) { return Cell_type::TWO_TWO; }
690  if (max_vertices == 1 && min_vertices == 3) { return Cell_type::THREE_ONE; }
691 
692  // If we got here, there's some kind of error
693  if (t_debug_flag)
694  {
695  fmt::print("This simplex has an error.\n");
696  fmt::print("Max timevalue is {}\n", maxtime);
697  fmt::print("There are {} vertices with the max timevalue.\n",
698  max_vertices);
699  fmt::print("Min timevalue is {}\n", mintime);
700  fmt::print("There are {} vertices with the min timevalue.\n",
701  min_vertices);
702  fmt::print("--\n");
703  }
704  return Cell_type::ERROR;
705  } // expected_cell_type
