88 [[nodiscard]] inline auto parse_move_run_cadence(
89 Int_precision const passes, Int_precision const checkpoint,
90 std::string_view const strategy_name) -> MoveRunCadence
91 {
92 auto cadence = MoveRunCadence::parse(passes, checkpoint);
93 if (cadence) { return *cadence; }
94
95 switch (cadence.error())
96 {
97 case MoveRunCadenceError::NONPOSITIVE_PASSES:
98 throw std::invalid_argument{
99 fmt::format("{} passes must be positive", strategy_name)};
100 case MoveRunCadenceError::NONPOSITIVE_CHECKPOINT:
101 throw std::invalid_argument{fmt::format(
102 "{} checkpoint interval must be positive", strategy_name)};
103 }
104 throw std::logic_error{"Unknown move-run cadence error"};
105 }

177 template <typename ManifoldType, typename StrategyState,
178 typename ExecutePass, typename Report, typename Checkpoint>
179 [[nodiscard]] auto execute_move_run(ManifoldType initial,
180 StrategyState initial_strategy_state,
181 MoveRunCadence const cadence,
182 MoveRunIdentity const identity,
183 bool const writes_files,
184 ExecutePass execute_pass, Report report,
185 Checkpoint checkpoint)
186 -> MoveRunResult<ManifoldType, StrategyState>
187 {
188 auto current = std::move(initial);
189 auto command_totals = MoveCommandResults<ManifoldType>{};
190 auto strategy_state = std::move(initial_strategy_state);
191 auto checkpoint_events = Int_precision{};
192
193 fmt::print("Starting {} algorithm in {}+1 dimensions ...\n",
194 identity.algorithm, ManifoldType::dimension - 1);
195 fmt::print("Effective random seed: {} (stream {}).\n", identity.seed,
196 identity.stream);
197 fmt::print("Making random moves ...\n");
198
199 for (auto pass_index = Int_precision{}; pass_index < cadence.passes();
200 ++pass_index)
201 {
202 auto const pass_number = pass_index + 1;
203 fmt::print("=== Pass {} ===\n", pass_number);
204 auto const attempts = current.N3();
205 auto pass = std::invoke(execute_pass, std::move(current),
206 std::move(strategy_state), attempts);
207 current = std::move(pass.manifold);
208 command_totals = accumulate_command_results(std::move(command_totals),
209 pass.command_results);
210 strategy_state = std::move(pass.strategy_state);
211
212 if (pass_number % cadence.checkpoint() == 0)
213 {
214 ++checkpoint_events;
215 std::invoke(report, std::as_const(current),
216 std::as_const(command_totals),
217 std::as_const(strategy_state));
218 if (writes_files)
219 {
220 fmt::print("Writing checkpoint for pass {}.\n", pass_number);
221 std::invoke(checkpoint, std::as_const(current),
222 std::as_const(command_totals),
223 std::as_const(strategy_state), pass_number);
224 }
225 }
226 }
227
228 fmt::print("=== Run results ===\n");
229 std::invoke(report, std::as_const(current), std::as_const(command_totals),
230 std::as_const(strategy_state));
231 return {.manifold = std::move(current),
232 .command_results = std::move(command_totals),
233 .strategy_state = std::move(strategy_state),
234 .checkpoint_events = checkpoint_events};
235 }
