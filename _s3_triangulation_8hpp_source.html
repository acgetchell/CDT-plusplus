129  {
130  if (cit->is_valid())
131  { // Valid cell
132  min_time = cit->vertex(0)->info();
133  max_time = min_time;
134 #ifdef DETAILED_DEBUGGING
135  bool this_cell_foliation_valid = true;
136 #endif
137  // Iterate over all vertices in the cell
138  for (int i = 0; i < 4; ++i)
139  {
140  auto current_time = cit->vertex(i)->info();
141 
142  // Classify extreme values
143  if (current_time < min_time) min_time = current_time;
144  if (current_time > max_time)
145  {
146  max_time = current_time;
147  max_vertex = i;
148  }
149  } // Finish iterating over vertices

162  }
163  else

165  ++valid;
166  }

168 #ifdef DETAILED_DEBUGGING
