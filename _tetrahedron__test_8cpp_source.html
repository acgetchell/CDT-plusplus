85SCENARIO("Find distances between points of the tetrahedron" *
86 doctest::test_suite("tetrahedron"))
87{
88 using Point = Point_t<3>;
89 using Causal_vertices = Causal_vertices_t<3>;
90 using FoliatedTriangulation = FoliatedTriangulation_3;
91 using squared_distance = TriangulationTraits<3>::squared_distance;
92 GIVEN("Points in a tetrahedron.")
93 {
94 auto origin = Point{0, 0, 0};
95 // These points have a radius of 1
96 auto v_1 = Point{1, 0, 0};
97 auto v_2 = Point{0, 1, 0};
98 auto v_3 = Point{0, 0, 1};
99 auto v_4 = Point{RADIUS_2, RADIUS_2, RADIUS_2};
100 Causal_vertices causal_vertices;
101 causal_vertices.emplace_back(v_1, 1);
102 causal_vertices.emplace_back(v_2, 1);
103 causal_vertices.emplace_back(v_3, 1);
104 causal_vertices.emplace_back(v_4, 2);
105 WHEN("The Foliated triangulation is constructed with these points.")
106 {
107 FoliatedTriangulation triangulation(causal_vertices);
108 squared_distance constexpr r_2;
109 THEN("The triangulation is initialized correctly.")
110 {
111 REQUIRE(triangulation.is_initialized());
112 }
113 THEN("The squared distances of vertices from origin are correct.")
114 {
115 fmt::print("v_1 is {}\n", utilities::point_to_str(v_1));
116 fmt::print("v_2 is {}\n", utilities::point_to_str(v_2));
117 fmt::print("v_3 is {}\n", utilities::point_to_str(v_3));
118 fmt::print("v_4 is {}\n", utilities::point_to_str(v_4));
119
120 auto d_1 = r_2(origin, v_1);
121 fmt::print("The squared distance between v_1 and the origin is {}\n",
122 d_1);
123 CHECK_EQ(d_1, doctest::Approx(1.0));
124
125 auto d_2 = r_2(origin, v_2);
126 fmt::print("The squared distance between v_2 and the origin is {}\n",
127 d_2);
128 CHECK_EQ(d_2, doctest::Approx(1.0));
129
130 auto d_3 = r_2(origin, v_3);
131 fmt::print("The squared distance between v_3 and the origin is {}\n",
132 d_3);
133 CHECK_EQ(d_3, doctest::Approx(1.0));
134
135 auto d_4 = r_2(origin, v_4);
136 fmt::print("The squared distance between v_4 and the origin is {}\n",
137 d_4);
138 CHECK_EQ(d_4, doctest::Approx(4.0));
139 }
140 THEN("The squared distance between radius=1 vertices are 2.")
141 {
142 auto d_1 = r_2(v_1, v_2);
143 CHECK_EQ(d_1, doctest::Approx(2.0));
144 fmt::print("The squared distance between v_1 and v_2 is {}\n", d_1);
145 auto d_2 = r_2(v_1, v_3);
146 CHECK_EQ(d_2, doctest::Approx(2.0));
147 fmt::print("The squared distance between v_1 and v_3 is {}\n", d_2);
148 auto d_3 = r_2(v_2, v_3);
149 CHECK_EQ(d_3, doctest::Approx(2.0));
150 fmt::print("The squared distance between v_2 and v_3 is {}\n", d_3);
151 }
152 THEN("All vertices have correct timevalues.")
153 {
154 CHECK(triangulation.check_all_vertices());
155 // Human verification
156 auto print = [&triangulation](Vertex_handle_t<3> const& vertex) {
157 fmt::print(
158 "Vertex ({}) with timevalue of {} has a squared radius of {} and "
159 "a squared expected radius of {} with an expected timevalue of "
160 "{}.\n",
161 utilities::point_to_str(vertex->point()), vertex->info(),
162 squared_radius<3>(vertex),
163 std::pow(triangulation.expected_radius(vertex), 2),
164 triangulation.expected_timevalue(vertex));
165 };
166 ranges::for_each(triangulation.get_vertices(), print);
167 }
