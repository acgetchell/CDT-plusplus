123 template <std::uniform_random_bit_generator Generator>
124 void execute(Generator& generator)
125 {
126 while (!m_moves.empty())
127 {
128 auto move_type = m_moves.back();
129 // Record attempted move
130 ++m_attempted[as_integer(move_type)];
131 if (auto result = apply_random_move(std::as_const(m_manifold), move_type,
132 generator);
133 result)
134 {
135 if (ergodic_moves::check_move(m_manifold, *result, move_type))
136 {
137 swap(result.value(), m_manifold);
138 ++m_succeeded[as_integer(move_type)];
139 }
140 else
141 {
142 spdlog::warn(
143 "Move violated a manifold invariant or geometry delta.\n");
144 ++m_failed[as_integer(move_type)];
145 }
146 }
147 else
148 {
149 // Routine inapplicable sites are represented by the failed counter.
150 ++m_failed[as_integer(move_type)];
151 }
152 // Remove move from queue
153 m_moves.pop_back();
154 }
155 } // execute

158 template <std::uniform_random_bit_generator Generator>
159 static auto apply_random_move(ManifoldType const& manifold,
160 move_tracker::move_type const move,
161 Generator& generator) -> ExpectedType
162 {
163 using enum move_tracker::move_type;
164 switch (move)
165 {
166 case TWO_THREE: return ergodic_moves::do_23_move(manifold, generator);
167 case THREE_TWO: return ergodic_moves::do_32_move(manifold, generator);
168 case TWO_SIX: return ergodic_moves::do_26_move(manifold, generator);
169 case SIX_TWO: return ergodic_moves::do_62_move(manifold, generator);
170 case FOUR_FOUR: return ergodic_moves::do_44_move(manifold, generator);
171 }
172 return std::unexpected("Unknown Pachner move.");
173 }
