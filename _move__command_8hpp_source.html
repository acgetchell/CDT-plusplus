89  void execute()
90  {
91 #ifndef NDEBUG

93  m_manifold.print_details();
94  fmt::print("===============================\n");
95 #endif
96 
97  while (!m_moves.empty())
98  {
99  auto move_type = m_moves.back();
100  // Record attempted move
101  m_attempted[move_tracker::as_integer(move_type)]++;
102  // Convert move_type to function
103  auto move_function = as_move_function(move_type);
104  // Execute move
105  if (auto result = apply_move(m_manifold, move_function); result)
106  {
107  result->update();
108  swap(result.value(), m_manifold);
109  m_succeeded[move_tracker::as_integer(move_type)]++;
110  }
111  else
112  {
113  fmt::print("{}\n", result.error());
114  // Track failed moves
115  m_failed[move_tracker::as_integer(move_type)]++;
116  m_moves_to_retry.push_front(move_type);
117  }
118  // Remove move from queue
119  m_moves.pop_back();
120  }
121 #ifndef NDEBUG
122  fmt::print("=== After moves ===\n");
123  print_attempts();
124  print_errors();
125  m_manifold.print_details();
126  fmt::print("===================\n");

128  } // execute
