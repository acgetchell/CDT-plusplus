123 void execute()
124 {
125#ifndef NDEBUG
126 fmt::print("=== Executing: Before moves ===\n");
127 m_manifold.print_details();
128 fmt::print("===============================\n");
129#endif
130
131 while (!m_moves.empty())
132 {
133 auto move_type = m_moves.back();
134 // Record attempted move
135 ++m_attempted[as_integer(move_type)];
136 // Convert move_type to function
137 auto move_function = as_move_function(move_type);
138 // Execute each move on a private candidate so rejection is atomic even
139 // when a move discovers an error after beginning a topology mutation.
140 ManifoldType candidate{m_manifold};
141 if (auto result = apply_move(candidate, move_function); result)
142 {
143 result->update();
144 if (result->is_correct())
145 {
146 swap(result.value(), m_manifold);
147 ++m_succeeded[as_integer(move_type)];
148 }
149 else
150 {
151 fmt::print("Move produced an invalid manifold.\n");
152 ++m_failed[as_integer(move_type)];
153 }
154 }
155 else
156 {
157 fmt::print("{}\n", result.error());
158 // Track failed moves
159 ++m_failed[as_integer(move_type)];
160 }
161 // Remove move from queue
162 m_moves.pop_back();
163 }
164#ifndef NDEBUG
165 fmt::print("=== After moves ===\n");
166 print_attempts();
167 print_successful();
168 print_errors();
169 m_manifold.print_details();
170 fmt::print("===================\n");
171#endif
172 } // execute
