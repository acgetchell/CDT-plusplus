80  void execute()
81  {
82 #ifndef NDEBUG
83  fmt::print("=== Executing: Before moves ===\n");
84  m_manifold.print_details();
85  fmt::print("===============================\n");
86 #endif
87 
88  while (!m_moves.empty())
89  {
90  auto move_type = m_moves.back();
91  // Record attempted move
92  m_attempted_moves[move_tracker::as_integer(move_type)]++;
93  // Convert move_type to function
94  auto move_function = as_move_function(move_type);
95  auto result = apply_move(m_manifold, move_function);
96  m_moves.pop_back();
97  if (result)
98  {
99  result->update();
100  swap(result.value(), m_manifold);
101  }
102  else
103  {
104  fmt::print("{}\n", result.error());
105  // Track failed moves
106  parse_unexpected(result.error());
107  }
108  }
109 #ifndef NDEBUG

111  print_attempts();
112  print_errors();
113  m_manifold.print_details();
114  fmt::print("===================\n");
