111  // Convert move_type to function

113  // Execute move
114  if (auto result = apply_move(m_manifold, move_function); result)
115  {
116  result->update();

118  m_succeeded[move_tracker::as_integer(move_type)]++;
119  }
120  else

123  // Track failed moves

125  m_moves_to_retry.push_front(move_type);
