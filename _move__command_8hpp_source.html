126 template <std::uniform_random_bit_generator Generator>
127 void execute(Generator& generator)
128 {
129 while (!m_moves.empty())
130 {
131 auto move_type = m_moves.back();
132 // Record attempted move
133 ++m_attempted[as_integer(move_type)];
134 if (auto result = apply_random_move(std::as_const(m_manifold),
135 move_type, generator);
136 result)
137 {
138 if (ergodic_moves::detail::check_move(m_manifold, *result, move_type))
139 {
140 swap(result.value(), m_manifold);
141 ++m_succeeded[as_integer(move_type)];
142 }
143 else
144 {
145 spdlog::warn(
146 "Move violated a manifold invariant or geometry delta.\n");
147 ++m_failed[as_integer(move_type)];
148 }
149 }
150 else
151 {
152 // Routine inapplicable sites are represented by the failed counter.
153 ++m_failed[as_integer(move_type)];
154 }
155 // Remove move from queue
156 m_moves.pop_back();
157 }
158 } // execute

161 template <std::uniform_random_bit_generator Generator>
162 static auto apply_random_move(ManifoldType const& manifold,
163 move_tracker::move_type const move,
164 Generator& generator) -> ExpectedType
165 {
166 using enum move_tracker::move_type;
167 switch (move)
168 {
169 case TWO_THREE: return ergodic_moves::do_23_move(manifold, generator);
170 case THREE_TWO: return ergodic_moves::do_32_move(manifold, generator);
171 case TWO_SIX: return ergodic_moves::do_26_move(manifold, generator);
172 case SIX_TWO: return ergodic_moves::do_62_move(manifold, generator);
173 case FOUR_FOUR: return ergodic_moves::do_44_move(manifold, generator);
174 }
175 return std::unexpected("Unknown Pachner move.");
176 }
