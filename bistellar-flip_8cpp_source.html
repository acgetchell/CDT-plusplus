90SCENARIO("Perform bistellar flip on Delaunay triangulation" *
91 doctest::test_suite("bistellar"))
92{
93 GIVEN("A triangulation setup for a bistellar flip")
94 {
95 auto vertices = bistellar_triangulation_vertices();
96 ergodic_moves::Delaunay triangulation(vertices.begin(), vertices.end());
97 WHEN("We have a valid triangulation")
98 {
99 CHECK(triangulation.is_valid());
100 THEN("We can perform a bistellar flip")
101 {
102 // Obtain top and bottom vertices by re-inserting, which returns the
103 // Vertex_handle
104 auto top = triangulation.insert(Point_t<3>{0, 0, 2});
105 auto bottom = triangulation.insert(Point_t<3>{0, 0, 0});
106 auto edges = foliated_triangulations::collect_edges<3>(triangulation);
107 auto pivot_edge = ergodic_moves::find_pivot_edge(triangulation, edges);
108 REQUIRE_MESSAGE(pivot_edge, "No pivot edge found.");
109
110 // Check this didn't actually change vertices in the triangulation
111 REQUIRE_EQ(vertices.size(), 6);
112
113 if (pivot_edge)
114 {
115 auto flipped_triangulation = ergodic_moves::bistellar_flip(
116 triangulation, pivot_edge.value(), top, bottom);
117
118 REQUIRE_MESSAGE(flipped_triangulation, "Bistellar flip failed.");
119 if (flipped_triangulation)
120 {
123 WARN(flipped_triangulation->is_valid());
124 }
125 }
126 }
127 }
