40 [[nodiscard]] inline auto make_physical_parameters(long double const alpha,
41 long double const k,
42 long double const lambda)

44 {
45 if (!std::isfinite(alpha) || !std::isfinite(k) || !std::isfinite(lambda))

109 auto const r3 = mpfr_values::multiply(r2, n1_tl); // r3 = 2*pi*k*n1_tl

114 auto const r6 = mpfr_values::add(r4, r5); // r6 = r4 + r5
115 auto const r7 = mpfr_values::multiply(n3_31, r6); // r7 = n3_31*r6

125 auto const total = mpfr_values::add(r11, r12); // total = r11+r12

221[[nodiscard]] inline auto S3_bulk_action(
222 Int_precision const N1_TL, Int_precision const N3_31_13,
223 Int_precision const N3_22, long double const Alpha, long double const K,
224 long double const Lambda) -> mpfr_values::Value
225{
226 auto const parameters = s3_action::make_physical_parameters(Alpha, K, Lambda);
227
228 // Set input parameters and constants to MPFR equivalents
229 auto const n1_tl = mpfr_values::from_integer(N1_TL);
230 auto const n3_31 = mpfr_values::from_integer(N3_31_13);
231 auto const n3_22 = mpfr_values::from_integer(N3_22);
232 auto const alpha = mpfr_values::from_long_double(parameters.alpha);
233 auto const k = mpfr_values::from_long_double(parameters.k);
234 auto const lambda = mpfr_values::from_long_double(parameters.lambda);
235 auto const two = mpfr_values::from_integer(2);
236 auto const pi = mpfr_values::pi();
237 auto const three = mpfr_values::from_integer(3);
238 auto const four = mpfr_values::from_integer(4);
239 auto const one = mpfr_values::from_integer(1);
240 auto const twelve = mpfr_values::from_integer(12);
241
242 // First term accumulates in r5
243 auto const r1 = mpfr_values::multiply(two, pi); // r1 = 2*pi
244 auto const r2 = mpfr_values::multiply(r1, k); // r2 = 2*pi*k
245 auto const r3 = mpfr_values::square_root(alpha); // r3 = sqrt(alpha)
246 auto const r4 =
247 mpfr_values::multiply(r3, r2); // r4 = r3*r2 = 2*pi*k*sqrt(alpha)
248 auto const r5 = mpfr_values::multiply(r4, n1_tl); // r5 = r4*n1_tl
249
250 // Second term accumulates in r30
251 auto const r6 = mpfr_values::square_root(three); // r6 = sqrt(3)
252 auto const r7 = mpfr_values::multiply(four, alpha); // r7 = 4*alpha
253 auto const r8 = mpfr_values::add(one, r7); // r8 = r7+1 = 4*alpha+1
254 auto const r9 =
255 mpfr_values::square_root(r8); // r9 = sqrt(r8) = sqrt(4*alpha+1)
256 auto const r10 = mpfr_values::multiply(r6, r9); // r10 = r6*r9
257 auto const r11 = mpfr_values::divide(one, r10); // r11 = 1/r10
258 auto const r12 =
259 mpfr_values::inverse_hyperbolic_sine(r11); // r12 = arcsinh(r11)
260 auto const r13 = mpfr_values::negate(three); // r13 = -3
261 auto const r14 = mpfr_values::multiply(r13, k); // r14 = r13*k = -3*k
262 // r15 = r14*r12 = -3*k*arcsinh(r11)
263 auto const r15 = mpfr_values::multiply(r14, r12);
264
265 auto const r16 = mpfr_values::multiply(two, alpha); // r16 = 2*alpha
266 auto const r17 = mpfr_values::add(r16, one); // r17 = 2*alpha+1
267 auto const r18 =
268 mpfr_values::divide(r17, r8); // r18 = (2*alpha+1)/(4*alpha+1)
269 auto const r19 = mpfr_values::arc_cosine(r18); // r19 = arccos(r18)
270 auto const r20 = mpfr_values::multiply(r14, r3); // r20 = -3*k*sqrt(alpha)
271 auto const r21 =
272 mpfr_values::multiply(r20, r19); // r21 = -3*k*sqrt(alpha)*arccos(r18)
273
274 auto const r22 = mpfr_values::multiply(three, alpha); // r22 = 3*alpha
275 auto const r23 = mpfr_values::add(r22, one); // r23 = 3*alpha+1
276 auto const r24 = mpfr_values::square_root(r23); // r24 = sqrt(3*alpha+1)
277 auto const r25 = mpfr_values::negate(lambda); // r25 = -lambda
278 auto const r26 = mpfr_values::divide(r25, twelve); // r26 = -lambda/12
279 auto const r27 =
280 mpfr_values::multiply(r26, r24); // r27 = (-lambda/12)*sqrt(3*alpha+1)
281
282 // Accumulate partial sums of second term
283 auto const r28 = mpfr_values::add(r15, r21); // r28 = r15+r21
284 auto const r29 = mpfr_values::add(r28, r27); // r29 = r28+r27
285
286 // Multiply by overall factor of n3_31
287 auto const r30 = mpfr_values::multiply(n3_31, r29); // r30 = n3_31*r29
288
289 // Third term accumulates in r51
290 auto const r31 = mpfr_values::multiply(two, k); // r31 = 2*k
291 auto const r32 = mpfr_values::square_root(two); // r32 = sqrt(2)
292 auto const r33 = mpfr_values::multiply(two, r32); // r33 = 2*sqrt(2)
293 auto const r34 = mpfr_values::square_root(r17); // r34 = sqrt(2*alpha+1)
294 auto const r35 = mpfr_values::multiply(r33, r34); // r35 = r33*r34
295 auto const r36 =
296 mpfr_values::divide(r35, r8); // r36 = 2*sqrt(2)*sqrt(2*alpha+1)/
297 // 4*alpha+1
298 auto const r37 =
299 mpfr_values::inverse_hyperbolic_sine(r36); // r37 = arcsinh(r36)
300 auto const r38 = mpfr_values::multiply(r31, r37); // r38 = 2*k*arcsinh(r36)
301
302 auto const r39 = mpfr_values::negate(one); // r39 = -1
303 auto const r40 = mpfr_values::divide(r39, r8); // r40 = -1/(4*alpha+1)
304 auto const r41 = mpfr_values::arc_cosine(r40); // r41 = arccos(r40)
305 auto const r42 = mpfr_values::negate(four); // r42 = -4
306 auto const r43 = mpfr_values::multiply(r42, k); // r43 = -4*k
307 auto const r44 = mpfr_values::multiply(r43, r3); // r44 = -4*k*sqrt(alpha)
308 auto const r45 =
309 mpfr_values::multiply(r44, r41); // r45 = -4*k*sqrt(alpha)*arccos(r40)
310
311 auto const r46 = mpfr_values::add(r7, two); // r46 = 4*alpha+2
312 auto const r47 = mpfr_values::square_root(r46); // r47 = sqrt(4*alpha+2)
313 auto const r48 =
314 mpfr_values::multiply(r26, r47); // r48 = (-lambda/12)*sqrt(4*alpha+2)
315
316 // Accumulate partial sums of third term
317 auto const r49 = mpfr_values::add(r38, r45); // r49 = r38+r45
318 auto const r50 = mpfr_values::add(r49, r48); // r50 = r49+r48
319
320 // Multiply by overall factor of n3_22
321 auto const r51 = mpfr_values::multiply(n3_22, r50); // r51 = n3_22*r50
322
323 auto const r52 = mpfr_values::add(r5, r30); // r52 = r5+r30
324 auto const total = mpfr_values::add(r51, r52); // total = r51+r52
325
326 return total;
327} // S3_bulk_action()
