76  for (eit = universe_ptr->finite_edges_begin();
77  eit != universe_ptr->finite_edges_end(); ++eit)

88

130 [[deprecated]] auto classify_simplices(T&& universe_ptr)
131 {
132 #ifndef NDEBUG
133  std::cout << "Classifying simplices....\n";
134 #endif
135  Delaunay3::Finite_cells_iterator cit;
136  std::vector<Cell_handle> three_one;
137  std::vector<Cell_handle> two_two;
138  std::vector<Cell_handle> one_three;
139 
140  // Iterate over all cells in the Delaunay triangulation
141  for (cit = universe_ptr->finite_cells_begin();
142  cit != universe_ptr->finite_cells_end(); ++cit)
143  {
144  int max_values{0};
145  int min_values{0};
146  // Push every time value of every vertex into a list
147  int timevalues[4] = {
148  cit->vertex(0)->info(),
149  cit->vertex(1)->info(),
150  cit->vertex(2)->info(),
151  cit->vertex(3)->info(),
152  };
153  int max_time =
154  *std::max_element(std::begin(timevalues), std::end(timevalues));
155  for (auto elt : timevalues)
156  {
157  if (elt == max_time) { ++max_values; }
158  else
159  {
160  ++min_values;
161  }
162  }
163 
164  // Classify simplex using max_values and write to cit->info()
165  if (min_values == 1 && max_values == 3)
166  {
167  cit->info() = 13;
168  one_three.emplace_back(cit);
169  }
170  else if (min_values == 2 && max_values == 2)
171  {
172  cit->info() = 22;
173  two_two.emplace_back(cit);
174  }
175  else if (min_values == 3 && max_values == 1)
176  {
177  cit->info() = 31;
178  three_one.emplace_back(cit);
179  }
180  else
181  {
182  throw std::runtime_error("Invalid simplex in classify_simplices()!");
183  } // endif
184  } // Finish iterating over cells
185 
186 // Display results if debugging
187 #ifndef NDEBUG

189  << two_two.size() << " (2,2) simplices\n";
190  std::cout << "and " << one_three.size() << " (1,3) simplices.\n";
191 #endif
192  return std::make_tuple(three_one, two_two, one_three);
