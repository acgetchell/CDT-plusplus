33  template <int dimension>
34  [[nodiscard]] inline auto get_vertices_from_cells(
35  std::vector<Cell_handle_t<dimension>> const& t_cells)
36  {
37  std::unordered_set<Vertex_handle_t<dimension>> cell_vertices;
38  for (auto const& cell : t_cells)
39  {
40  // Emplace each vertex in the cell into the container of unique elements
41  for (int j = 0; j < dimension + 1; ++j)
42  {
43  cell_vertices.emplace(cell->vertex(j));
44  }
45  }
46  std::vector<Vertex_handle_t<dimension>> result(cell_vertices.begin(),
47  cell_vertices.end());
48  return result;
49  } // get_vertices_from_cells
