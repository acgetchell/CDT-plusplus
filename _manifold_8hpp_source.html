24  [[nodiscard]] inline auto get_vertices_from_cells(
25  std::vector<Cell_handle_t<dimension>> const& t_cells)
26  {
27  std::unordered_set<Vertex_handle_t<dimension>> cell_vertices;
28  auto get_vertices = [&cell_vertices](auto const& t_cell) {
29  for (int i = 0; i < dimension + 1; ++i)
30  {
31  cell_vertices.emplace(t_cell->vertex(i));
32  }
33  };
34  std::for_each(t_cells.begin(), t_cells.end(), get_vertices);
35  std::vector<Vertex_handle_t<dimension>> result(cell_vertices.begin(),
36  cell_vertices.end());
37  return result;
38  } // get_vertices_from_cells
