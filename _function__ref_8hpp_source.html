317 template <typename Ret, typename... Params>
318 class function_ref<Ret(Params...)>
319 {
320  Ret (*callback)(intptr_t callable, Params... params) = nullptr;
321  intptr_t callable{};
322 
323  template <typename Callable>
324  static auto callback_fn(intptr_t callable, Params... params) -> Ret
325  {
326  return (*reinterpret_cast<Callable*>(callable))( // NOLINT
327  std::forward<Params>(params)...);
328  }
329 
330  public:
331  function_ref() = default;
332  explicit function_ref(std::nullptr_t) {}
333 
334  template <typename Callable>
335  function_ref( // NOLINT
336  Callable&& callable,
337  // This is not the copy-constructor.
338  std::enable_if_t<
339  !std::is_same<std::remove_cv_t<std::remove_reference_t<Callable>>,
340  function_ref>::value>* /*unused*/
341  = nullptr,
342  // Functor must be callable and return a suitable type.
343  std::enable_if_t<std::is_void<Ret>::value ||
344  std::is_convertible<decltype(std::declval<Callable>()(
345  std::declval<Params>()...)),
346  Ret>::value>* /*unused*/
347  = nullptr)
348  : callback(callback_fn<typename std::remove_reference<Callable>::type>)
349  , callable(reinterpret_cast<intptr_t>(&callable)) // NOLINT
350  {}
351 
352  auto operator()(Params... params) const -> Ret
353  {
354  return callback(callable, std::forward<Params>(params)...);
355  }
356 
357  explicit operator bool() const { return callback; }
358 };
