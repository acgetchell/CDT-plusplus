111 auto operator()(ManifoldType const& t_manifold) -> ManifoldType
112 {
113#ifndef NDEBUG
114 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
115#endif
116 fmt::print("Starting Move Always algorithm in {}+1 dimensions ...\n",
117 ManifoldType::dimension - 1);
118 fmt::print("Effective random seed: {} (stream {}).\n", m_random.seed(),
119 m_random.stream());
120
121 m_attempted_moves.reset();
122 m_successful_moves.reset();
123 m_failed_moves.reset();
124
125 // Start the move command
126 MoveCommand command(t_manifold);
127
128 fmt::print("Making random moves ...\n");
129
130 // Loop through passes
131 for (auto pass_number = 1; pass_number <= m_passes; ++pass_number)
132 {
133 fmt::print("=== Pass {} ===\n", pass_number);
134 auto total_simplices_this_pass = command.get_const_results().N3();
135 // Make a random move per simplex
136 for (auto move_attempt = 0; move_attempt < total_simplices_this_pass;
137 ++move_attempt)
138 {
139 // Pick a move to attempt
140 command.enqueue(move_tracker::generate_random_move_3(m_random));
141 }
142 command.execute(m_random);
143 // Update attempted, successful, and failed moves
144 m_attempted_moves += command.get_attempted();
145 m_successful_moves += command.get_succeeded();
146 m_failed_moves += command.get_failed();
147 command.reset_counters();
148
149 if (pass_number % m_checkpoint == 0)
150 {
151 fmt::print("Writing checkpoint for pass {}.\n", pass_number);
152 print_results();
153 utilities::write_file(command.get_results(), m_random.seed(),
154 pass_number);
155 }
156 }
157 print_results();
158 return command.get_results();
159 }
