109 auto operator()(ManifoldType const& t_manifold) -> ManifoldType
110 {
111#ifndef NDEBUG
112 spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
113#endif
114 fmt::print("Starting Move Always algorithm in {}+1 dimensions ...\n",
115 ManifoldType::dimension - 1);
116 fmt::print("Effective random seed: {} (stream {}).\n", m_random.seed(),
117 m_random.stream());
118
119 m_attempted_moves.reset();
120 m_successful_moves.reset();
121 m_failed_moves.reset();
122
123 // Start the move command
124 MoveCommand command(t_manifold);
125
126 fmt::print("Making random moves ...\n");
127
128 // Loop through passes
129 for (auto pass_number = 1; pass_number <= m_passes; ++pass_number)
130 {
131 fmt::print("=== Pass {} ===\n", pass_number);
132 auto total_simplices_this_pass = command.get_const_results().N3();
133 // Make a random move per simplex
134 for (auto move_attempt = 0; move_attempt < total_simplices_this_pass;
135 ++move_attempt)
136 {
137 // Pick a move to attempt
138 command.enqueue(move_tracker::generate_random_move_3(m_random));
139 }
140 command.execute(m_random);
141 // Update attempted, successful, and failed moves
142 m_attempted_moves += command.get_attempted();
143 m_successful_moves += command.get_succeeded();
144 m_failed_moves += command.get_failed();
145 command.reset_counters();
146
147 if (pass_number % m_checkpoint == 0)
148 {
149 fmt::print("Writing checkpoint for pass {}.\n", pass_number);
150 print_results();
151 utilities::write_file(command.get_results(), m_random.seed(),
152 pass_number);
153 }
154 }
155 print_results();
156 return command.get_results();
157 }
