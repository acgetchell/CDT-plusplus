32 auto vertex_state(Vertex_handle_t<3> const& vertex) -> VertexState
33 {
34 auto const& point = vertex->point();
35 return {CGAL::to_double(point.x()), CGAL::to_double(point.y()),
36 CGAL::to_double(point.z()), vertex->info()};
37 }

39 auto vertex_states(Manifold_3 const& manifold) -> std::vector<VertexState>
40 {
41 std::vector<VertexState> states;
42 states.reserve(static_cast<std::size_t>(manifold.vertices()));
43 for (auto vertex = manifold.get_delaunay().finite_vertices_begin();
44 vertex != manifold.get_delaunay().finite_vertices_end(); ++vertex)
45 {
46 states.push_back(vertex_state(vertex));
47 }
48 std::ranges::sort(states);
49 return states;
50 }
51
52 auto cell_states(Manifold_3 const& manifold) -> std::vector<CellState>
53 {
54 std::vector<CellState> states;
55 states.reserve(static_cast<std::size_t>(manifold.simplices()));
56 for (auto cell = manifold.get_delaunay().finite_cells_begin();
57 cell != manifold.get_delaunay().finite_cells_end(); ++cell)
58 {
59 std::array<VertexState, 4> vertices;
60 for (auto index = 0; index < 4; ++index)
61 {
62 vertices.at(static_cast<std::size_t>(index)) =
63 vertex_state(cell->vertex(index));
64 }
65 std::ranges::sort(vertices);
66 states.emplace_back(vertices, cell->info());
67 }
68 std::ranges::sort(states);
69 return states;
70 }

100 void check_single_move_outcome(MoveCommand<Manifold_3> const& command,
101 Manifold_3 const& before,
102 move_tracker::move_type const move,
103 Int_precision const cell_delta)
104 {
105 auto const attempted = command.get_attempted()[move];
106 auto const succeeded = command.get_succeeded()[move];
107 auto const failed = command.get_failed()[move];
108 CHECK_EQ(attempted, 1);
109 CHECK_EQ(succeeded + failed, 1);
110
111 auto const& result = command.get_const_results();
112 CHECK(result.is_valid());
113 if (succeeded == 1)
114 {
115 CHECK_EQ(result.simplices(), before.simplices() + cell_delta);
116 CHECK(ergodic_moves::check_move(before, result, move));
117 }
118 else
119 {
120 CHECK(result.get_delaunay() == before.get_delaunay());
121 CHECK(vertex_states(result) == vertex_states(before));
122 CHECK(cell_states(result) == cell_states(before));
123 CHECK(manifold_counts(result) == manifold_counts(before));
124 }
125 }
