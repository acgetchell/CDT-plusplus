413 [[nodiscard]] inline auto expected_points_per_timeslice(
414  std::size_t const t_dimension, Int_precision const t_number_of_simplices,
415  Int_precision const t_number_of_timeslices, bool const t_output_flag = true)
416 {
417  if (t_output_flag)
418  {
419  fmt::print("{} simplices on {} timeslices desired.\n",
420  t_number_of_simplices, t_number_of_timeslices);
421  }
422 
423  auto const simplices_per_timeslice =
424  t_number_of_simplices / t_number_of_timeslices;
425  if (t_dimension == 3)
426  {
427  // Avoid segfaults for small values
428  if (t_number_of_simplices == t_number_of_timeslices)
429  { return 2 * simplices_per_timeslice; }
430  else if (t_number_of_simplices < 1000)
431  {
432  return static_cast<Int_precision>(
433  0.4L * static_cast<long double>(simplices_per_timeslice));
434  }
435  else if (t_number_of_simplices < 10000)
436  {
437  return static_cast<Int_precision>(
438  0.2L * static_cast<long double>(simplices_per_timeslice));
439  }
440  else if (t_number_of_simplices < 100000)
441  {
442  return static_cast<Int_precision>(
443  0.15L * static_cast<long double>(simplices_per_timeslice));
444  }
445  else
446  {
447  return static_cast<Int_precision>(
448  0.1L * static_cast<long double>(simplices_per_timeslice));
449  }
450  }
451  else
452  {
453  throw std::invalid_argument("Currently, dimensions cannot be >3.");
454  }
455 } // expected_points_per_timeslice
