337 [[nodiscard]] inline auto expected_points_per_timeslice(
338  std::size_t const t_dimension, Int_precision const t_number_of_simplices,
339  Int_precision const t_number_of_timeslices, bool const t_output_flag = true)
340 {
341  if (t_output_flag)
342  {
343  fmt::print("{} simplices on {} timeslices desired.\n",
344  t_number_of_simplices, t_number_of_timeslices);
345  }
346 
347  auto const simplices_per_timeslice =
348  t_number_of_simplices / t_number_of_timeslices;
349  if (t_dimension == 3)
350  {
351  // Avoid segfaults for small values
352  if (t_number_of_simplices == t_number_of_timeslices)
353  { return 2 * simplices_per_timeslice; }
354  if (t_number_of_simplices < 1000) // NOLINT
355  {
356  return static_cast<Int_precision>(
357  0.4L * static_cast<long double>(simplices_per_timeslice)); // NOLINT
358  }
359  if (t_number_of_simplices < 10000) // NOLINT
360  {
361  return static_cast<Int_precision>(
362  0.2L * static_cast<long double>(simplices_per_timeslice)); // NOLINT
363  }
364  if (t_number_of_simplices < 100000) // NOLINT
365  {
366  return static_cast<Int_precision>(
367  0.15L * static_cast<long double>(simplices_per_timeslice)); // NOLINT
368  }
369 
370  return static_cast<Int_precision>(
371  0.1L * static_cast<long double>(simplices_per_timeslice)); // NOLINT
372  }
373  throw std::invalid_argument("Currently, dimensions cannot be >3.");
374 
375 } // expected_points_per_timeslice
