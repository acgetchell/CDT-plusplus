290  [[nodiscard]] inline auto expected_points_per_timeslice(
291  Int_precision t_dimension, Int_precision t_number_of_simplices,
292  Int_precision t_number_of_timeslices)
293  {
294 #ifndef NDEBUG
295  spdlog::debug("{} simplices on {} timeslices desired.\n",
296  t_number_of_simplices, t_number_of_timeslices);
297 #endif
298 
299  auto const simplices_per_timeslice =
300  t_number_of_simplices / t_number_of_timeslices;
301  if (t_dimension == 3)
302  {
303  // Avoid segfaults for small values
304  if (t_number_of_simplices == t_number_of_timeslices)
305  {
306  return 2 * simplices_per_timeslice;
307  }
308  if (t_number_of_simplices < 1000) // NOLINT
309  {
310  return static_cast<Int_precision>(
311  0.4L * // NOLINT
312  static_cast<long double>(simplices_per_timeslice));
313  }
314  if (t_number_of_simplices < 10000) // NOLINT
315  {
316  return static_cast<Int_precision>(
317  0.2L * // NOLINT
318  static_cast<long double>(simplices_per_timeslice));
319  }
320  if (t_number_of_simplices < 100000) // NOLINT
321  {
322  return static_cast<Int_precision>(
323  0.15L * // NOLINT
324  static_cast<long double>(simplices_per_timeslice));
325  }
326 
327  return static_cast<Int_precision>(
328  0.1L * static_cast<long double>(simplices_per_timeslice)); // NOLINT
329  }
330  throw std::invalid_argument("Currently, dimensions cannot be >3.");
331 
332  } // expected_points_per_timeslice
