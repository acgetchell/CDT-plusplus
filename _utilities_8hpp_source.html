434 [[nodiscard]] inline auto expected_points_per_timeslice(
435  std::size_t const t_dimension, Int_precision const t_number_of_simplices,
436  Int_precision const t_number_of_timeslices, bool const t_output_flag = true)
437 {
438  if (t_output_flag)
439  {
440  fmt::print("{} simplices on {} timeslices desired.\n",
441  t_number_of_simplices, t_number_of_timeslices);
442  }
443 
444  auto const simplices_per_timeslice =
445  t_number_of_simplices / t_number_of_timeslices;
446  if (t_dimension == 3)
447  {
448  // Avoid segfaults for small values
449  if (t_number_of_simplices == t_number_of_timeslices)
450  { return 2 * simplices_per_timeslice; }
451  else if (t_number_of_simplices < 1000)
452  {
453  return static_cast<Int_precision>(
454  0.4L * static_cast<long double>(simplices_per_timeslice));
455  }
456  else if (t_number_of_simplices < 10000)
457  {
458  return static_cast<Int_precision>(
459  0.2L * static_cast<long double>(simplices_per_timeslice));
460  }
461  else if (t_number_of_simplices < 100000)
462  {
463  return static_cast<Int_precision>(
464  0.15L * static_cast<long double>(simplices_per_timeslice));
465  }
466  else
467  {
468  return static_cast<Int_precision>(
469  0.1L * static_cast<long double>(simplices_per_timeslice));
470  }
471  }
472  else
473  {
474  throw std::invalid_argument("Currently, dimensions cannot be >3.");
475  }
476 } // expected_points_per_timeslice
