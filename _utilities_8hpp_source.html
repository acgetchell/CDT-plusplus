298 [[nodiscard]] inline auto expected_points_per_timeslice(
299  std::size_t const t_dimension, Int_precision const t_number_of_simplices,
300  Int_precision const t_number_of_timeslices, bool const t_output_flag = true)
301 {
302  if (t_output_flag)
303  {
304  fmt::print("{} simplices on {} timeslices desired.\n",
305  t_number_of_simplices, t_number_of_timeslices);
306  }
307 
308  auto const simplices_per_timeslice =
309  t_number_of_simplices / t_number_of_timeslices;
310  if (t_dimension == 3)
311  {
312  // Avoid segfaults for small values
313  if (t_number_of_simplices == t_number_of_timeslices)
314  { return 2 * simplices_per_timeslice; }
315  if (t_number_of_simplices < 1000) // NOLINT
316  {
317  return static_cast<Int_precision>(
318  0.4L * static_cast<long double>(simplices_per_timeslice)); // NOLINT
319  }
320  if (t_number_of_simplices < 10000) // NOLINT
321  {
322  return static_cast<Int_precision>(
323  0.2L * static_cast<long double>(simplices_per_timeslice)); // NOLINT
324  }
325  if (t_number_of_simplices < 100000) // NOLINT
326  {
327  return static_cast<Int_precision>(
328  0.15L * static_cast<long double>(simplices_per_timeslice)); // NOLINT
329  }
330 
331  return static_cast<Int_precision>(
332  0.1L * static_cast<long double>(simplices_per_timeslice)); // NOLINT
333  }
334  throw std::invalid_argument("Currently, dimensions cannot be >3.");
335 
336 } // expected_points_per_timeslice
