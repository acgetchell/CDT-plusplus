178 [[nodiscard]] inline auto payload_integrity(
179 std::filesystem::path const& filename) -> Payload_integrity
180 {
181 std::ifstream input(filename, std::ios::in | std::ios::binary);
182 if (!input.is_open())
183 {
184 throw std::filesystem::filesystem_error(
185 "Could not open payload for integrity validation", filename,
186 std::make_error_code(std::errc::bad_file_descriptor));
187 }
188
189 std::uint64_t digest{14695981039346656037ULL};
190 std::uint64_t size{};
191 std::array<char, 8192> buffer{};
192 while (input)
193 {
194 input.read(buffer.data(), static_cast<std::streamsize>(buffer.size()));
195 auto const count = input.gcount();
196 for (std::streamsize index = 0; index < count; ++index)
197 {
198 digest ^= static_cast<unsigned char>(
199 buffer[static_cast<std::size_t>(index)]);
200 digest *= 1099511628211ULL;
201 }
202 size += static_cast<std::uint64_t>(count);
203 }
204 if (!input.eof())
205 {
206 throw std::filesystem::filesystem_error(
207 "Could not read payload for integrity validation", filename,
208 std::make_error_code(std::errc::io_error));
209 }
210 return {size, digest};
211 }

259 [[nodiscard]] inline auto fingerprint_records(
260 std::vector<std::string> const& records) -> std::uint64_t
261 {
262 std::uint64_t digest{14695981039346656037ULL};
263 for (auto const& record : records)
264 {
265 for (auto const byte : record)
266 {
267 digest ^= static_cast<unsigned char>(byte);
268 digest *= 1099511628211ULL;
269 }
270 digest ^= 0xFFU;
271 digest *= 1099511628211ULL;
272 }
273 return digest;
274 }

530 template <typename TriangulationType>
531 void read_causal_info(std::istream& input, TriangulationType& triangulation,
532 std::filesystem::path const& path)
533 {
534 std::string line;
535 if (!std::getline(input, line) || line != CAUSAL_INFO_HEADER)
536 {
537 throw std::filesystem::filesystem_error(
538 "Unexpected trailing data after triangulation", path,
539 std::make_error_code(std::errc::illegal_byte_sequence));
540 }
541
542 if (!std::getline(input, line))
543 {
544 throw std::filesystem::filesystem_error(
545 "Truncated causal triangulation metadata", path,
546 std::make_error_code(std::errc::illegal_byte_sequence));
547 }
548 auto const vertex_count = parse_count_line(line, "vertices=", path);
549 if (vertex_count !=
550 static_cast<std::uint64_t>(triangulation.number_of_vertices()))
551 {
552 throw std::filesystem::filesystem_error(
553 "Causal vertex metadata count does not match triangulation", path,
554 std::make_error_code(std::errc::illegal_byte_sequence));
555 }
556
557 std::map<std::string, Int_precision> vertex_info;
558 for (std::uint64_t index = 0; index < vertex_count; ++index)
559 {
560 if (!std::getline(input, line))
561 {
562 throw std::filesystem::filesystem_error(
563 "Truncated causal vertex metadata", path,
564 std::make_error_code(std::errc::illegal_byte_sequence));
565 }
566 auto [key, value] = parse_record(line, "v=", path);
567 if (!vertex_info.emplace(std::move(key), value).second)
568 {
569 throw std::filesystem::filesystem_error(
570 "Duplicate causal vertex metadata", path,
571 std::make_error_code(std::errc::illegal_byte_sequence));
572 }
573 }
574
575 if (!std::getline(input, line))
576 {
577 throw std::filesystem::filesystem_error(
578 "Truncated causal triangulation metadata", path,
579 std::make_error_code(std::errc::illegal_byte_sequence));
580 }
581 auto const cell_count = parse_count_line(line, "cells=", path);
582 if (cell_count !=
583 static_cast<std::uint64_t>(triangulation.number_of_finite_cells()))
584 {
585 throw std::filesystem::filesystem_error(
586 "Causal cell metadata count does not match triangulation", path,
587 std::make_error_code(std::errc::illegal_byte_sequence));
588 }
589
590 std::map<std::string, Int_precision> cell_info;
591 for (std::uint64_t index = 0; index < cell_count; ++index)
592 {
593 if (!std::getline(input, line))
594 {
595 throw std::filesystem::filesystem_error(
596 "Truncated causal cell metadata", path,
597 std::make_error_code(std::errc::illegal_byte_sequence));
598 }
599 auto [key, value] = parse_record(line, "c=", path);
600 if (!cell_info.emplace(std::move(key), value).second)
601 {
602 throw std::filesystem::filesystem_error(
603 "Duplicate causal cell metadata", path,
604 std::make_error_code(std::errc::illegal_byte_sequence));
605 }
606 }
607
608 for (auto vertex = triangulation.finite_vertices_begin();
609 vertex != triangulation.finite_vertices_end(); ++vertex)
610 {
611 auto const found = vertex_info.find(point_key(vertex->point()));
612 if (found == vertex_info.end())
613 {
614 throw std::filesystem::filesystem_error(
615 "Causal vertex metadata does not match triangulation", path,
616 std::make_error_code(std::errc::illegal_byte_sequence));
617 }
618 vertex->info() = found->second;
619 vertex_info.erase(found);
620 }
621 for (auto cell = triangulation.finite_cells_begin();
622 cell != triangulation.finite_cells_end(); ++cell)
623 {
624 auto const found = cell_info.find(cell_key(cell));
625 if (found == cell_info.end())
626 {
627 throw std::filesystem::filesystem_error(
628 "Causal cell metadata does not match triangulation", path,
629 std::make_error_code(std::errc::illegal_byte_sequence));
630 }
631 cell->info() = found->second;
632 cell_info.erase(found);
633 }
634 if (!vertex_info.empty() || !cell_info.empty())
635 {
636 throw std::filesystem::filesystem_error(
637 "Causal metadata contains records outside the triangulation", path,
638 std::make_error_code(std::errc::illegal_byte_sequence));
639 }
640 }

663 {

666 {
667 throw std::filesystem::filesystem_error(
668 "Could not open persistence metadata", path,
669 std::make_error_code(std::errc::bad_file_descriptor));
670 }
671
672 std::string line;
673 if (!std::getline(input, line) || line != "cdt-plusplus-metadata-v1")
674 {
675 throw std::filesystem::filesystem_error(
676 "Unsupported or malformed persistence metadata", path,
677 std::make_error_code(std::errc::illegal_byte_sequence));
678 }
679
680 std::map<std::string, std::string> values;
681 while (std::getline(input, line))
682 {
683 auto const separator = line.find('=');
684 if (separator == std::string::npos || separator == 0 ||
685 separator + 1 == line.size())
686 {
687 throw std::filesystem::filesystem_error(
688 "Malformed persistence metadata", path,
689 std::make_error_code(std::errc::illegal_byte_sequence));
690 }
691 auto [unused, inserted] = values.emplace(line.substr(0, separator),
692 line.substr(separator + 1));
693 if (!inserted)
694 {
695 throw std::filesystem::filesystem_error(
696 "Duplicate persistence metadata field", path,
697 std::make_error_code(std::errc::illegal_byte_sequence));
698 }
699 }
700 if (!input.eof())
701 {
702 throw std::filesystem::filesystem_error(
703 "Could not read persistence metadata", path,
704 std::make_error_code(std::errc::io_error));
705 }
706
707 for (auto const required : {"payload.size",
708 "payload.fnv1a64",
709 "artifact",
710 "resume_supported",
711 "fresh_topology_replay_supported",
712 "transition_replay_requires_identical_start",
713 "cdt.version",
714 "build.compiler_id",
715 "build.compiler_version",

954 }

1013 template <typename TriangulationType>
1014 [[nodiscard]] auto parse_payload(std::filesystem::path const& filename)
1015 -> TriangulationType
1016 {
1017 std::ifstream file(filename, std::ios::in);
1018 if (!file.is_open())
1019 {
1020 throw std::filesystem::filesystem_error(
1021 "Could not open file for reading", filename,
1022 std::make_error_code(std::errc::bad_file_descriptor));
1023 }
1024 TriangulationType triangulation;
1025 file >> triangulation;
1026 if (!file)
1027 {
1028 throw std::filesystem::filesystem_error(
1029 "Could not parse triangulation", filename,
1030 std::make_error_code(std::errc::illegal_byte_sequence));
1031 }
1032 file >> std::ws;
1033 if (!file.eof())
1034 {
1035 if constexpr (HAS_CAUSAL_INFO<TriangulationType>)
1036 {
1037 read_causal_info(file, triangulation, filename);
1038 file >> std::ws;
1039 }
1040 }
1041 if (!file.eof())
1042 {
1043 throw std::filesystem::filesystem_error(
1044 "Unexpected trailing data after triangulation", filename,
1045 std::make_error_code(std::errc::illegal_byte_sequence));
1046 }
1047 if constexpr (requires(TriangulationType const& value) {
1048 { value.tds().is_valid() } -> std::convertible_to<bool>;
1049 })
1050 {
1051 if (!triangulation.tds().is_valid())
1052 {
1053 throw std::filesystem::filesystem_error(
1054 "Parsed triangulation data structure failed its integrity check",
1055 filename, std::make_error_code(std::errc::illegal_byte_sequence));
1056 }
1057 }
1058 // Evolved CDT states are valid abstract causal triangulations but need
1059 // not satisfy the Euclidean Delaunay empty-sphere property. For CGAL
1060 // triangulations the TDS check above is therefore the authoritative
1061 // integrity check. Use a type's broader validator only when it does not
1062 // expose a distinct triangulation data structure.
1063 if constexpr (
1064 !requires(TriangulationType const& value) {
1065 { value.tds().is_valid() } -> std::convertible_to<bool>;
1066 } &&
1067 requires(TriangulationType const& value) {
1068 { value.is_valid() } -> std::convertible_to<bool>;
1069 })
1070 {
1071 if (!triangulation.is_valid())
1072 {
1073 throw std::filesystem::filesystem_error(
1074 "Parsed triangulation failed its integrity check", filename,
1075 std::make_error_code(std::errc::illegal_byte_sequence));
1076 }
1077 }
1078 return triangulation;
1079 }

1081 template <typename TriangulationType>
1082 void reconcile_payload_metadata(Reproducibility_metadata& metadata,
1083 TriangulationType const& triangulation)
1084 {
1085 if constexpr (requires(TriangulationType const& value) {
1086 value.dimension();
1087 value.number_of_vertices();
1088 value.number_of_finite_edges();
1089 value.number_of_finite_facets();
1090 value.number_of_finite_cells();
1091 })
1092 {
1093 metadata.dimension =
1094 gsl::narrow<Int_precision>(triangulation.dimension());
1095 metadata.actual_vertices =
1096 gsl::narrow<Int_precision>(triangulation.number_of_vertices());
1097 metadata.actual_edges =
1098 gsl::narrow<Int_precision>(triangulation.number_of_finite_edges());
1099 metadata.actual_faces =
1100 gsl::narrow<Int_precision>(triangulation.number_of_finite_facets());
1101 metadata.actual_simplices =
1102 gsl::narrow<Int_precision>(triangulation.number_of_finite_cells());
1103 }
1104 if constexpr (HAS_CAUSAL_INFO<TriangulationType>)
1105 {
1106 if (triangulation.number_of_vertices() == 0)
1107 {
1108 metadata.minimum_timeslice = 0;
1109 metadata.maximum_timeslice = 0;
1110 }
1111 else
1112 {
1113 auto vertex = triangulation.finite_vertices_begin();
1114 auto minimum_timeslice = static_cast<Int_precision>(vertex->info());
1115 auto maximum_timeslice = minimum_timeslice;
1116 for (++vertex; vertex != triangulation.finite_vertices_end();
1117 ++vertex)
1118 {
1119 auto const time = static_cast<Int_precision>(vertex->info());
1120 minimum_timeslice = std::min(minimum_timeslice, time);
1121 maximum_timeslice = std::max(maximum_timeslice, time);
1122 }
1123 metadata.minimum_timeslice = minimum_timeslice;
1124 metadata.maximum_timeslice = maximum_timeslice;
1125 }
1126 metadata.placement_fingerprint =
1127 canonical_placement_fingerprint(triangulation);
1128 metadata.topology_fingerprint =
1129 canonical_topology_fingerprint(triangulation);
1130 }
1131 }

1162 template <typename TriangulationType>
1163 void validate_serialized_payload(std::filesystem::path const& filename,
1164 TriangulationType const& original)
1165 {
1166 if constexpr (requires(std::istream& input, TriangulationType& value) {
1167 input >> value;
1168 } && std::default_initializable<TriangulationType>)
1169 {
1170 auto const parsed = parse_payload<TriangulationType>(filename);
1171 if constexpr (requires(TriangulationType const& value) {
1172 value.dimension();
1173 value.number_of_vertices();
1174 value.number_of_finite_edges();
1175 value.number_of_finite_facets();
1176 value.number_of_finite_cells();
1177 })
1178 {
1179 if (parsed.dimension() != original.dimension() ||
1180 parsed.number_of_vertices() != original.number_of_vertices() ||
1181 parsed.number_of_finite_edges() !=
1182 original.number_of_finite_edges() ||
1183 parsed.number_of_finite_facets() !=
1184 original.number_of_finite_facets() ||
1185 parsed.number_of_finite_cells() !=
1186 original.number_of_finite_cells())
1187 {
1188 throw std::filesystem::filesystem_error(
1189 "Serialized triangulation changed its incidence counts",
1190 filename,
1191 std::make_error_code(std::errc::illegal_byte_sequence));
1192 }
1193 if constexpr (HAS_CAUSAL_INFO<TriangulationType>)
1194 {
1195 if (canonical_topology_fingerprint(parsed) !=
1196 canonical_topology_fingerprint(original))
1197 {
1198 throw std::filesystem::filesystem_error(
1199 "Serialized triangulation changed its causal topology",
1200 filename,
1201 std::make_error_code(std::errc::illegal_byte_sequence));
1202 }
1203 }
1204 }
1205 else if constexpr (requires(TriangulationType const& left,
1206 TriangulationType const& right) {
1207 { left == right } -> std::convertible_to<bool>;
1208 })
1209 {
1210 if (!(parsed == original))
1211 {
1212 throw std::filesystem::filesystem_error(
1213 "Serialized triangulation did not round-trip exactly", filename,
1214 std::make_error_code(std::errc::illegal_byte_sequence));
1215 }
1216 }
1217 }

1253 template <typename TriangulationType>
1254 void write_payload(std::filesystem::path const& filename,
1255 TriangulationType const& triangulation,
1256 std::optional<Reproducibility_metadata> const& metadata)
1257 {
1258 WriteFileOperation const operation;
1259 fmt::print("Writing to file {}\n", filename.string());
1260 std::scoped_lock const lock(write_file_mutex());
1261 auto temporary = filename;
1262 temporary += ".tmp";
1263 auto const metadata_destination = metadata_filename(filename);
1264 auto metadata_temporary = metadata_destination;
1265 metadata_temporary += ".tmp";
1266 auto resolved_metadata = metadata;
1267 if (resolved_metadata)
1268 {
1269 reconcile_payload_metadata(*resolved_metadata, triangulation);
1270 }
1271
1272 std::error_code cleanup_error;
1273 std::filesystem::remove(temporary, cleanup_error);
1274 std::filesystem::remove(metadata_temporary, cleanup_error);
1275 try
1276 {
1277 std::ofstream file(temporary, std::ios::out | std::ios::trunc);
1278 if (!file.is_open())
1279 {
1280 throw std::filesystem::filesystem_error(
1281 "Could not open temporary file for writing", filename,
1282 std::make_error_code(std::errc::bad_file_descriptor));
1283 }
1284 file << std::setprecision(std::numeric_limits<double>::max_digits10)
1285 << triangulation;
1286 write_causal_info(file, triangulation);
1287 if (!file)
1288 {
1289 throw std::filesystem::filesystem_error(
1290 "Could not serialize triangulation", filename,
1291 std::make_error_code(std::errc::io_error));
1292 }
1293 file.flush();
1294 if (!file)
1295 {
1296 throw std::filesystem::filesystem_error(
1297 "Could not flush serialized triangulation", filename,
1298 std::make_error_code(std::errc::io_error));
1299 }
1300 file.close();
1301 if (!file)
1302 {
1303 throw std::filesystem::filesystem_error(
1304 "Could not close serialized triangulation", filename,
1305 std::make_error_code(std::errc::io_error));
1306 }
1307
1308 validate_serialized_payload(temporary, triangulation);
1309 if (resolved_metadata)
1310 {
1311 auto const integrity = payload_integrity(temporary);
1312 write_text(metadata_temporary,
1313 metadata_text(*resolved_metadata, integrity));
1314 auto const recorded = read_persistence_metadata(metadata_temporary);
1315 if (recorded.payload.size != integrity.size ||
1316 recorded.payload.digest != integrity.digest)
1317 {
1318 throw std::filesystem::filesystem_error(
1319 "Persistence metadata did not round-trip exactly",
1320 metadata_temporary,
1321 std::make_error_code(std::errc::illegal_byte_sequence));
1322 }
1323 validate_persistence_metadata(recorded, triangulation, temporary,
1324 metadata_temporary);
1325
1326 // Publish the manifest first. A process interrupted between these
1327 // replacements observes a detectable checksum mismatch, never a
1328 // payload silently paired with stale provenance.
1329 replace_file(metadata_temporary, metadata_destination);
1330 }
1331 replace_file(temporary, filename);
1332 if (!resolved_metadata)
1333 {
1334 std::filesystem::remove(metadata_destination, cleanup_error);
1335 }
1336 }
1337 catch (...)
1338 {
1339 std::filesystem::remove(temporary, cleanup_error);
1340 std::filesystem::remove(metadata_temporary, cleanup_error);
1341 throw;
1342 }
1343 }

1370 [[nodiscard]] inline auto make_filename(topology_type const& t_topology,
1371 Int_precision t_dimension,
1372 Int_precision t_number_of_simplices,
1373 Int_precision t_number_of_timeslices,
1374 double t_initial_radius,
1375 double t_foliation_spacing)
1376 -> std::filesystem::path
1377 {
1378 std::string filename;
1379 if (t_topology == topology_type::SPHERICAL) { filename += "S"; }
1380 else
1381 {
1382 filename += "T";
1383 }
1384 // std::to_string() works in C++11, but not earlier
1385 filename += std::to_string(t_dimension);
1386
1387 filename += "-";
1388
1389 filename += std::to_string(t_number_of_timeslices);
1390
1391 filename += "-";
1392
1393 filename += std::to_string(t_number_of_simplices);
1394
1395 filename += "-I";
1396
1397 filename += std::to_string(t_initial_radius);
1398
1399 filename += "-R";
1400
1401 filename += std::to_string(t_foliation_spacing);
1402
1403 // Append current time
1404 filename += "-";
1405 auto timestamp = current_date_time();
1406 std::replace(timestamp.begin(), timestamp.end(), ':', '-');
1407 filename += timestamp;

1716 inline auto expected_points_per_timeslice(
1717 Int_precision const t_dimension, Int_precision t_number_of_simplices,
1718 Int_precision t_number_of_timeslices)
1719 {

1721 spdlog::debug("{} simplices on {} timeslices desired.\n",

1724
1725 if (t_dimension != 3)
1726 {
1727 throw std::invalid_argument(
1728 "Only three-dimensional triangulations are supported.");

1730 if (t_number_of_simplices <= 0 || t_number_of_timeslices <= 0)

1732 throw std::invalid_argument(
1733 "Simplices and timeslices must both be positive.");
1734 }
1735

1737 t_number_of_simplices / t_number_of_timeslices;
1738 // Avoid segfaults for small values
1739 if (t_number_of_simplices == t_number_of_timeslices)
1740 {

1742 }
1743 if (t_number_of_simplices < 1000) // NOLINT
1744 {
1745 return static_cast<Int_precision>(
1746 0.4L * // NOLINT
1747 static_cast<long double>(simplices_per_timeslice));
1748 }
1749 if (t_number_of_simplices < 10000) // NOLINT
1750 {
1751 return static_cast<Int_precision>(
1752 0.2L * // NOLINT
1753 static_cast<long double>(simplices_per_timeslice));
1754 }
1755 if (t_number_of_simplices < 100000) // NOLINT
1756 {
1757 return static_cast<Int_precision>(
1758 0.15L * // NOLINT
1759 static_cast<long double>(simplices_per_timeslice));
1760 }
1761
1762 return static_cast<Int_precision>(
1763 0.1L * static_cast<long double>(simplices_per_timeslice)); // NOLINT

1765 } // expected_points_per_timeslice
