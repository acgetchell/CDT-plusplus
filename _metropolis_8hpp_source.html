150  auto CalculateA1(move_tracker::move_type move) const noexcept
151  {
152  auto all_moves = m_proposed_moves.total();
153  auto this_move = m_proposed_moves[move];
154  // Set precision for initialization and assignment functions
155  mpfr_set_default_prec(PRECISION);
156 
157  // Initialize for MPFR
158  mpfr_t r1;
159  mpfr_t r2;
160  mpfr_t a1;
161  mpfr_inits2(PRECISION, r1, r2, a1, nullptr); // NOLINT
162 
163  mpfr_init_set_si(r1, this_move, MPFR_RNDD); // r1 = this_move NOLINT
164  mpfr_init_set_si(r2, all_moves, MPFR_RNDD); // r2 = total_moves NOLINT
165 
166  // The result
167  mpfr_div(a1, r1, r2, MPFR_RNDD); // a1 = r1/r2 NOLINT
168 
169  // Convert mpfr_t total to Gmpzf result by using Gmpzf(double d)
170  auto result = mpfr_get_ld(a1, MPFR_RNDD); // NOLINT
171 
172  // Free memory
173  mpfr_clears(r1, r2, a1, nullptr); // NOLINT
174 
175 #ifndef NDEBUG
176  spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
177  spdlog::trace("Total proposed moves = {}\n", all_moves);
178  spdlog::trace("A1 is {}\n", result);
179 #endif
180 
181  return result;
182  } // CalculateA1()

189  template <int dimension>
190  auto CalculateA2(move_tracker::move_type move) const noexcept
191  {
192  if (dimension == 3)
193  {
194  auto currentS3Action =
195  S3_bulk_action(m_geometry.N1_TL, m_geometry.N3_31_13,
196  m_geometry.N3_22, m_Alpha, m_K, m_Lambda);
197  auto newS3Action = static_cast<Gmpzf>(0);
198  switch (move)
199  {
200  case move_tracker::move_type::TWO_THREE:
201  // A (2,3) move adds a timelike edge
202  // and a (2,2) simplex
203  newS3Action =
204  S3_bulk_action(m_geometry.N1_TL + 1, m_geometry.N3_31_13,
205  m_geometry.N3_22 + 1, m_Alpha, m_K, m_Lambda);
206  break;
207  case move_tracker::move_type::THREE_TWO:
208  // A (3,2) move removes a timelike edge
209  // and a (2,2) simplex
210  newS3Action =
211  S3_bulk_action(m_geometry.N1_TL - 1, m_geometry.N3_31_13,
212  m_geometry.N3_22 - 1, m_Alpha, m_K, m_Lambda);
213  break;
214  case move_tracker::move_type::TWO_SIX:
215  // A (2,6) move adds 2 timelike edges and
216  // 2 (1,3) and 2 (3,1) simplices
217  newS3Action =
218  S3_bulk_action(m_geometry.N1_TL + 2, m_geometry.N3_31_13 + 4,
219  m_geometry.N3_22, m_Alpha, m_K, m_Lambda);
220  break;
221  case move_tracker::move_type::SIX_TWO:
222  // A (6,2) move removes 2 timelike edges and
223  // 2 (1,3) and 2 (3,1) simplices
224  newS3Action =
225  S3_bulk_action(m_geometry.N1_TL - 2, m_geometry.N3_31_13,
226  m_geometry.N3_22 - 4, m_Alpha, m_K, m_Lambda);
227  break;
228  case move_tracker::move_type::FOUR_FOUR:
229  // A (4,4) move changes nothing with respect to the action,
230  // and e^0==1
231 #ifndef NDEBUG
232  spdlog::trace("A2 is 1\n");
233 #endif
234  return static_cast<long double>(1);
235  default:
236  break;
237  }
238 
239  auto exponent = currentS3Action - newS3Action;
240  auto exponent_double = utilities::Gmpzf_to_double(exponent);
241 
242  // if exponent > 0 then e^exponent >=1 so according to Metropolis
243  // algorithm return A2=1
244  if (exponent >= 0) { return static_cast<long double>(1); }
245 
246  // Set precision for initialization and assignment functions
247  mpfr_set_default_prec(PRECISION);
248 
249  // Initialize for MPFR
250  mpfr_t r1;
251  mpfr_t a2;
252  mpfr_inits2(PRECISION, r1, a2, nullptr); // NOLINT
253 
254  // Set input parameters and constants to mpfr_t equivalents
255  mpfr_init_set_ld(r1, exponent_double, // NOLINT
256  MPFR_RNDD); // r1 = exponent
257 
258  // e^exponent
259  mpfr_exp(a2, r1, MPFR_RNDD); // NOLINT
260 
261  // Convert mpfr_t total to Gmpzf result by using Gmpzf(double d)
262  auto result = mpfr_get_ld(a2, MPFR_RNDD); // NOLINT
263 
264  // Free memory
265  mpfr_clears(r1, a2, nullptr); // NOLINT
266 
267 #ifndef NDEBUG
268  spdlog::trace("A2 is {}\n", result);
269 #endif
270 
271  return result;
272  }

282  auto try_move(move_tracker::move_type move) -> bool
283  {
284  // Record the trial move
285  m_proposed_moves[move_tracker::as_integer(move)]++;
286 
287  // Calculate probability
288  auto a1 = CalculateA1(move);
289 
290  // Make move if random number < probability
291  auto a2 = CalculateA2<3>(move);
292 
293  if (auto const trial_value = utilities::generate_probability();
294  trial_value <= a1 * a2)
295  {
296 #ifndef NDEBUG
297  spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
298  spdlog::trace("Trying move.\n");
299  spdlog::trace("Move type = {}\n", move_tracker::as_integer(move));
300  spdlog::trace("Trial_value = {}\n", trial_value);
301  spdlog::trace("A1 = {}\n", a1);
302  spdlog::trace("A2 = {}\n", a2);
303  spdlog::trace("A1*A2 = {}\n", a1 * a2);
304  spdlog::trace("{}\n", (trial_value <= a1 * a2) ? "Move accepted."
305  : "Move rejected.");
306 #endif
307  m_accepted_moves[move_tracker::as_integer(move)]++;
308  return true;
309  }
310 
311  m_rejected_moves[move_tracker::as_integer(move)]++;
312  return false;
313
