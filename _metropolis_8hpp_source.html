154  {

157  // Set precision for initialization and assignment functions
158  mpfr_set_default_prec(PRECISION);
159 
160  // Initialize for MPFR
161  mpfr_t r1, r2, a1; // NOLINT
162  mpfr_inits2(PRECISION, r1, r2, a1, nullptr); // NOLINT
163 
164  mpfr_init_set_ui(r1, this_move, MPFR_RNDD); // r1 = this_move NOLINT
165  mpfr_init_set_ui(r2, all_moves, MPFR_RNDD); // r2 = total_moves NOLINT
166 
167  // The result
168  mpfr_div(a1, r1, r2, MPFR_RNDD); // a1 = r1/r2 NOLINT

173  auto result = mpfr_get_d(a1, MPFR_RNDD); // NOLINT
174 
175  // Free memory
176  mpfr_clears(r1, r2, a1, nullptr); // NOLINT

178 #ifndef NDEBUG

192  auto CalculateA2(move_tracker::move_type move) const noexcept
193  {
194  if (dimension == 3)
195  {
196  auto currentS3Action =
197  S3_bulk_action(m_geometry.N1_TL, m_geometry.N3_31_13,
198  m_geometry.N3_22, m_Alpha, m_K, m_Lambda);
199  auto newS3Action = static_cast<Gmpzf>(0);
200  switch (move)
201  {
202  case move_tracker::move_type::TWO_THREE:
203  // A (2,3) move adds a timelike edge
204  // and a (2,2) simplex
205  newS3Action =
206  S3_bulk_action(m_geometry.N1_TL + 1, m_geometry.N3_31_13,
207  m_geometry.N3_22 + 1, m_Alpha, m_K, m_Lambda);
208  break;
209  case move_tracker::move_type::THREE_TWO:
210  // A (3,2) move removes a timelike edge
211  // and a (2,2) simplex
212  newS3Action =
213  S3_bulk_action(m_geometry.N1_TL - 1, m_geometry.N3_31_13,
214  m_geometry.N3_22 - 1, m_Alpha, m_K, m_Lambda);
215  break;
216  case move_tracker::move_type::TWO_SIX:
217  // A (2,6) move adds 2 timelike edges and
218  // 2 (1,3) and 2 (3,1) simplices
219  newS3Action =
220  S3_bulk_action(m_geometry.N1_TL + 2, m_geometry.N3_31_13 + 4,
221  m_geometry.N3_22, m_Alpha, m_K, m_Lambda);
222  break;
223  case move_tracker::move_type::SIX_TWO:
224  // A (6,2) move removes 2 timelike edges and
225  // 2 (1,3) and 2 (3,1) simplices
226  newS3Action =
227  S3_bulk_action(m_geometry.N1_TL - 2, m_geometry.N3_31_13,
228  m_geometry.N3_22 - 4, m_Alpha, m_K, m_Lambda);
229  break;
230  case move_tracker::move_type::FOUR_FOUR:
231  // A (4,4) move changes nothing with respect to the action,
232  // and e^0==1
233 #ifndef NDEBUG
234  fmt::print("A2 is 1\n");
235 #endif
236  return static_cast<double>(1);
237  default:
238  break;
239  }
240 
241  auto exponent = currentS3Action - newS3Action;
242  auto exponent_double = Gmpzf_to_double(exponent);
243 
244  // if exponent > 0 then e^exponent >=1 so according to Metropolis
245  // algorithm return A2=1
246  if (exponent >= 0) { return static_cast<double>(1); }
247 
248  // Set precision for initialization and assignment functions
249  mpfr_set_default_prec(PRECISION);
250 
251  // Initialize for MPFR
252  mpfr_t r1, a2; // NOLINT
253  mpfr_inits2(PRECISION, r1, a2, nullptr); // NOLINT
254 
255  // Set input parameters and constants to mpfr_t equivalents
256  mpfr_init_set_d(r1, exponent_double, // NOLINT
257  MPFR_RNDD); // r1 = exponent
258 
259  // e^exponent
260  mpfr_exp(a2, r1, MPFR_RNDD); // NOLINT
261 
262  // Convert mpfr_t total to Gmpzf result by using Gmpzf(double d)
263  auto result = mpfr_get_d(a2, MPFR_RNDD); // NOLINT
264 
265  // Free memory
266  mpfr_clears(r1, a2, nullptr); // NOLINT
267 
268 #ifndef NDEBUG
269  fmt::print("A2 is {}\n", result);
270 #endif
271 
272  return result;
273  }

283  auto try_move(move_tracker::move_type move) -> bool
284  {
285  // Record the trial move
286  m_trial_moves[move_tracker::as_integer(move)]++;
287 
288  // Calculate probability
289  auto a1 = CalculateA1(move);
290 
291  // Make move if random number < probability
292  auto a2 = CalculateA2<3>(move);
293 
294  const auto trial_value = generate_probability();
295  // Convert to Gmpzf because trial_value will be set to 0 when
296  // comparing with a1 and a2!
297  const auto trial = static_cast<double>(trial_value);
298 
299 #ifndef NDEBUG
300  fmt::print("{} called.\n", __PRETTY_FUNCTION__);
301  fmt::print("Trying move.\n");
302  fmt::print("Move type = {}\n", move_tracker::as_integer(move));
303  fmt::print("Trial_value = {}\n", trial_value);
304  fmt::print("Trial = {}\n", trial);
305  fmt::print("A1 = {}\n", a1);
306  fmt::print("A2 = {}\n", a2);
307  fmt::print("A1*A2 = {}\n", a1 * a2);
308  fmt::print("{}\n",
309  (trial <= a1 * a2) ? "Move accepted." : "Move rejected.");
310 #endif
311 
312  if (trial <= a1 * a2)
313  {
314  m_accepted_moves[move_tracker::as_integer(move)]++;
315  return true;
316  }
