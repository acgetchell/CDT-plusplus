247 [[nodiscard]] static auto proposal_probability(
248 Geometry<ManifoldType::dimension> const& geometry,
249 move_tracker::move_type const move) -> mpfr_values::Value
250 {
251 auto const sites = proposal_site_count(geometry, move);
252 if (sites <= 0) { return mpfr_values::zero(); }
253 auto const move_count =
254 mpfr_values::from_integer(move_tracker::NUMBER_OF_3D_MOVES);
255 auto const site_count = mpfr_values::from_integer(sites);
256 auto const denominator = mpfr_values::multiply(move_count, site_count);
257 return mpfr_values::divide(mpfr_values::from_integer(1), denominator);
258 }

263 [[nodiscard]] static auto CalculateA1(
264 Geometry<ManifoldType::dimension> const& current,
265 Geometry<ManifoldType::dimension> const& proposed,
266 move_tracker::move_type const move) -> mpfr_values::Value
267 {
268 auto const forward = proposal_probability(current, move);
269 auto const reverse = proposal_probability(proposed, reverse_move(move));
270 if (mpfr_zero_p(forward.fr()) != 0 || mpfr_zero_p(reverse.fr()) != 0)
271 {
272 throw std::logic_error(
273 "A successful reversible proposal must have nonzero forward and reverse probabilities.");
274 }
275 return mpfr_values::divide(reverse, forward);
276 }

281 [[nodiscard]] auto CalculateA2(
282 Geometry<ManifoldType::dimension> const& current,
283 Geometry<ManifoldType::dimension> const& proposed) const
284 -> mpfr_values::Value
285 {
286 auto const current_action = S3_bulk_action(
287 current.N1_TL, current.N3_31_13, current.N3_22, m_Alpha, m_K, m_Lambda);
288 auto const proposed_action =
289 S3_bulk_action(proposed.N1_TL, proposed.N3_31_13, proposed.N3_22,
290 m_Alpha, m_K, m_Lambda);
291 return mpfr_values::exponential(
292 mpfr_values::subtract(current_action, proposed_action));
293 }

296 [[nodiscard]] auto acceptance_probability(
297 Geometry<ManifoldType::dimension> const& current,
298 Geometry<ManifoldType::dimension> const& proposed,
299 move_tracker::move_type const move) const -> mpfr_values::Value
300 {
301 auto const ratio = mpfr_values::multiply(
302 CalculateA1(current, proposed, move), CalculateA2(current, proposed));
303 auto const one = mpfr_values::from_integer(1);
304 return mpfr_cmp(ratio.fr(), one.fr()) < 0 ? ratio : one;
305 }

308 [[nodiscard]] auto propose_candidate(ManifoldType const& current,
309 move_tracker::move_type const move)
310 -> std::expected<ManifoldType, std::string>
311 {
312 using enum move_tracker::move_type;
313 switch (move)
314 {
315 case TWO_THREE:
316 return ergodic_moves::propose_23_move(current, m_generator);
317 case THREE_TWO:
318 return ergodic_moves::propose_32_move(current, m_generator);
319 case TWO_SIX: return ergodic_moves::propose_26_move(current, m_generator);
320 case SIX_TWO: return ergodic_moves::propose_62_move(current, m_generator);
321 case FOUR_FOUR:
322 return ergodic_moves::propose_44_move(current, m_generator);
323 }
324 return std::unexpected("Unknown 3D Pachner move.\n");
325 }

333 auto attempt_transition(ManifoldType& current,
334 move_tracker::move_type const move,
335 long double const trial_value) -> bool
336 {
337 if (!std::isfinite(trial_value) || trial_value < 0.0L || trial_value > 1.0L)
338 {
339 throw std::invalid_argument("MH trial value must lie in [0, 1].");
340 }
341
342 m_geometry = current.get_geometry();
343 ++m_proposed_moves[move];
344 ++m_attempted_moves[move];
345
346 auto candidate = propose_candidate(current, move);
347 if (!candidate || !ergodic_moves::check_move(current, *candidate, move))
348 {
349 ++m_failed_moves[move];
350 ++m_rejected_moves[move];
351 return false;
352 }
353
354 ++m_succeeded_moves[move];
355 auto const probability =
356 acceptance_probability(m_geometry, candidate->get_geometry(), move);
357 if (mpfr_cmp_ld(probability.fr(), trial_value) >= 0)
358 {
359 swap(*candidate, current);
360 m_geometry = current.get_geometry();
361 ++m_accepted_moves[move];
362 return true;
363 }
364
365 ++m_rejected_moves[move];
366 return false;
367 }

374 auto operator()(ManifoldType const& t_manifold) -> ManifoldType
375 {
376#ifndef NDEBUG
377 spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
378#endif
379
380 fmt::print(
381 "Starting Metropolis-Hastings algorithm in {}+1 dimensions ...\n",
382 ManifoldType::dimension - 1);
383 fmt::print("Effective random seed: {} (stream {}).\n", m_generator.seed(),
384 m_generator.stream());
385
386 m_proposed_moves.reset();
387 m_accepted_moves.reset();
388 m_rejected_moves.reset();
389 m_attempted_moves.reset();
390 m_succeeded_moves.reset();
391 m_failed_moves.reset();
392
393 auto current = t_manifold;
394 initialize(current);
395 std::uniform_real_distribution<long double> acceptance_draw{0.0L, 1.0L};
396
397 fmt::print("Making random moves ...\n");
398 for (auto pass_number = 1; pass_number <= m_passes; ++pass_number)
399 {
400 fmt::print("=== Pass {} ===\n", pass_number);
401 auto const attempts_this_pass = current.N3();
402 for (auto move_attempt = 0; move_attempt < attempts_this_pass;
403 ++move_attempt)
404 {
405 auto const move = move_tracker::generate_random_move_3(m_generator);
406 static_cast<void>(
407 attempt_transition(current, move, acceptance_draw(m_generator)));
408 }
409
410 if (pass_number % m_checkpoint == 0)
411 {
412 print_results();
413 if (m_write_files)
414 {
415 fmt::print("Writing to file.\n");
416 utilities::write_file(current, m_generator.seed(), pass_number);
417 }
418 }
419 }
420
421 fmt::print("=== Run results ===\n");
422 print_results();
423 return current;
424 } // operator()
