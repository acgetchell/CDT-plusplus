194  auto CalculateA2(move_tracker::move_type move) const noexcept
195  {
196  if (dimension == 3)
197  {
198  auto currentS3Action =
199  S3_bulk_action(m_geometry.N1_TL, m_geometry.N3_31_13,
200  m_geometry.N3_22, m_Alpha, m_K, m_Lambda);
201  auto newS3Action = static_cast<Gmpzf>(0);
202  switch (move)
203  {
204  case move_tracker::move_type::TWO_THREE:
205  // A (2,3) move adds a timelike edge
206  // and a (2,2) simplex
207  newS3Action =
208  S3_bulk_action(m_geometry.N1_TL + 1, m_geometry.N3_31_13,
209  m_geometry.N3_22 + 1, m_Alpha, m_K, m_Lambda);
210  break;
211  case move_tracker::move_type::THREE_TWO:
212  // A (3,2) move removes a timelike edge
213  // and a (2,2) simplex
214  newS3Action =
215  S3_bulk_action(m_geometry.N1_TL - 1, m_geometry.N3_31_13,
216  m_geometry.N3_22 - 1, m_Alpha, m_K, m_Lambda);
217  break;
218  case move_tracker::move_type::TWO_SIX:
219  // A (2,6) move adds 2 timelike edges and
220  // 2 (1,3) and 2 (3,1) simplices
221  newS3Action =
222  S3_bulk_action(m_geometry.N1_TL + 2, m_geometry.N3_31_13 + 4,
223  m_geometry.N3_22, m_Alpha, m_K, m_Lambda);
224  break;
225  case move_tracker::move_type::SIX_TWO:
226  // A (6,2) move removes 2 timelike edges and
227  // 2 (1,3) and 2 (3,1) simplices
228  newS3Action =
229  S3_bulk_action(m_geometry.N1_TL - 2, m_geometry.N3_31_13,
230  m_geometry.N3_22 - 4, m_Alpha, m_K, m_Lambda);
231  break;
232  case move_tracker::move_type::FOUR_FOUR:
233  // A (4,4) move changes nothing with respect to the action,
234  // and e^0==1
235 #ifndef NDEBUG
236  fmt::print("A2 is 1\n");
237 #endif
238  return static_cast<double>(1);
239  default:
240  break;
241  }
242 
243  auto exponent = currentS3Action - newS3Action;
244  auto exponent_double = Gmpzf_to_double(exponent);
245 
246  // if exponent > 0 then e^exponent >=1 so according to Metropolis
247  // algorithm return A2=1
248  if (exponent >= 0) { return static_cast<double>(1); }
249 
250  // Set precision for initialization and assignment functions
251  mpfr_set_default_prec(PRECISION);
252 
253  // Initialize for MPFR
254  mpfr_t r1, a2; // NOLINT
255  mpfr_inits2(PRECISION, r1, a2, nullptr); // NOLINT
256 
257  // Set input parameters and constants to mpfr_t equivalents
258  mpfr_init_set_d(r1, exponent_double, // NOLINT
259  MPFR_RNDD); // r1 = exponent
260 
261  // e^exponent
262  mpfr_exp(a2, r1, MPFR_RNDD); // NOLINT
263 
264  // Convert mpfr_t total to Gmpzf result by using Gmpzf(double d)
265  auto result = mpfr_get_d(a2, MPFR_RNDD); // NOLINT
266 
267  // Free memory
268  mpfr_clears(r1, a2, nullptr); // NOLINT
269 
270 #ifndef NDEBUG
271  fmt::print("A2 is {}\n", result);
272 #endif
273 
274  return result;
275  }
