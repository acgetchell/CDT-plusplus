348 [[nodiscard]] static auto proposal_probability(
349 Geometry<ManifoldType::dimension> const& geometry,
350 move_tracker::move_type const move) -> mpfr_values::Value
351 {
352 auto const sites = proposal_site_count(geometry, move);
353 if (sites <= 0) { return mpfr_values::zero(); }
354 auto const move_count =
355 mpfr_values::from_integer(move_tracker::NUMBER_OF_3D_MOVES);
356 auto const site_count = mpfr_values::from_integer(sites);
357 auto const denominator = mpfr_values::multiply(move_count, site_count);
358 return mpfr_values::divide(mpfr_values::from_integer(1), denominator);
359 }

364 [[nodiscard]] static auto CalculateA1(
365 Geometry<ManifoldType::dimension> const& current,
366 Geometry<ManifoldType::dimension> const& proposed,
367 move_tracker::move_type const move) -> mpfr_values::Value
368 {
369 auto const forward = proposal_probability(current, move);
370 auto const reverse = proposal_probability(proposed, reverse_move(move));
371 if (mpfr_zero_p(forward.fr()) != 0 || mpfr_zero_p(reverse.fr()) != 0)
372 {
373 throw std::logic_error(
374 "A successful reversible proposal must have nonzero forward and reverse probabilities.");
375 }
376 return mpfr_values::divide(reverse, forward);
377 }

382 [[nodiscard]] auto CalculateA2(
383 Geometry<ManifoldType::dimension> const& current,
384 Geometry<ManifoldType::dimension> const& proposed) const
385 -> mpfr_values::Value
386 {
387 auto const current_action =
388 s3_action::S3_bulk_action(current.N1_TL, current.N3_31_13,
389 current.N3_22, m_Alpha, m_K, m_Lambda);
390 auto const proposed_action =
391 s3_action::S3_bulk_action(proposed.N1_TL, proposed.N3_31_13,
392 proposed.N3_22, m_Alpha, m_K, m_Lambda);
393 return mpfr_values::exponential(
394 mpfr_values::subtract(current_action, proposed_action));
395 }

398 [[nodiscard]] auto acceptance_probability(
399 Geometry<ManifoldType::dimension> const& current,
400 Geometry<ManifoldType::dimension> const& proposed,
401 move_tracker::move_type const move) const -> mpfr_values::Value
402 {
403 auto const ratio = mpfr_values::multiply(
404 CalculateA1(current, proposed, move), CalculateA2(current, proposed));
405 auto const one = mpfr_values::from_integer(1);
406 return mpfr_cmp(ratio.fr(), one.fr()) < 0 ? ratio : one;
407 }

431 public:
438 auto attempt_transition(ManifoldType& current,
439 move_tracker::move_type const move,
440 long double const trial_value) -> bool
441 {
442 if (!std::isfinite(trial_value) || trial_value < 0.0L ||
443 trial_value > 1.0L)
444 {
445 throw std::invalid_argument("MH trial value must lie in [0, 1].");
446 }
447
448 m_geometry = current.get_geometry();
449 ++m_proposed_moves[move];
450 ++m_attempted_moves[move];
451
452 auto candidate = propose_candidate(current, move);
453 if (!candidate ||
454 !ergodic_moves::detail::check_move(current, *candidate, move))
455 {
456 ++m_failed_moves[move];
457 ++m_rejected_moves[move];
458 record_transition(move, Transition_outcome::CANDIDATE_FAILED);
459 return false;
460 }
461
462 ++m_succeeded_moves[move];
463 auto const probability =
464 acceptance_probability(m_geometry, candidate->get_geometry(), move);
465 if (mpfr_cmp_ld(probability.fr(), trial_value) >= 0)
466 {
467 swap(*candidate, current);
468 m_geometry = current.get_geometry();
469 ++m_accepted_moves[move];
470 record_transition(move, Transition_outcome::ACCEPTED);
471 return true;
472 }
473
474 ++m_rejected_moves[move];
475 record_transition(move, Transition_outcome::REJECTED);
476 return false;
477 }

484 auto operator()(ManifoldType const& t_manifold) -> ManifoldType
485 {
486#ifndef NDEBUG
487 spdlog::debug("{} called.\n", CDT_PRETTY_FUNCTION);
488#endif
489
490 fmt::print(
491 "Starting Metropolis-Hastings algorithm in {}+1 dimensions ...\n",
492 ManifoldType::dimension - 1);
493 fmt::print("Effective random seed: {} (stream {}).\n", m_generator.seed(),
494 m_generator.stream());
495
496 m_proposed_moves.reset();
497 m_accepted_moves.reset();
498 m_rejected_moves.reset();
499 m_attempted_moves.reset();
500 m_succeeded_moves.reset();
501 m_failed_moves.reset();
502 m_transition_trace = 14695981039346656037ULL;
503 m_transition_count = 0;
504
505 auto current = t_manifold;
506 initialize(current);
507 std::uniform_real_distribution<long double> acceptance_draw{0.0L, 1.0L};
508
509 fmt::print("Making random moves ...\n");
510 for (auto pass_number = 1; pass_number <= m_passes; ++pass_number)
511 {
512 fmt::print("=== Pass {} ===\n", pass_number);
513 auto const attempts_this_pass = current.N3();
514 for (auto move_attempt = 0; move_attempt < attempts_this_pass;
515 ++move_attempt)
516 {
517 auto const move = move_tracker::generate_random_move_3(m_generator);
518 static_cast<void>(
519 attempt_transition(current, move, acceptance_draw(m_generator)));
520 }
521
522 if (pass_number % m_checkpoint == 0)
523 {
524 print_results();
525 if (m_write_files)
526 {
527 fmt::print("Writing to file.\n");
528 utilities::write_file(
529 current, reproducibility_metadata(
530 current, utilities::Artifact_kind::CHECKPOINT,
531 pass_number));
532 }
533 }
534 }
535
536 fmt::print("=== Run results ===\n");
537 print_results();
538 return current;
539 } // operator()
