119 static void record_transition(RunStatistics& statistics,
120 move_tracker::move_type const move,
121 Transition_outcome const outcome) noexcept
122 {
123 auto const append = [&statistics](std::uint8_t const value) {
124 statistics.transition_trace ^= value;
125 statistics.transition_trace *= 1099511628211ULL;
126 };
127 append(static_cast<std::uint8_t>(move));
128 append(static_cast<std::uint8_t>(outcome));
129 ++statistics.transition_count;

327 [[nodiscard]] static auto proposal_probability(
328 Geometry<ManifoldType::dimension> const& geometry,
329 move_tracker::move_type const move) -> mpfr_values::Value
330 {
331 auto const sites = proposal_site_count(geometry, move);
332 if (sites <= 0) { return mpfr_values::zero(); }
333 auto const move_count =
334 mpfr_values::from_integer(move_tracker::NUMBER_OF_3D_MOVES);
335 auto const site_count = mpfr_values::from_integer(sites);
336 auto const denominator = mpfr_values::multiply(move_count, site_count);
337 return mpfr_values::divide(mpfr_values::from_integer(1), denominator);
338 }

343 [[nodiscard]] static auto CalculateA1(
344 Geometry<ManifoldType::dimension> const& current,
345 Geometry<ManifoldType::dimension> const& proposed,
346 move_tracker::move_type const move) -> mpfr_values::Value
347 {
348 auto const forward = proposal_probability(current, move);
349 auto const reverse = proposal_probability(proposed, reverse_move(move));
350 if (mpfr_zero_p(forward.fr()) != 0 || mpfr_zero_p(reverse.fr()) != 0)
351 {
352 throw std::logic_error(
353 "A successful reversible proposal must have nonzero forward and reverse probabilities.");
354 }
355 return mpfr_values::divide(reverse, forward);
356 }

361 [[nodiscard]] auto CalculateA2(
362 Geometry<ManifoldType::dimension> const& current,
363 Geometry<ManifoldType::dimension> const& proposed) const
364 -> mpfr_values::Value
365 {
366 auto const current_action =
367 s3_action::S3_bulk_action(current.N1_TL, current.N3_31_13,
368 current.N3_22, m_Alpha, m_K, m_Lambda);
369 auto const proposed_action =
370 s3_action::S3_bulk_action(proposed.N1_TL, proposed.N3_31_13,
371 proposed.N3_22, m_Alpha, m_K, m_Lambda);
372 return mpfr_values::exponential(
373 mpfr_values::subtract(current_action, proposed_action));
374 }

377 [[nodiscard]] auto acceptance_probability(
378 Geometry<ManifoldType::dimension> const& current,
379 Geometry<ManifoldType::dimension> const& proposed,
380 move_tracker::move_type const move) const -> mpfr_values::Value
381 {
382 auto const ratio = mpfr_values::multiply(
383 CalculateA1(current, proposed, move), CalculateA2(current, proposed));
384 auto const one = mpfr_values::from_integer(1);
385 return mpfr_cmp(ratio.fr(), one.fr()) < 0 ? ratio : one;
386 }

433 auto attempt_transition(ManifoldType& current,
434 CommandResults& command_results,
435 RunStatistics& statistics,
436 move_tracker::move_type const move,
437 long double const trial_value) -> bool
438 {
439 if (!std::isfinite(trial_value) || trial_value < 0.0L ||
440 trial_value > 1.0L)
441 {
442 throw std::invalid_argument("MH trial value must lie in [0, 1].");
443 }
444
445 statistics.geometry = current.get_geometry();
446 ++statistics.proposed[move];
447 ++command_results.attempted[move];
448
449 auto candidate = propose_candidate(current, move);
450 if (!candidate ||
451 !ergodic_moves::detail::check_move(current, *candidate, move))
452 {
453 ++command_results.failed[move];
454 ++statistics.rejected[move];
455 record_transition(statistics, move,
456 Transition_outcome::CANDIDATE_FAILED);
457 return false;
458 }
459
460 ++command_results.succeeded[move];
461 auto const probability = acceptance_probability(
462 statistics.geometry, candidate->get_geometry(), move);
463 if (mpfr_cmp_ld(probability.fr(), trial_value) >= 0)
464 {
465 swap(*candidate, current);
466 statistics.geometry = current.get_geometry();
467 ++statistics.accepted[move];
468 record_transition(statistics, move, Transition_outcome::ACCEPTED);
469 return true;
470 }
471
472 ++statistics.rejected[move];
473 record_transition(statistics, move, Transition_outcome::REJECTED);
474 return false;
475 }

477 [[nodiscard]] auto execute_pass(ManifoldType current,
478 RunStatistics statistics,
479 Int_precision const attempts) -> PassResult
480 {
481 auto command_results = CommandResults{};
482 std::uniform_real_distribution<long double> acceptance_draw{0.0L, 1.0L};
483 for (auto move_attempt = Int_precision{0}; move_attempt < attempts;
484 ++move_attempt)
485 {
486 auto const move = move_tracker::generate_random_move_3(m_generator);
487 static_cast<void>(attempt_transition(current, command_results,
488 statistics, move,
489 acceptance_draw(m_generator)));
490 }
491 return {.manifold = std::move(current),
492 .command_results = std::move(command_results),
493 .strategy_state = std::move(statistics)};
494 }
