173 auto CalculateA1(move_tracker::move_type move) const noexcept
174 {
175 auto all_moves = m_proposed_moves.total();
176 auto this_move = m_proposed_moves[move];
177 // Set precision for initialization and assignment functions
178 mpfr_set_default_prec(PRECISION);
179
180 // Initialize for MPFR
181 mpfr_t r_1;
182 mpfr_t r_2;
183 mpfr_t a_1;
184 mpfr_inits2(PRECISION, a_1, nullptr); // NOLINT
185
186 mpfr_init_set_si(r_1, this_move, MPFR_RNDD); // r_1 = this_move NOLINT
187 mpfr_init_set_si(r_2, all_moves, MPFR_RNDD); // r_2 = total_moves NOLINT
188
189 // The result
190 mpfr_div(a_1, r_1, r_2, MPFR_RNDD); // a_1 = r_1/r_2 NOLINT
191
192 // Convert mpfr_t total to Gmpzf result by using Gmpzf(double d)
193 auto result = mpfr_get_ld(a_1, MPFR_RNDD); // NOLINT
194
195 // Free memory
196 mpfr_clears(r_1, r_2, a_1, nullptr); // NOLINT
197
198#ifndef NDEBUG
199 spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
200 spdlog::trace("Total proposed moves = {}\n", all_moves);
201 spdlog::trace("A1 is {}\n", result);
202#endif
203
204 return result;
205 } // CalculateA1()

211 auto CalculateA2(move_tracker::move_type const move) const noexcept
212 {
213 auto currentS3Action =
214 S3_bulk_action(m_geometry.N1_TL, m_geometry.N3_31_13, m_geometry.N3_22,
215 m_Alpha, m_K, m_Lambda);
216 auto newS3Action = static_cast<Gmpzf>(0);
217 switch (move)
218 {
219 case move_tracker::move_type::TWO_THREE:
220 // A (2,3) move adds a timelike edge
221 // and a (2,2) simplex
222 newS3Action =
223 S3_bulk_action(m_geometry.N1_TL + 1, m_geometry.N3_31_13,
224 m_geometry.N3_22 + 1, m_Alpha, m_K, m_Lambda);
225 break;
226 case move_tracker::move_type::THREE_TWO:
227 // A (3,2) move removes a timelike edge
228 // and a (2,2) simplex
229 newS3Action =
230 S3_bulk_action(m_geometry.N1_TL - 1, m_geometry.N3_31_13,
231 m_geometry.N3_22 - 1, m_Alpha, m_K, m_Lambda);
232 break;
233 case move_tracker::move_type::TWO_SIX:
234 // A (2,6) move adds 2 timelike edges and
235 // 2 (1,3) and 2 (3,1) simplices
236 newS3Action =
237 S3_bulk_action(m_geometry.N1_TL + 2, m_geometry.N3_31_13 + 4,
238 m_geometry.N3_22, m_Alpha, m_K, m_Lambda);
239 break;
240 case move_tracker::move_type::SIX_TWO:
241 // A (6,2) move removes 2 timelike edges and
242 // 2 (1,3) and 2 (3,1) simplices
243 newS3Action =
244 S3_bulk_action(m_geometry.N1_TL - 2, m_geometry.N3_31_13 - 4,
245 m_geometry.N3_22, m_Alpha, m_K, m_Lambda);
246 break;
247 case move_tracker::move_type::FOUR_FOUR:
248 // A (4,4) move changes nothing with respect to the action,
249 // and e^0==1
250#ifndef NDEBUG
251 spdlog::trace("A2 is 1\n");
252#endif
253 return static_cast<long double>(1);
254 default: break;
255 }
256
257 auto exponent = currentS3Action - newS3Action;
258 auto exponent_double = utilities::Gmpzf_to_double(exponent);
259
260 // if exponent > 0 then e^exponent >=1 so according to Metropolis
261 // algorithm return A2=1
262 if (exponent >= 0) { return static_cast<long double>(1); }
263
264 // Set precision for initialization and assignment functions
265 mpfr_set_default_prec(PRECISION);
266
267 // Initialize for MPFR
268 mpfr_t r_1;
269 mpfr_t a_2;
270 mpfr_inits2(PRECISION, a_2, nullptr); // NOLINT
271
272 // Set input parameters and constants to mpfr_t equivalents
273 mpfr_init_set_ld(r_1, exponent_double, // NOLINT
274 MPFR_RNDD); // r1 = exponent
275
276 // e^exponent
277 mpfr_exp(a_2, r_1, MPFR_RNDD); // NOLINT
278
279 // Convert mpfr_t total to Gmpzf result by using Gmpzf(double d)
280 auto result = mpfr_get_ld(a_2, MPFR_RNDD); // NOLINT
281
282 // Free memory
283 mpfr_clears(r_1, a_2, nullptr); // NOLINT
284
285#ifndef NDEBUG
286 spdlog::trace("A2 is {}\n", result);
287#endif
288
289 return result;
290 } // CalculateA2()

298 auto try_move(move_tracker::move_type const move) -> bool
299 {
300 // Record the proposed move
301 ++m_proposed_moves[as_integer(move)];
302
303 // Calculate probability
304 auto a_1 = CalculateA1(move);
305
306 // Make move if random number < probability
307 auto a_2 = CalculateA2(move);
308
309 if (auto const trial_value = utilities::generate_probability();
310 trial_value <= a_1 * a_2)
311 {
312#ifndef NDEBUG
313 spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
314 spdlog::trace("Trying move.\n");
315 spdlog::trace("Move type = {}\n", as_integer(move));
316 spdlog::trace("Trial_value = {}\n", trial_value);
317 spdlog::trace("A1 = {}\n", a_1);
318 spdlog::trace("A2 = {}\n", a_2);
319 spdlog::trace("A1*A2 = {}\n", a_1 * a_2);
320 spdlog::trace("{}\n", trial_value <= a_1 * a_2 ? "Move accepted."
321 : "Move rejected.");
322#endif
323 // Accept the move
324 ++m_accepted_moves[as_integer(move)];
325 return true;
326 }
327
328 // Reject the move
329 ++m_rejected_moves[as_integer(move)];
330 return false;
331
332 } // try_move()

340 [[nodiscard]] auto initialize(ManifoldType t_manifold)
341 -> std::optional<MoveCommand<ManifoldType>>
342 try
343 {
344 MoveCommand command(t_manifold);
345 fmt::print("Making initial moves ...\n");
346
347 // Make a move of each type
348 for (auto move = 0; move < move_tracker::NUMBER_OF_3D_MOVES; ++move)
349 {
350#ifndef NDEBUG
351 spdlog::trace("Making move {} ...\n", move);
352#endif
353 command.enqueue(move_tracker::as_move(move));
354 ++m_proposed_moves[move];
355 ++m_accepted_moves[move];
356 }
357
358 // Execute the initial moves
359 command.execute();
360 command.print_successful();
361 command.print_errors();
362
363 // Update attempted, succeeded, and failed moves from MoveCommand
364 m_attempted_moves += command.get_attempted();
365 m_succeeded_moves += command.get_succeeded();
366 m_failed_moves += command.get_failed();
367
368 // Reset the move counters
369 command.reset_counters();
370
371 // print initial results
372 auto initial_results = command.get_results();
373 initial_results.print();
374 initial_results.print_details();
375
376 return command;
377 }
378 catch (std::system_error const& SystemError)
379 {
380 spdlog::debug("Metropolis initialization failed with {} ... exiting.\n",
381 SystemError.what());
382 spdlog::trace("{}\n", SystemError.code().message());
383 return std::nullopt;
384 }

393 auto operator()(ManifoldType const& t_manifold) -> ManifoldType
394 {
395#ifndef NDEBUG
396 spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
397#endif
398
399 fmt::print(
400 "Starting Metropolis-Hastings algorithm in {}+1 dimensions ...\n",
401 ManifoldType::dimension - 1);
402
403 m_proposed_moves.reset();
404 m_accepted_moves.reset();
405 m_rejected_moves.reset();
406 m_attempted_moves.reset();
407 m_succeeded_moves.reset();
408 m_failed_moves.reset();
409
410 auto initialized = initialize(t_manifold);
411 auto command = initialized ? std::move(*initialized)
412 : MoveCommand<ManifoldType>{t_manifold};
413
414 fmt::print("Making random moves ...\n");
415
416 // Loop through m_passes
417 for (auto pass_number = 1; pass_number <= m_passes; ++pass_number)
418 {
419 fmt::print("=== Pass {} ===\n", pass_number);
420 auto total_simplices_this_pass = command.get_const_results().N3();
421 // Attempt a random move per simplex
422 for (auto move_attempt = 0; move_attempt < total_simplices_this_pass;
423 ++move_attempt)
424 {
425 // Pick a move to attempt
426
427 if (auto move = move_tracker::generate_random_move_3(); try_move(move))
428 {
429 command.enqueue(move);
430 }
431 } // Ends loop through CurrentTotalSimplices
432
433 // Do the moves
434 command.execute();
435
436 // Update attempted and failed moves
437 this->m_attempted_moves += command.get_attempted();
438 this->m_succeeded_moves += command.get_succeeded();
439 this->m_failed_moves += command.get_failed();
440 command.reset_counters();
441
442 // Do stuff on checkpoint
443 if (pass_number % m_checkpoint == 0)
444 {
445 fmt::print("=== Pass {} ===\n", pass_number);
446 print_results();
447 if (m_write_files)
448 {
449 fmt::print("Writing to file.\n");
450 utilities::write_file(command.get_results());
451 }
452 }
453 } // Ends loop through m_passes
454
455 // output results
456 fmt::print("=== Run results ===\n");
457 print_results();
458 return command.get_results();
459 } // operator()
