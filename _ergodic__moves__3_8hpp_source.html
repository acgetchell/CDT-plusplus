196  [[nodiscard]] inline auto do_26_move(Manifold3& t_manifold)
197  {
198 #ifndef NDEBUG
199  fmt::print("{} called.\n", __PRETTY_FUNCTION__);
200 #endif
201  auto one_three = t_manifold.get_triangulation().get_one_three();
202  // Shuffle the container to pick a random sequence of (1,3) cells to try
203  std::shuffle(one_three.begin(), one_three.end(), make_random_generator());
204  for (auto& bottom : one_three)
205  {
206  if (auto neighboring_31_index = find_26_move(bottom);
207  neighboring_31_index)
208  {
209 #ifndef NDEBUG
210  fmt::print("neighboring_31_index is {}\n", *neighboring_31_index);
211 #endif
212  Cell_handle top = bottom->neighbor(*neighboring_31_index);
213  // Calculate the common face with respect to the bottom cell
214  auto common_face_index = std::numeric_limits<int>::max();
215  Expects(bottom->has_neighbor(top, common_face_index));
216 
217  // Get indices of vertices of common face with respect to bottom cell
218  // A face is denoted by the index of the opposite vertex
219  // Thus, the indices of the vertices of the face are all other indices
220  // except the common_face_index
221  // CGAL uses bitwise operations, e.g.
222  // auto i1 = (common_face_index + 1) & 3;
223  // auto i2 = (common_face_index + 2) & 3;
224  // auto i3 = (common_face_index + 3) & 3;
225  // We use % 4 which is equivalent and doesn't trigger clang-tidy
226  auto i1 = (common_face_index + 1) % 4;
227  auto i2 = (common_face_index + 2) % 4;
228  auto i3 = (common_face_index + 3) % 4;
229 
230  // Get vertices of common face from indices
231  auto v1 = bottom->vertex(i1);
232  auto v2 = bottom->vertex(i2);
233  auto v3 = bottom->vertex(i3);
234 
235  // Timeslice of vertices should be same
236  Expects(v1->info() == v2->info() && v2->info() == v3->info());
237 
238  // Do the (2,6) move
239  // Insert new vertex
240  Vertex_handle v_center =
241  t_manifold.triangulation().delaunay().tds().insert_in_facet(
242  bottom, *neighboring_31_index);
243 
244  // Checks
245  std::vector<Cell_handle> incident_cells;
246  t_manifold.triangulation().delaunay().tds().incident_cells(
247  v_center, std::back_inserter(incident_cells));
248  // the (2,6) center vertex should be bounded by 6 simplices
249  Expects(incident_cells.size() == 6);
250  // Each incident cell should be combinatorially and geometrically valid
251  for (auto const& cell : incident_cells)
252  {
253  Expects(t_manifold.get_triangulation().get_delaunay().tds().is_valid(
254  cell));
255  }
256 
257  // Now assign a geometric point to the center vertex
258  auto center_point =
259  CGAL::centroid(v1->point(), v2->point(), v3->point());
260 #ifndef NDEBUG
261  std::cout << "Center point is: " << center_point << "\n";
262 // fmt::print("Center point is: {}\n", center_point);
263 #endif
264  v_center->set_point(center_point);
265 
266  // Assign a timevalue to the new vertex
267  auto timevalue = v1->info();
268  v_center->info() = timevalue;
269 
270 #ifndef NDEBUG
271  if (t_manifold.is_vertex(v_center))
272  { fmt::print("It's a vertex in the TDS.\n"); }
273  else
274  {
275  fmt::print("It's not a vertex in the TDS.\n");
276  }
277  std::cout << "Spacelike face timevalue is " << timevalue << "\n";
278  std::cout << "Inserted vertex (" << v_center->point()
279  << ") with timevalue " << v_center->info() << "\n";
280 #endif
281 
282  // Final check
283  Expects(t_manifold.get_triangulation().get_delaunay().tds().is_valid(
284  v_center, true, 1));
285 
286  return t_manifold;
287  }
288  // Try next cell
289 #ifndef NDEBUG
290  fmt::print("Cell not insertable.\n");
291 #endif
292  }
293  // We've run out of (1,3) simplices to try
294  throw std::domain_error("No (2,6) move possible.");

307  [[nodiscard]] inline auto is_62_movable(Manifold3& manifold,
308  Vertex_handle const& candidate)
309  {
310  Expects(manifold.dim() == 3); // Precondition of incident_cells()
311  Expects(manifold.is_vertex(candidate));
312 
313  auto incident_edges = manifold.degree(candidate);
314 
315  // We must have 5 incident edges to have 6 incident cells
316  if (incident_edges != 5)
317  {
318 #ifdef DETAILED_DEBUGGING
319  fmt::print("Vertex has {} incident edges/vertices.\n", incident_edges);
320 #endif
321  return false;
322  }
323 
324  // Obtain all incident cells
325  auto incident_cells = manifold.incident_cells(candidate);
326  // We must have 6 cells incident to the vertex to make a (6,2) move
327  if (incident_cells.size() != 6)
328  {
329 #ifdef DETAILED_DEBUGGING
330  fmt::print(
331  "Vertex has {} incident edges/vertices and {} incident cells.\n",
332  incident_edges, incident_cells.size());
333 #endif
334  return false;
335  }
336 
337  auto incident_31 = manifold.get_triangulation().filter_cells(
338  incident_cells, Cell_type::THREE_ONE);
339  auto incident_22 = manifold.get_triangulation().filter_cells(
340  incident_cells, Cell_type::TWO_TWO);
341  auto incident_13 = manifold.get_triangulation().filter_cells(
342  incident_cells, Cell_type::ONE_THREE);
343 
344  // All cells should be classified
345  if ((incident_13.size() + incident_22.size() + incident_31.size()) != 6)
346  {
347  fmt::print("Some incident cells on this vertex need to be fixed.\n");
348  }
349 
350 #ifndef NDEBUG
351  fmt::print(
352  "Vertex has {} incident edges/vertices and {} incident (3,1) simplices "
353  "and {} incident (2,2) simplices and {} incident (1,3) simplices.\n",
354  incident_edges, incident_31.size(), incident_22.size(),
355  incident_13.size());
356  manifold.get_triangulation().print_cells(incident_cells);
357 #endif
358  return ((incident_31.size() == 3) && (incident_22.empty()) &&
359  (incident_13.size() == 3));

361  } // find_62_moves()
