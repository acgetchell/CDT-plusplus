43  {

203  [[nodiscard]] inline auto do_26_move(
204  manifolds::Manifold3& t_manifold) noexcept -> Expected
205  {
206 #ifndef NDEBUG
207  spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
208 #endif
209  auto one_three = t_manifold.get_triangulation().get_one_three();
210  // Shuffle the container to pick a random sequence of (1,3) cells to try
211  std::shuffle(one_three.begin(), one_three.end(),
212  utilities::make_random_generator());
213  for (auto const& bottom : one_three)
214  {
215  if (auto neighboring_31_index = find_adjacent_31_cell(bottom);
216  neighboring_31_index)
217  {
218 #ifndef NDEBUG
219  spdlog::trace("neighboring_31_index is {}.\n", *neighboring_31_index);
220 #endif
221  Cell_handle_t<3> const top =
222  bottom->neighbor(neighboring_31_index.value());
223  // Calculate the common face with respect to the bottom cell
224  auto common_face_index = std::numeric_limits<int>::max();
225  if (!bottom->has_neighbor(top, common_face_index))
226  {
227  std::string msg = "Bottom cell does not have a neighbor.\n";
228 #ifndef NDEBUG
229  spdlog::trace(msg);
230 #endif
231  return tl::make_unexpected(msg);
232  }
233 
234  // Get indices of vertices of common face with respect to bottom cell
235  // A face is denoted by the index of the opposite vertex
236  // Thus, the indices of the vertices of the face are all other indices
237  // except the common_face_index
238  // CGAL uses bitwise operations like (common_face_index +1) & 3
239  // We use % 4 which is equivalent and doesn't trigger clang-tidy
240  auto const i1 = (common_face_index + 1) % 4;
241  auto const i2 = (common_face_index + 2) % 4;
242  auto const i3 = (common_face_index + 3) % 4;
243 
244  // Get vertices of common face from indices
245  auto const v1 = bottom->vertex(i1);
246  auto const v2 = bottom->vertex(i2);
247  auto const v3 = bottom->vertex(i3);
248 
249  // Timeslice of vertices should be same
250  if (v1->info() != v2->info() || v2->info() != v3->info())
251  {
252  std::string msg = "Vertices have different timeslices.\n";
253 #ifndef NDEBUG
254  spdlog::trace(msg);
255 #endif
256  return tl::make_unexpected(msg);
257  }
258 
259  // Do the (2,6) move
260  // Insert new vertex
261  Vertex_handle_t<3> v_center =
262  t_manifold.triangulation().delaunay().tds().insert_in_facet(
263  bottom, *neighboring_31_index);
264 
265  // Checks
266  std::vector<Cell_handle_t<3>> incident_cells;
267  t_manifold.triangulation().delaunay().tds().incident_cells(
268  v_center, std::back_inserter(incident_cells));
269  // the (2,6) center vertex should be bounded by 6 simplices
270  if (incident_cells.size() != 6)
271  {
272  std::string msg = "Center vertex is not bounded by 6 simplices.\n";
273 #ifndef NDEBUG
274  spdlog::trace(msg);
275 #endif
276  return tl::make_unexpected(msg);
277  }
278 
279  // Each incident cell should be combinatorially and geometrically valid
280  if (auto check_cells =
281  std::all_of(incident_cells.begin(), incident_cells.end(),
282  [&t_manifold](auto const& cell) {
283  return t_manifold.get_triangulation()
284  .get_delaunay()
285  .tds()
286  .is_cell(cell);
287  });
288  !check_cells)
289  {
290  std::string msg = "A cell is invalid.\n";
291 #ifndef NDEBUG
292  spdlog::trace(msg);
293 #endif
294  return tl::make_unexpected(msg);
295  }
296 
297  // Now assign a geometric point to the center vertex
298  auto center_point =
299  CGAL::centroid(v1->point(), v2->point(), v3->point());
300 #ifndef NDEBUG
301  spdlog::trace("Center point is: ({}).\n", center_point);
302 #endif
303  v_center->set_point(center_point);
304 
305  // Assign a timevalue to the new vertex
306  auto timevalue = v1->info();
307  v_center->info() = timevalue;
308 
309 #ifndef NDEBUG
310  if (t_manifold.is_vertex(v_center))
311  {
312  spdlog::trace("It's a vertex in the TDS.\n");
313  }
314  else
315  {
316  spdlog::trace("It's not a vertex in the TDS.\n");
317  }
318  spdlog::trace("Spacelike face timevalue is {}.\n", timevalue);
319  spdlog::trace("Inserted vertex ({}) with timevalue {}.\n",
320  v_center->point(), v_center->info());
321 #endif
322 
323  // Final checks
324  // is_valid() checks for combinatorial and geometric validity
325  // and outputs to std::cerr
326  if (!t_manifold.get_triangulation().get_delaunay().tds().is_valid(
327  v_center, true, 1))
328  {
329  std::string msg = "v_center is invalid.\n";
330 #ifndef NDEBUG
331  spdlog::trace(msg);
332 #endif
333  return tl::make_unexpected(msg);
334  }
335 
336  return t_manifold;
337  }
338  // Try next cell
339 #ifndef NDEBUG
340  spdlog::debug("Cell not insertable.\n");
341 #endif
342  }
343  // We've run out of (1,3) simplices to try
344  std::string msg = "No (2,6) move possible.\n";
345  spdlog::warn(msg);
346  return tl::make_unexpected(msg);
347  } // do_26_move()

359  [[nodiscard]] inline auto is_62_movable(
360  manifolds::Manifold3 const& manifold,
361  Vertex_handle_t<3> const& candidate) noexcept -> bool
362  {
363  if (manifold.dimensionality() != 3)
364  {
365 #ifndef NDEBUG
366  spdlog::trace("Manifold is not 3-dimensional.\n");
367 #endif
368  return false;
369  }
370 
371  if (!manifold.is_vertex(candidate))
372  {
373 #ifndef NDEBUG
374  spdlog::trace("Candidate is not a vertex.\n");
375 #endif
376  return false;
377  }
378 
379  // We must have 5 incident edges to have 6 incident cells
380  if (auto incident_edges = manifold.degree(candidate);
381  incident_edges != 5) // NOLINT
382  {
383 #ifndef NDEBUG
384  spdlog::trace("Vertex has {} incident edges.\n", incident_edges);
385 #endif
386  return false;
387  }
388 
389  // Obtain all incident cells
390  auto const incident_cells = manifold.incident_cells(candidate);
391  // We must have 6 cells incident to the vertex to make a (6,2) move
392  if (incident_cells.size() != 6) // NOLINT
393  {
394 #ifndef NDEBUG
395  spdlog::trace("Vertex has {} incident cells.\n", incident_cells.size());
396 #endif
397  return false;
398  }
399 
400  // Run until all vertices are fixed
401  while (foliated_triangulations::fix_vertices<3>(
402  manifold.get_triangulation().get_delaunay(),
403  manifold.get_triangulation().initial_radius(),
404  manifold.get_triangulation().foliation_spacing()))
405  {
406  spdlog::warn("Fixing vertices found by is_62_movable().\n");
407  }
408  // Run until all cells fixed or 50 passes
409  for (auto passes = 1; passes < 51; ++passes) // NOLINT
410  {
411  if (foliated_triangulations::fix_cells<3>(
412  manifold.get_triangulation().get_delaunay()))
413  {
414  spdlog::warn("Fixing cells found by is_62_movable() pass {}.\n",
415  passes);
416  }
417  }
418 
419  auto const incident_31 = foliated_triangulations::filter_cells<3>(
420  incident_cells, Cell_type::THREE_ONE);
421  auto const incident_22 = foliated_triangulations::filter_cells<3>(
422  incident_cells, Cell_type::TWO_TWO);
423  auto const incident_13 = foliated_triangulations::filter_cells<3>(
424  incident_cells, Cell_type::ONE_THREE);
425 
426  // All cells should be classified
427  if ((incident_13.size() + incident_22.size() + incident_31.size())
428  != 6) // NOLINT
429  {
430  spdlog::warn("Some incident cells on this vertex need to be fixed.\n");
431  }
432 
433 #ifndef NDEBUG
434  spdlog::trace(
435  "Vertex has {} incident cells with {} incident (3,1) simplices and {} "
436  "incident (2,2) simplices and {} incident (1,3) simplices.\n",
437  incident_cells.size(), incident_31.size(), incident_22.size(),
438  incident_13.size());
439  foliated_triangulations::debug_print_cells<3>(std::span{incident_cells});
440 #endif
441  return ((incident_31.size() == 3) && (incident_22.empty())
442  && (incident_13.size() == 3));
443 
444  } // find_62_moves()

502  manifolds::Manifold3 const& t_manifold,
503  Edge_handle_t<3> const& t_edge_candidate) noexcept
504  -> std::optional<std::vector<Cell_handle_t<3>>>
505  {

507

512

515  incident_cells.emplace_back(circulator++);

519 #endif
520 
521  if (incident_cells.size() == 4) { return incident_cells; }

523  return std::nullopt;
