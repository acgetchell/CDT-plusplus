202  [[nodiscard]] inline auto do_26_move(
203  manifolds::Manifold3& t_manifold) noexcept -> Expected
204  {
205 #ifndef NDEBUG
206  spdlog::debug("{} called.\n", __PRETTY_FUNCTION__);
207 #endif
208  static auto constexpr INCIDENT_CELLS_FOR_6_2_MOVE = 6;
209  auto one_three = t_manifold.get_triangulation().get_one_three();
210  // Shuffle the container to pick a random sequence of (1,3) cells to try
211  std::shuffle(one_three.begin(), one_three.end(),
212  utilities::make_random_generator());
213  for (auto const& bottom : one_three)
214  {
215  if (auto neighboring_31_index = find_adjacent_31_cell(bottom);
216  neighboring_31_index)
217  {
218 #ifndef NDEBUG
219  spdlog::trace("neighboring_31_index is {}.\n", *neighboring_31_index);
220 #endif
221  Cell_handle_t<3> const top =
222  bottom->neighbor(neighboring_31_index.value());
223  // Calculate the common face with respect to the bottom cell
224  auto common_face_index = std::numeric_limits<int>::max();
225  if (!bottom->has_neighbor(top, common_face_index))
226  {
227  std::string msg = "Bottom cell does not have a neighbor.\n";
228 #ifndef NDEBUG
229  spdlog::trace(msg);
230 #endif
231  return tl::make_unexpected(msg);
232  }
233 
234  // Get indices of vertices of common face with respect to bottom cell
235  // A face is denoted by the index of the opposite vertex
236  // Thus, the indices of the vertices of the face are all other indices
237  // except the common_face_index
238  // CGAL uses bitwise operations like (common_face_index +1) & 3
239  // We use % 4 which is equivalent and doesn't trigger clang-tidy
240  auto const i_1 = (common_face_index + 1) % 4;
241  auto const i_2 = (common_face_index + 2) % 4;
242  auto const i_3 = (common_face_index + 3) % 4;
243 
244  // Get vertices of common face from indices
245  auto const v_1 = bottom->vertex(i_1);
246  auto const v_2 = bottom->vertex(i_2);
247  auto const v_3 = bottom->vertex(i_3);
248 
249  // Timeslice of vertices should be same
250  if (v_1->info() != v_2->info() || v_2->info() != v_3->info())
251  {
252  std::string msg = "Vertices have different timeslices.\n";
253 #ifndef NDEBUG
254  spdlog::trace(msg);
255 #endif
256  return tl::make_unexpected(msg);
257  }
258 
259  // Do the (2,6) move
260  // Insert new vertex
261  Vertex_handle_t<3> v_center =
262  t_manifold.triangulation().delaunay().tds().insert_in_facet(
263  bottom, *neighboring_31_index);
264 
265  // Checks
266  std::vector<Cell_handle_t<3>> incident_cells;
267  t_manifold.triangulation().delaunay().tds().incident_cells(
268  v_center, std::back_inserter(incident_cells));
269  // the (2,6) center vertex should be bounded by 6 simplices
270  if (incident_cells.size() != INCIDENT_CELLS_FOR_6_2_MOVE)
271  {
272  std::string msg = "Center vertex is not bounded by 6 simplices.\n";
273 #ifndef NDEBUG
274  spdlog::trace(msg);
275 #endif
276  return tl::make_unexpected(msg);
277  }
278 
279  // Each incident cell should be combinatorially and geometrically valid
280  if (auto check_cells =
281  std::all_of(incident_cells.begin(), incident_cells.end(),
282  [&t_manifold](auto const& cell) {
283  return t_manifold.get_triangulation()
284  .get_delaunay()
285  .tds()
286  .is_cell(cell);
287  });
288  !check_cells)
289  {
290  std::string msg = "A cell is invalid.\n";
291 #ifndef NDEBUG
292  spdlog::trace(msg);
293 #endif
294  return tl::make_unexpected(msg);
295  }
296 
297  // Now assign a geometric point to the center vertex
298  auto center_point =
299  CGAL::centroid(v_1->point(), v_2->point(), v_3->point());
300 #ifndef NDEBUG
301  spdlog::trace("Center point is: ({}).\n", center_point);
302 #endif
303  v_center->set_point(center_point);
304 
305  // Assign a timevalue to the new vertex
306  auto timevalue = v_1->info();
307  v_center->info() = timevalue;
308 
309 #ifndef NDEBUG
310  if (t_manifold.is_vertex(v_center))
311  {
312  spdlog::trace("It's a vertex in the TDS.\n");
313  }
314  else { spdlog::trace("It's not a vertex in the TDS.\n"); }
315  spdlog::trace("Spacelike face timevalue is {}.\n", timevalue);
316  spdlog::trace("Inserted vertex ({}) with timevalue {}.\n",
317  v_center->point(), v_center->info());
318 #endif
319 
320  // Final checks
321  // is_valid() checks for combinatorial and geometric validity
322  // and outputs to std::cerr
323  if (!t_manifold.get_triangulation().get_delaunay().tds().is_valid(
324  v_center, true, 1))
325  {
326  std::string msg = "v_center is invalid.\n";
327 #ifndef NDEBUG
328  spdlog::trace(msg);
329 #endif
330  return tl::make_unexpected(msg);
331  }
332 
333  return t_manifold;
334  }
335  // Try next cell
336 #ifndef NDEBUG
337  spdlog::debug("Cell not insertable.\n");
338 #endif
339  }
340  // We've run out of (1,3) simplices to try
341  std::string msg = "No (2,6) move possible.\n";
342  spdlog::warn(msg);
343  return tl::make_unexpected(msg);
344  } // do_26_move()

356  [[nodiscard]] inline auto is_62_movable(
357  manifolds::Manifold3 const& manifold,
358  Vertex_handle_t<3> const& candidate) noexcept -> bool
359  {
360  if (manifold.dimensionality() != 3)
361  {
362 #ifndef NDEBUG
363  spdlog::trace("Manifold is not 3-dimensional.\n");
364 #endif
365  return false;
366  }
367 
368  if (!manifold.is_vertex(candidate))
369  {
370 #ifndef NDEBUG
371  spdlog::trace("Candidate is not a vertex.\n");
372 #endif
373  return false;
374  }
375 
376  // We must have 5 incident edges to have 6 incident cells
377  if (auto incident_edges = manifold.degree(candidate);
378  incident_edges != 5) // NOLINT
379  {
380 #ifndef NDEBUG
381  spdlog::trace("Vertex has {} incident edges.\n", incident_edges);
382 #endif
383  return false;
384  }
385 
386  // Obtain all incident cells
387  auto const incident_cells = manifold.incident_cells(candidate);
388  // We must have 6 cells incident to the vertex to make a (6,2) move
389  if (incident_cells.size() != 6) // NOLINT
390  {
391 #ifndef NDEBUG
392  spdlog::trace("Vertex has {} incident cells.\n", incident_cells.size());
393 #endif
394  return false;
395  }
396 
397  // Run until all vertices are fixed
398  while (foliated_triangulations::fix_vertices<3>(
399  manifold.get_triangulation().get_delaunay(), manifold.initial_radius(),
400  manifold.foliation_spacing()))
401  {
402  spdlog::warn("Fixing vertices found by is_62_movable().\n");
403  }
404  // Run until all cells fixed or 10 passes
405  for (auto passes = 1; passes < 11; ++passes) // NOLINT
406  {
407  if (foliated_triangulations::fix_cells<3>(
408  manifold.get_triangulation().get_delaunay()))
409  {
410  spdlog::warn("Fixing cells found by is_62_movable() pass {}.\n",
411  passes);
412  }
413  }
414 
415  auto const incident_31 = foliated_triangulations::filter_cells<3>(
416  incident_cells, Cell_type::THREE_ONE);
417  auto const incident_22 = foliated_triangulations::filter_cells<3>(
418  incident_cells, Cell_type::TWO_TWO);
419  auto const incident_13 = foliated_triangulations::filter_cells<3>(
420  incident_cells, Cell_type::ONE_THREE);
421 
422  // All cells should be classified
423  if ((incident_13.size() + incident_22.size() + incident_31.size()) !=
424  6) // NOLINT
425  {
426  spdlog::warn("Some incident cells on this vertex need to be fixed.\n");
427  }
428 
429 #ifndef NDEBUG
430  spdlog::trace(
431  "Vertex has {} incident cells with {} incident (3,1) simplices and {} "
432  "incident (2,2) simplices and {} incident (1,3) simplices.\n",
433  incident_cells.size(), incident_31.size(), incident_22.size(),
434  incident_13.size());
435  foliated_triangulations::debug_print_cells<3>(std::span{incident_cells});
436 #endif
437  return ((incident_31.size() == 3) && (incident_22.empty()) &&
438  (incident_13.size() == 3));
439 
440  } // find_62_moves()

535  !b_4->is_valid())

537  return false;
538  }

543  Cell_handle_t<3> n_3 = b_2->neighbor(b_2->index(pivot_from_vertex_1));
544  Cell_handle_t<3> n_4 = b_2->neighbor(b_2->index(pivot_from_vertex_2));

547  Cell_handle_t<3> n_7 = b_4->neighbor(b_4->index(pivot_from_vertex_1));
548  Cell_handle_t<3> n_8 = b_4->neighbor(b_4->index(pivot_from_vertex_2));
549 
550  // Next, delete the old cells
551  t_manifold.triangulation().delaunay().tds().delete_cell(b_1);
552  t_manifold.triangulation().delaunay().tds().delete_cell(b_2);
553  t_manifold.triangulation().delaunay().tds().delete_cell(b_3);
554  t_manifold.triangulation().delaunay().tds().delete_cell(b_4);

558  Cell_handle_t<3> a_1 =

561  Cell_handle_t<3> a_2 =
562  t_manifold.triangulation().delaunay().tds().create_cell(
563  top, pivot_from_vertex_2, pivot_to_vertex_1, pivot_to_vertex_2);
564  Cell_handle_t<3> a_3 =
565  t_manifold.triangulation().delaunay().tds().create_cell(
566  bottom, pivot_from_vertex_1, pivot_to_vertex_1, pivot_to_vertex_2);
567  Cell_handle_t<3> a_4 =
568  t_manifold.triangulation().delaunay().tds().create_cell(
569  bottom, pivot_from_vertex_2, pivot_to_vertex_1, pivot_to_vertex_2);
570

572  // https://doc.cgal.org/latest/TDS_3/classTriangulationDataStructure__3_1_1Cell.html#ace214d6e7a06de2976adbbc18c90a0d1
573  a_1->set_neighbors(n_1, n_4, a_2, a_3);
574  a_2->set_neighbors(n_2, n_3, a_1, a_4);
575  a_3->set_neighbors(n_5, n_8, a_4, a_1);

581  t_manifold.triangulation().delaunay().tds().reorient();
582 
583  return a_1->is_valid() && a_2->is_valid() && a_3->is_valid() &&
584  a_4->is_valid();
585 
586  // Move succeeded
587  // return true;
588  }
