78 Edge_handle const& edge) noexcept -> bool
79 {

81 auto const valid_index = [](Int_precision const index) {

92 [[nodiscard]] inline auto finite_incident_cells(
93 Delaunay const& triangulation, Edge_handle const& edge)
94 -> std::optional<Cell_container>
95 {
96 if (!is_well_formed_edge(edge) || triangulation.dimension() != 3 ||
97 !triangulation.tds().is_edge(edge.first, edge.second, edge.third))
98 {
99 return std::nullopt;
100 }
101
102 auto circulator = triangulation.incident_cells(edge, edge.first);
103 Cell_container incident_cells;
104 do
105 { // NOLINT(cppcoreguidelines-avoid-do-while)
106 if (triangulation.is_infinite(circulator)) { return std::nullopt; }
107 incident_cells.emplace_back(circulator);
108 }
109 while (++circulator != edge.first);
110 return incident_cells;
111 }

193 template <typename Container, std::uniform_random_bit_generator Generator,
194 typename Comparator>
195 [[nodiscard]] inline auto canonical_random_element(Container& candidates,
196 Generator& generator,
197 Comparator comparator)
198 -> std::optional<typename Container::value_type>
199 {
200 if (candidates.empty()) { return std::nullopt; }
201 std::uniform_int_distribution<std::size_t> distribution{
202 0, candidates.size() - 1};
203 auto const index = distribution(generator);
204 auto const nth = candidates.begin() +
205 static_cast<typename Container::difference_type>(index);
206 std::ranges::nth_element(candidates, nth, comparator);
207 return candidates[index];
208 }

295 [[nodiscard]] inline auto detail::try_23_move(Delaunay& triangulation,
296 Cell_handle const& to_be_moved)
297 -> bool
298 {
299 if (to_be_moved == nullptr || triangulation.dimension() != 3 ||
300 !triangulation.tds().is_cell(to_be_moved) ||
301 !foliated_triangulations::is_cell_type_correct<3>(to_be_moved) ||
302 foliated_triangulations::expected_cell_type<3>(to_be_moved) !=
303 Cell_type::TWO_TWO)
304 {
305 return false;
306 }
307 std::array facet_indices{0, 1, 2, 3};
308 std::ranges::sort(facet_indices, [&](auto const left, auto const right) {
309 return detail::point_less(to_be_moved->vertex(left)->point(),
310 to_be_moved->vertex(right)->point());
311 });
312 auto flipped = false;
313 // Try every facet of the (2,2) cell
314 for (auto const i : facet_indices)
315 {
316 auto const neighbor = to_be_moved->neighbor(i);
317 if (triangulation.is_infinite(neighbor)) { continue; }
318
319 auto const neighbor_type =
320 foliated_triangulations::expected_cell_type<3>(neighbor);
321 if (!foliated_triangulations::is_cell_type_correct<3>(neighbor) ||
322 (neighbor_type != Cell_type::THREE_ONE &&
323 neighbor_type != Cell_type::ONE_THREE))
324 {
325 continue;
326 }
327
328 // A causal (2,3) move must replace the facet with a timelike edge.
329 // CGAL also permits topological flips that create a spacelike edge.
330 auto const mirror_index = neighbor->index(to_be_moved);
331 auto const first_time =
332 static_cast<long long>(to_be_moved->vertex(i)->info());
333 auto const second_time =
334 static_cast<long long>(neighbor->vertex(mirror_index)->info());
335 auto const time_difference = first_time > second_time
336 ? first_time - second_time
337 : second_time - first_time;
338 if (time_difference != 1) { continue; }
339
340 if (triangulation.flip(to_be_moved, i))
341 {
342 flipped = true;
343 break;
344 }
345 }
346 return flipped;
347 } // try_23_move

366 template <std::uniform_random_bit_generator Generator>
367 [[nodiscard]] inline auto do_23_move(Manifold const& t_manifold,
368 Generator& generator) -> Expected
369 {
370 Delaunay triangulation{t_manifold.delaunay_snapshot()};
371 auto two_two = foliated_triangulations::filter_cells<3>(
372 foliated_triangulations::collect_cells<3>(triangulation),
373 Cell_type::TWO_TWO);
374 detail::canonicalize(two_two);
375 // Shuffle the container to create a random sequence of (2,2) cells
376 std::ranges::shuffle(two_two, generator);
377 // Try a (2,3) move on successive cells in the sequence
378 if (std::ranges::any_of(two_two, [&](auto& cell) {
379 return detail::try_23_move(triangulation, cell);
380 }))
381 {
382 return detail::make_manifold(std::move(triangulation), t_manifold);
383 }
384 // We've run out of (2,2) cells
385 std::string const msg = "No (2,3) move possible.\n";
386 return std::unexpected(msg);
387 }

412 [[nodiscard]] inline auto is_32_movable(Delaunay const& triangulation,
413 Edge_handle const& candidate)
414 -> bool
415 {
416 auto const incident_cells =
417 finite_incident_cells(triangulation, candidate);
418 if (!incident_cells || incident_cells->size() != 3) { return false; }
419
420 auto const first_time = static_cast<long long>(
421 candidate.first->vertex(candidate.second)->info());
422 auto const second_time = static_cast<long long>(
423 candidate.first->vertex(candidate.third)->info());
424 auto const time_difference = first_time > second_time
425 ? first_time - second_time
426 : second_time - first_time;
427 if (time_difference != 1) { return false; }
428
429 if (!std::ranges::all_of(*incident_cells, [](auto const& cell) {
430 return foliated_triangulations::is_cell_type_correct<3>(cell);
431 }))
432 {
433 return false;
434 }
435
436 auto const cell_type_count = [&](Cell_type const type) {
437 return std::ranges::count_if(*incident_cells, [&](auto const& cell) {
438 return foliated_triangulations::expected_cell_type<3>(cell) == type;
439 });
440 };
441 auto const incident_31 = cell_type_count(Cell_type::THREE_ONE);
442 auto const incident_22 = cell_type_count(Cell_type::TWO_TWO);
443 auto const incident_13 = cell_type_count(Cell_type::ONE_THREE);
444 return incident_22 == 2 && ((incident_31 == 1 && incident_13 == 0) ||
445 (incident_31 == 0 && incident_13 == 1));
446 }

525 [[nodiscard]] inline auto detail::find_adjacent_31_cell(
526 Cell_handle const& t_cell) -> std::optional<int>
527 {
528 if (t_cell == nullptr ||
529 !foliated_triangulations::is_cell_type_correct<3>(t_cell) ||
530 foliated_triangulations::expected_cell_type<3>(t_cell) !=
531 Cell_type::ONE_THREE)
532 {
533 return std::nullopt;
534 }
535 std::vector<int> candidates;
536 for (auto i = 0; i < 4; ++i)
537 {
538 auto const neighbor = t_cell->neighbor(i);
539 if (foliated_triangulations::is_cell_type_correct<3>(neighbor) &&
540 foliated_triangulations::expected_cell_type<3>(neighbor) ==
541 Cell_type::THREE_ONE)
542 {
543 candidates.emplace_back(i);
544 }
545 }
546 if (candidates.empty()) { return std::nullopt; }
547 std::ranges::sort(candidates, [&](auto const left, auto const right) {
548 auto const left_points =
549 detail::canonical_cell_points(t_cell->neighbor(left));
550 auto const right_points =
551 detail::canonical_cell_points(t_cell->neighbor(right));
552 return std::lexicographical_compare(
553 left_points.begin(), left_points.end(), right_points.begin(),
554 right_points.end(), detail::point_less);
555 });
556 return candidates.front();
557 } // find_26_move()

570 // Internal validation seam used to test rejection after mutation.
571 template <std::uniform_random_bit_generator Generator,
572 typename Post_mutation_validator>
573 requires std::predicate<Post_mutation_validator&, Delaunay const&>
574 [[nodiscard]] inline auto detail::do_26_move_impl(
575 Manifold const& t_manifold, Generator& generator,
576 bool const only_first_site,
577 Post_mutation_validator post_mutation_validator) -> Expected
578 {
579 static auto constexpr INCIDENT_CELLS_FOR_6_2_MOVE = 6;
580 Delaunay triangulation{t_manifold.delaunay_snapshot()};
581 auto one_three = foliated_triangulations::filter_cells<3>(
582 foliated_triangulations::collect_cells<3>(triangulation),
583 Cell_type::ONE_THREE);
584 if (only_first_site)
585 {
586 auto const candidate =
587 detail::canonical_random_element(one_three, generator);
588 if (!candidate)
589 {
590 return std::unexpected("No (2,6) proposal site is available.\n");
591 }
592 one_three = {*candidate};
593 }
594 else
595 {
596 detail::canonicalize(one_three);
597 // Shuffle the container to pick a random sequence of (1,3) cells to
598 // try.
599 std::ranges::shuffle(one_three, generator);
600 }
601 for (auto const& bottom : one_three)
602 {
603 if (auto neighboring_31_index = find_adjacent_31_cell(bottom);
604 neighboring_31_index)
605 {
606 Cell_handle const top = bottom->neighbor(neighboring_31_index.value());
607 // Calculate the common face with respect to the bottom cell
608 auto common_face_index = std::numeric_limits<int>::max();
609 if (!bottom->has_neighbor(top, common_face_index))
610 {
611 std::string const msg = "Bottom cell does not have a neighbor.\n";
612 return std::unexpected(msg);
613 }
614
615 // Get indices of vertices of common face with respect to bottom cell
616 // A face is denoted by the index of the opposite vertex
617 // Thus, the indices of the vertices of the face are all other indices
618 // except the common_face_index
619 // CGAL uses bitwise operations like (common_face_index +1) & 3
620 // We use % 4 which is equivalent and doesn't trigger clang-tidy
621 auto const i_1 = (common_face_index + 1) % 4;
622 auto const i_2 = (common_face_index + 2) % 4;
623 auto const i_3 = (common_face_index + 3) % 4;
624
625 // Get vertices of common face from indices
626 auto const v_1 = bottom->vertex(i_1);
627 auto const v_2 = bottom->vertex(i_2);
628 auto const v_3 = bottom->vertex(i_3);
629
630 // Timeslice of vertices should be same
631 if (v_1->info() != v_2->info() || v_2->info() != v_3->info())
632 {
633 std::string const msg = "Vertices have different timeslices.\n";
634 return std::unexpected(msg);
635 }
636
637 // Do the (2,6) move
638 // Insert new vertex
639 Vertex_handle const v_center =
640 triangulation.tds().insert_in_facet(bottom, *neighboring_31_index);
641
642 // Checks
643 Cell_container incident_cells;
644 triangulation.tds().incident_cells(v_center,
645 std::back_inserter(incident_cells));
646 // the (2,6) center vertex should be bounded by 6 simplices
647 if (incident_cells.size() != INCIDENT_CELLS_FOR_6_2_MOVE)
648 {
649 std::string const msg =
650 "Center vertex is not bounded by 6 simplices.\n";
651 return std::unexpected(msg);
652 }
653
654 // Each incident cell should be combinatorially and geometrically
655 // valid
656 if (auto check_cells =
657 std::ranges::all_of(incident_cells,
658 [&triangulation](auto const& cell) {
659 return triangulation.tds().is_cell(cell);
660 });
661 !check_cells)
662 {
663 std::string const msg = "A cell is invalid.\n";
664 return std::unexpected(msg);
665 }
666
667 // Now assign a geometric point to the center vertex
668 std::array face_points{v_1->point(), v_2->point(), v_3->point()};
669 std::ranges::sort(face_points, detail::point_less);
670 auto const center_point =
671 CGAL::centroid(face_points[0], face_points[1], face_points[2]);
672 v_center->set_point(center_point);
673
674 // Assign a timevalue to the new vertex
675 auto timevalue = v_1->info();
676 v_center->info() = timevalue;
677
678 // Final checks
679 // is_valid() checks for combinatorial and geometric validity
680 if (!post_mutation_validator(
681 static_cast<Delaunay const&>(triangulation)) ||
682 !triangulation.tds().is_valid(v_center, true, 1))
683 {
684 std::string const msg = "v_center is invalid.\n";
685 return std::unexpected(msg);
686 }
687
688 return make_manifold(std::move(triangulation), t_manifold);
689 }
690 }
691 // We've run out of (1,3) simplices to try
692 std::string const msg = "No (2,6) move possible.\n";
693 return std::unexpected(msg);
694 } // do_26_move_impl()

743 [[nodiscard]] inline auto detail::is_62_movable(
744 Delaunay const& triangulation, Vertex_handle const& candidate) -> bool
745 {
746 if (triangulation.dimension() != 3) { return false; }
747
748 if (!triangulation.tds().is_vertex(candidate)) { return false; }
749
750 // We must have 5 incident edges to have 6 incident cells
751 if (auto incident_edges = triangulation.degree(candidate);
752 incident_edges != 5) // NOLINT
753 {
754 return false;
755 }
756
757 // Obtain all incident cells
758 Cell_container incident_cells;
759 triangulation.tds().incident_cells(candidate,
760 std::back_inserter(incident_cells));
761
762 // We must have 6 cells incident to the vertex to make a (6,2) move
763 if (incident_cells.size() != 6) // NOLINT
764 {
765 return false;
766 }
767
768 // Check that none of the incident cells are infinite
769 for (auto const& cell : incident_cells)
770 {
771 if (triangulation.is_infinite(cell)) { return false; }
772 }
773
774 auto const cell_type_count = [&](Cell_type const type) {
775 return std::ranges::count_if(incident_cells, [&](auto const& cell) {
776 return foliated_triangulations::expected_cell_type<3>(cell) == type;
777 });
778 };
779 auto const incident_31 = cell_type_count(Cell_type::THREE_ONE);
780 auto const incident_22 = cell_type_count(Cell_type::TWO_TWO);
781 auto const incident_13 = cell_type_count(Cell_type::ONE_THREE);
782
783 // All cells should be causally classified and carry matching metadata.
784 if (incident_13 + incident_22 + incident_31 != 6 || // NOLINT
785 !std::ranges::all_of(incident_cells, [](auto const& cell) {
786 return foliated_triangulations::is_cell_type_correct<3>(cell);
787 }))
788 {
789 return false;
790 }
791
792 return incident_31 == 3 && incident_22 == 0 && incident_13 == 3;
793
794 } // find_62_moves()

808 // Internal validation seam used to test rejection after mutation.
809 template <std::uniform_random_bit_generator Generator,
810 typename Post_mutation_validator>
811 requires std::predicate<Post_mutation_validator&, Delaunay const&>
812 [[nodiscard]] inline auto detail::try_62_move_impl(
813 Delaunay const& source_triangulation,
814 Vertex_handle const source_candidate, Generator& generator,
815 Post_mutation_validator post_mutation_validator)
816 -> std::optional<Delaunay>
817 {
818 if (!source_triangulation.tds().is_vertex(source_candidate) ||
819 source_triangulation.is_infinite(source_candidate))
820 {
821 return std::nullopt;
822 }
823
824 auto const candidate_point = source_candidate->point();
825 Delaunay triangulation{source_triangulation};
826 auto const copied_candidate =
827 foliated_triangulations::find_vertex<3>(triangulation, candidate_point);
828 if (!copied_candidate) { return std::nullopt; }
829
830 auto const candidate = *copied_candidate;
831 auto& tds = triangulation.tds();
832 auto const old_cells = triangulation.number_of_finite_cells();
833 auto const old_vertices = triangulation.number_of_vertices();
834
835 Edge_container incident_edges;
836 triangulation.finite_incident_edges(candidate,
837 std::back_inserter(incident_edges));
838 detail::canonicalize(incident_edges);
839 std::ranges::shuffle(incident_edges, generator);
840
841 auto const is_timelike = [](Edge_handle const& edge) {
842 auto const first_time = edge.first->vertex(edge.second)->info();
843 auto const second_time = edge.first->vertex(edge.third)->info();
844 return first_time != second_time;
845 };
846 auto const flipped = std::ranges::any_of(
847 incident_edges,
848 [&](auto const& edge) { return is_timelike(edge) && tds.flip(edge); });
849 if (!flipped || tds.degree(candidate) != 4 || !tds.is_valid())
850 {
851 return std::nullopt;
852 }
853
854 tds.remove_from_maximal_dimension_simplex(candidate);
855 if (!post_mutation_validator(static_cast<Delaunay const&>(triangulation)) ||
856 !tds.is_valid() ||
857 triangulation.number_of_finite_cells() + 4 != old_cells ||
858 triangulation.number_of_vertices() + 1 != old_vertices)
859 {
860 return std::nullopt;
861 }
862
863 for (auto const cell : triangulation.finite_cell_handles())
864 {
865 auto const type = foliated_triangulations::expected_cell_type<3>(cell);
866 if (type == Cell_type::ACAUSAL || type == Cell_type::UNCLASSIFIED)
867 {
868 return std::nullopt;
869 }
870 cell->info() = static_cast<Int_precision>(type);
871 }
872
873 return triangulation;
874 } // try_62_move_impl()

1022 // Internal validation seam used to test rejection after mutation.
1023 template <typename Post_mutation_validator>
1024 requires std::predicate<Post_mutation_validator&, Delaunay const&>
1025 [[nodiscard]] inline auto detail::bistellar_flip_impl(
1026 Delaunay const& source_triangulation, Edge_handle const source_edge,
1027 Vertex_handle const source_top, Vertex_handle const source_bottom,
1028 Post_mutation_validator post_mutation_validator)
1029 -> std::optional<Delaunay>
1030 {
1031 if (!is_well_formed_edge(source_edge) || source_top == nullptr ||
1032 source_bottom == nullptr ||
1033 !source_triangulation.tds().is_edge(
1034 source_edge.first, source_edge.second, source_edge.third) ||
1035 !source_triangulation.tds().is_vertex(source_top) ||
1036 !source_triangulation.tds().is_vertex(source_bottom) ||
1037 source_triangulation.is_infinite(source_top) ||
1038 source_triangulation.is_infinite(source_bottom))
1039 {
1040 return std::nullopt;
1041 }
1042
1043 // CGAL copy construction duplicates its vertices and cells, invalidating
1044 // all handles from the source triangulation. Capture stable point values,
1045 // then resolve the corresponding handles in the private copy before
1046 // attempting the mutation.
1047 auto const pivot_from_1_point =
1048 source_edge.first->vertex(source_edge.second)->point();
1049 auto const pivot_from_2_point =
1050 source_edge.first->vertex(source_edge.third)->point();
1051 auto const top_point = source_top->point();
1052 auto const bottom_point = source_bottom->point();
1053
1054 Delaunay triangulation{source_triangulation};
1055 auto const pivot_from_1_handle = foliated_triangulations::find_vertex<3>(
1056 triangulation, pivot_from_1_point);
1057 auto const pivot_from_2_handle = foliated_triangulations::find_vertex<3>(
1058 triangulation, pivot_from_2_point);
1059 auto const top_handle =
1060 foliated_triangulations::find_vertex<3>(triangulation, top_point);
1061 auto const bottom_handle =
1062 foliated_triangulations::find_vertex<3>(triangulation, bottom_point);
1063 if (!pivot_from_1_handle || !pivot_from_2_handle || !top_handle ||
1064 !bottom_handle)
1065 {
1066 return std::nullopt;
1067 }
1068
1069 auto const pivot_from_1 = *pivot_from_1_handle;
1070 auto const pivot_from_2 = *pivot_from_2_handle;
1071 auto const top = *top_handle;
1072 auto const bottom = *bottom_handle;
1073
1074 Cell_handle copied_edge_cell = nullptr;
1075 int copied_edge_first_index{};
1076 int copied_edge_second_index{};
1077 if (!triangulation.is_edge(pivot_from_1, pivot_from_2, copied_edge_cell,
1078 copied_edge_first_index,
1079 copied_edge_second_index))
1080 {
1081 return std::nullopt;
1082 }
1083 Edge_handle const edge{copied_edge_cell, copied_edge_first_index,
1084 copied_edge_second_index};
1085
1086 auto const incident_cells = incident_cells_from_edge(triangulation, edge);
1087 if (!incident_cells || incident_cells->size() != 4 || top == bottom ||
1088 top == pivot_from_1 || top == pivot_from_2 || bottom == pivot_from_1 ||
1089 bottom == pivot_from_2 ||
1090 std::ranges::any_of(*incident_cells, [&](auto const& cell) {
1091 return triangulation.is_infinite(cell) || !cell->is_valid();
1092 }))
1093 {
1094 return std::nullopt;
1095 }
1096
1097 auto const incident_count = [&](Vertex_handle const vertex) {
1098 return std::ranges::count_if(
1099 *incident_cells,
1100 [&](Cell_handle const cell) { return cell->has_vertex(vertex); });
1101 };
1102 if (incident_count(top) != 2 || incident_count(bottom) != 2)
1103 {
1104 return std::nullopt;
1105 }
1106
1107 // A 3D 4-to-4 bistellar move is the composition of CGAL's checked TDS
1108 // 2-to-3 facet flip and checked 3-to-2 edge flip. The TDS operations are
1109 // intentional: either geometrically checked Triangulation_3 facet flip
1110 // can reject the transient, degenerate intermediate even when the final
1111 // 4-to-4 configuration is valid. Flipping a facet that contains the old
1112 // edge and either boundary vertex creates the new diagonal; the old edge
1113 // then has degree three and can be removed by the second flip.
1114 Cell_handle boundary_facet_cell = nullptr;
1115 int pivot_from_1_index{};
1116 int pivot_from_2_index{};
1117 int boundary_index{};
1118 if (!triangulation.is_facet(pivot_from_1, pivot_from_2, bottom,
1119 boundary_facet_cell, pivot_from_1_index,
1120 pivot_from_2_index, boundary_index))
1121 {
1122 return std::nullopt;
1123 }
1124 auto constexpr cell_index_sum = 0 + 1 + 2 + 3;
1125 auto const boundary_facet_index = cell_index_sum - pivot_from_1_index -
1126 pivot_from_2_index - boundary_index;
1127 if (!triangulation.tds().flip(
1128 Delaunay::Facet{boundary_facet_cell, boundary_facet_index}))
1129 {
1130 return std::nullopt;
1131 }
1132
1133 Cell_handle old_edge_cell = nullptr;
1134 int old_edge_first_index{};
1135 int old_edge_second_index{};
1136 if (!triangulation.is_edge(pivot_from_1, pivot_from_2, old_edge_cell,
1137 old_edge_first_index, old_edge_second_index) ||
1138 !triangulation.tds().flip(Delaunay::Edge{
1139 old_edge_cell, old_edge_first_index, old_edge_second_index}))
1140 {
1141 return std::nullopt;
1142 }
1143
1144 if (!post_mutation_validator(static_cast<Delaunay const&>(triangulation)) ||
1145 !triangulation.tds().is_valid())
1146 {
1147 return std::nullopt;
1148 }
1149
1150 for (auto const cell : triangulation.finite_cell_handles())
1151 {
1152 cell->info() = static_cast<Int_precision>(
1153 foliated_triangulations::expected_cell_type<3>(cell));
1154 }
1155
1156 return triangulation;
1157 } // bistellar_flip_impl()

1233 template <std::uniform_random_bit_generator Generator>
1234 [[nodiscard]] inline auto do_44_move(Manifold const& t_manifold,
1235 Generator& generator) -> Expected
1236 {
1237 auto triangulation = t_manifold.delaunay_snapshot();
1238 auto spacelike_edges = foliated_triangulations::filter_edges<3>(
1239 foliated_triangulations::collect_edges<3>(triangulation), false);
1240 detail::canonicalize(spacelike_edges);
1241 // Shuffle the container to pick a random sequence of edges to try
1242 std::ranges::shuffle(spacelike_edges, generator);
1243 for (auto const& edge : spacelike_edges)
1244 {
1245 // Obtain all incident cells
1246 if (auto const incident_cells =
1247 detail::find_bistellar_flip_location(triangulation, edge);
1248 incident_cells)
1249 {
1250 // Get edge vertices
1251 auto const& v1 = edge.first->vertex(edge.second);
1252 auto const& v2 = edge.first->vertex(edge.third);
1253
1254 // Find top and bottom vertices
1255 Vertex_handle top = nullptr;
1256 Vertex_handle bottom = nullptr;
1257
1258 // Analyze cells to find the top and bottom vertices
1259 for (auto const& cell : *incident_cells)
1260 {
1261 for (int i = 0; i < 4; ++i)
1262 {
1263 auto vertex = cell->vertex(i);
1264 if (vertex != v1 && vertex != v2)
1265 {
1266 // Use timevalue to determine if it's a top or bottom vertex
1267 if (top == nullptr || detail::vertex_precedes(top, vertex))
1268 {
1269 top = vertex;
1270 }
1271 if (bottom == nullptr || detail::vertex_precedes(vertex, bottom))
1272 {
1273 bottom = vertex;
1274 }
1275 }
1276 }
1277 }
1278
1279 // Try the bistellar flip
1280 if (auto flipped_triangulation =
1281 detail::bistellar_flip(triangulation, edge, top, bottom);
1282 flipped_triangulation.has_value())
1283 {
1284 return detail::make_manifold(std::move(*flipped_triangulation),
1285 t_manifold);
1286 }
1287
1288 // The flip failed; continue trying other edges.
1289 }
1290 // Try next edge
1291 }
1292 // We've run out of edges to try
1293 std::string const msg = "No (4,4) move possible.\n";
1294 return std::unexpected(msg);
1295 } // do_44_move()
