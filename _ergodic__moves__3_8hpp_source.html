90 [[nodiscard]] inline auto finite_incident_cells(
91 Delaunay const& triangulation, Edge_handle const& edge)
92 -> std::optional<Cell_container>
93 {
94 if (!is_well_formed_edge(edge) || triangulation.dimension() != 3 ||
95 !triangulation.tds().is_edge(edge.first, edge.second, edge.third))
96 {
97 return std::nullopt;
98 }
99
100 auto circulator = triangulation.incident_cells(edge, edge.first);
101 Cell_container incident_cells;
102 do
103 { // NOLINT(cppcoreguidelines-avoid-do-while)
104 if (triangulation.is_infinite(circulator)) { return std::nullopt; }
105 incident_cells.emplace_back(circulator);
106 }
107 while (++circulator != edge.first);
108 return incident_cells;
109 }

139 [[nodiscard]] inline auto try_23_move(Delaunay& triangulation,
140 Cell_handle const& to_be_moved) -> bool
141 {
142 if (to_be_moved == nullptr || triangulation.dimension() != 3 ||
143 !triangulation.tds().is_cell(to_be_moved) ||
144 !foliated_triangulations::is_cell_type_correct<3>(to_be_moved) ||
145 foliated_triangulations::expected_cell_type<3>(to_be_moved) !=
146 Cell_type::TWO_TWO)
147 {
148 return false;
149 }
150 auto flipped = false;
151 // Try every facet of the (2,2) cell
152 for (auto i = 0; i < 4; ++i)
153 {
154 auto const neighbor = to_be_moved->neighbor(i);
155 if (triangulation.is_infinite(neighbor)) { continue; }
156
157 auto const neighbor_type =
158 foliated_triangulations::expected_cell_type<3>(neighbor);
159 if (!foliated_triangulations::is_cell_type_correct<3>(neighbor) ||
160 (neighbor_type != Cell_type::THREE_ONE &&
161 neighbor_type != Cell_type::ONE_THREE))
162 {
163 continue;
164 }
165
166 // A causal (2,3) move must replace the facet with a timelike edge.
167 // CGAL also permits topological flips that create a spacelike edge.
168 auto const mirror_index = neighbor->index(to_be_moved);
169 auto const first_time =
170 static_cast<long long>(to_be_moved->vertex(i)->info());
171 auto const second_time =
172 static_cast<long long>(neighbor->vertex(mirror_index)->info());
173 auto const time_difference = first_time > second_time
174 ? first_time - second_time
175 : second_time - first_time;
176 if (time_difference != 1) { continue; }
177
178 if (triangulation.flip(to_be_moved, i))
179 {
180 flipped = true;
181 break;
182 }
183 }
184 return flipped;
185 } // try_23_move

205 [[nodiscard]] inline auto do_23_move(Manifold const& t_manifold,
206 Generator& generator) -> Expected
207 {
208 Delaunay triangulation{t_manifold.delaunay_snapshot()};
209 auto two_two = foliated_triangulations::filter_cells<3>(
210 foliated_triangulations::collect_cells<3>(triangulation),
211 Cell_type::TWO_TWO);
212 // Shuffle the container to create a random sequence of (2,2) cells
213 std::ranges::shuffle(two_two, generator);
214 // Try a (2,3) move on successive cells in the sequence
215 if (std::ranges::any_of(two_two, [&](auto& cell) {
216 return try_23_move(triangulation, cell);
217 }))
218 {
219 return detail::make_manifold(std::move(triangulation), t_manifold);
220 }
221 // We've run out of (2,2) cells
222 std::string const msg = "No (2,3) move possible.\n";
223 return std::unexpected(msg);

249 [[nodiscard]] inline auto is_32_movable(Delaunay const& triangulation,
250 Edge_handle const& candidate)
251 -> bool
252 {
253 auto const incident_cells =
254 finite_incident_cells(triangulation, candidate);
255 if (!incident_cells || incident_cells->size() != 3) { return false; }
256
257 auto const first_time = static_cast<long long>(
258 candidate.first->vertex(candidate.second)->info());
259 auto const second_time = static_cast<long long>(
260 candidate.first->vertex(candidate.third)->info());
261 auto const time_difference = first_time > second_time
262 ? first_time - second_time
263 : second_time - first_time;
264 if (time_difference != 1) { return false; }
265
266 if (!std::ranges::all_of(*incident_cells, [](auto const& cell) {
267 return foliated_triangulations::is_cell_type_correct<3>(cell);
268 }))
269 {
270 return false;
271 }
272
273 auto const cell_type_count = [&](Cell_type const type) {
274 return std::ranges::count_if(*incident_cells, [&](auto const& cell) {
275 return foliated_triangulations::expected_cell_type<3>(cell) == type;
276 });
277 };
278 auto const incident_31 = cell_type_count(Cell_type::THREE_ONE);
279 auto const incident_22 = cell_type_count(Cell_type::TWO_TWO);
280 auto const incident_13 = cell_type_count(Cell_type::ONE_THREE);
281 return incident_22 == 2 && ((incident_31 == 1 && incident_13 == 0) ||
282 (incident_31 == 0 && incident_13 == 1));
283 }

393 // Internal validation seam used to test rejection after mutation.
394 template <std::uniform_random_bit_generator Generator,
395 typename Post_mutation_validator>
396 requires std::predicate<Post_mutation_validator&, Delaunay const&>
397 [[nodiscard]] inline auto detail::do_26_move_impl(
398 Manifold const& t_manifold, Generator& generator,
399 bool const only_first_site,
400 Post_mutation_validator post_mutation_validator) -> Expected
401 {
402 static auto constexpr INCIDENT_CELLS_FOR_6_2_MOVE = 6;
403 Delaunay triangulation{t_manifold.delaunay_snapshot()};
404 auto one_three = foliated_triangulations::filter_cells<3>(
405 foliated_triangulations::collect_cells<3>(triangulation),
406 Cell_type::ONE_THREE);
407 if (only_first_site)
408 {
409 auto const candidate = detail::random_element(one_three, generator);
410 if (!candidate)
411 {
412 return std::unexpected("No (2,6) proposal site is available.\n");
413 }
414 one_three = {*candidate};
415 }
416 else
417 {
418 // Shuffle the container to pick a random sequence of (1,3) cells to
419 // try.
420 std::ranges::shuffle(one_three, generator);
421 }
422 for (auto const& bottom : one_three)
423 {
424 if (auto neighboring_31_index = find_adjacent_31_cell(bottom);
425 neighboring_31_index)
426 {
427 Cell_handle const top = bottom->neighbor(neighboring_31_index.value());
428 // Calculate the common face with respect to the bottom cell
429 auto common_face_index = std::numeric_limits<int>::max();
430 if (!bottom->has_neighbor(top, common_face_index))
431 {
432 std::string const msg = "Bottom cell does not have a neighbor.\n";
433 return std::unexpected(msg);
434 }
435
436 // Get indices of vertices of common face with respect to bottom cell
437 // A face is denoted by the index of the opposite vertex
438 // Thus, the indices of the vertices of the face are all other indices
439 // except the common_face_index
440 // CGAL uses bitwise operations like (common_face_index +1) & 3
441 // We use % 4 which is equivalent and doesn't trigger clang-tidy
442 auto const i_1 = (common_face_index + 1) % 4;
443 auto const i_2 = (common_face_index + 2) % 4;
444 auto const i_3 = (common_face_index + 3) % 4;
445
446 // Get vertices of common face from indices
447 auto const v_1 = bottom->vertex(i_1);
448 auto const v_2 = bottom->vertex(i_2);
449 auto const v_3 = bottom->vertex(i_3);
450
451 // Timeslice of vertices should be same
452 if (v_1->info() != v_2->info() || v_2->info() != v_3->info())
453 {
454 std::string const msg = "Vertices have different timeslices.\n";
455 return std::unexpected(msg);
456 }
457
458 // Do the (2,6) move
459 // Insert new vertex
460 Vertex_handle const v_center =
461 triangulation.tds().insert_in_facet(bottom, *neighboring_31_index);
462
463 // Checks
464 Cell_container incident_cells;
465 triangulation.tds().incident_cells(v_center,
466 std::back_inserter(incident_cells));
467 // the (2,6) center vertex should be bounded by 6 simplices
468 if (incident_cells.size() != INCIDENT_CELLS_FOR_6_2_MOVE)
469 {
470 std::string const msg =
471 "Center vertex is not bounded by 6 simplices.\n";
472 return std::unexpected(msg);
473 }
474
475 // Each incident cell should be combinatorially and geometrically
476 // valid
477 if (auto check_cells =
478 std::ranges::all_of(incident_cells,
479 [&triangulation](auto const& cell) {
480 return triangulation.tds().is_cell(cell);
481 });
482 !check_cells)
483 {
484 std::string const msg = "A cell is invalid.\n";
485 return std::unexpected(msg);
486 }
487
488 // Now assign a geometric point to the center vertex
489 auto const center_point =
490 CGAL::centroid(v_1->point(), v_2->point(), v_3->point());
491 v_center->set_point(center_point);
492
493 // Assign a timevalue to the new vertex
494 auto timevalue = v_1->info();
495 v_center->info() = timevalue;
496
497 // Final checks
498 // is_valid() checks for combinatorial and geometric validity
499 if (!post_mutation_validator(
500 static_cast<Delaunay const&>(triangulation)) ||
501 !triangulation.tds().is_valid(v_center, true, 1))
502 {
503 std::string const msg = "v_center is invalid.\n";
504 return std::unexpected(msg);
505 }
506
507 return make_manifold(std::move(triangulation), t_manifold);
508 }
509 }
510 // We've run out of (1,3) simplices to try
511 std::string const msg = "No (2,6) move possible.\n";
512 return std::unexpected(msg);
513 } // do_26_move_impl()

569 [[nodiscard]] inline auto is_62_movable(Delaunay const& triangulation,
570 Vertex_handle const& candidate)
571 -> bool
572 {
573 if (triangulation.dimension() != 3) { return false; }
574
575 if (!triangulation.tds().is_vertex(candidate)) { return false; }
576
577 // We must have 5 incident edges to have 6 incident cells
578 if (auto incident_edges = triangulation.degree(candidate);
579 incident_edges != 5) // NOLINT
580 {
581 return false;
582 }
583
584 // Obtain all incident cells
585 Cell_container incident_cells;
586 triangulation.tds().incident_cells(candidate,
587 std::back_inserter(incident_cells));
588
589 // We must have 6 cells incident to the vertex to make a (6,2) move
590 if (incident_cells.size() != 6) // NOLINT
591 {
592 return false;
593 }
594
595 // Check that none of the incident cells are infinite
596 for (auto const& cell : incident_cells)
597 {
598 if (triangulation.is_infinite(cell)) { return false; }
599 }
600
601 auto const cell_type_count = [&](Cell_type const type) {
602 return std::ranges::count_if(incident_cells, [&](auto const& cell) {
603 return foliated_triangulations::expected_cell_type<3>(cell) == type;
604 });
605 };
606 auto const incident_31 = cell_type_count(Cell_type::THREE_ONE);
607 auto const incident_22 = cell_type_count(Cell_type::TWO_TWO);
608 auto const incident_13 = cell_type_count(Cell_type::ONE_THREE);
609
610 // All cells should be causally classified and carry matching metadata.
611 if (incident_13 + incident_22 + incident_31 != 6 || // NOLINT
612 !std::ranges::all_of(incident_cells, [](auto const& cell) {
613 return foliated_triangulations::is_cell_type_correct<3>(cell);
614 }))
615 {
616 return false;
617 }
618
619 return incident_31 == 3 && incident_22 == 0 && incident_13 == 3;
620
621 } // find_62_moves()

636 template <std::uniform_random_bit_generator Generator,
637 typename Post_mutation_validator>
638 requires std::predicate<Post_mutation_validator&, Delaunay const&>
639 [[nodiscard]] inline auto detail::try_62_move_impl(
640 Delaunay const& source_triangulation,
641 Vertex_handle const source_candidate, Generator& generator,
642 Post_mutation_validator post_mutation_validator)
643 -> std::optional<Delaunay>
644 {
645 if (!source_triangulation.tds().is_vertex(source_candidate) ||
646 source_triangulation.is_infinite(source_candidate))
647 {
648 return std::nullopt;
649 }
650
651 auto const candidate_point = source_candidate->point();
652 Delaunay triangulation{source_triangulation};
653 auto const copied_candidate =
654 foliated_triangulations::find_vertex<3>(triangulation, candidate_point);
655 if (!copied_candidate) { return std::nullopt; }
656
657 auto const candidate = *copied_candidate;
658 auto& tds = triangulation.tds();
659 auto const old_cells = triangulation.number_of_finite_cells();
660 auto const old_vertices = triangulation.number_of_vertices();
661
662 Edge_container incident_edges;
663 triangulation.finite_incident_edges(candidate,
664 std::back_inserter(incident_edges));
665 std::ranges::shuffle(incident_edges, generator);
666
667 auto const is_timelike = [](Edge_handle const& edge) {
668 auto const first_time = edge.first->vertex(edge.second)->info();
669 auto const second_time = edge.first->vertex(edge.third)->info();
670 return first_time != second_time;
671 };
672 auto const flipped = std::ranges::any_of(
673 incident_edges,
674 [&](auto const& edge) { return is_timelike(edge) && tds.flip(edge); });
675 if (!flipped || tds.degree(candidate) != 4 || !tds.is_valid())
676 {
677 return std::nullopt;
678 }
679
680 tds.remove_from_maximal_dimension_simplex(candidate);
681 if (!post_mutation_validator(static_cast<Delaunay const&>(triangulation)) ||
682 !tds.is_valid() ||
683 triangulation.number_of_finite_cells() + 4 != old_cells ||
684 triangulation.number_of_vertices() + 1 != old_vertices)
685 {
686 return std::nullopt;
687 }
688
689 for (auto const cell : triangulation.finite_cell_handles())
690 {
691 auto const type = foliated_triangulations::expected_cell_type<3>(cell);
692 if (type == Cell_type::ACAUSAL || type == Cell_type::UNCLASSIFIED)
693 {
694 return std::nullopt;
695 }
696 cell->info() = static_cast<Int_precision>(type);
697 }
698
699 return triangulation;
700 } // try_62_move_impl()

856 // Internal validation seam used to test rejection after mutation.
857 template <typename Post_mutation_validator>
858 requires std::predicate<Post_mutation_validator&, Delaunay const&>
859 [[nodiscard]] inline auto detail::bistellar_flip_impl(
860 Delaunay const& source_triangulation, Edge_handle const source_edge,
861 Vertex_handle const source_top, Vertex_handle const source_bottom,
862 Post_mutation_validator post_mutation_validator)
863 -> std::optional<Delaunay>
864 {
865 if (!is_well_formed_edge(source_edge) || source_top == nullptr ||
866 source_bottom == nullptr ||
867 !source_triangulation.tds().is_edge(
868 source_edge.first, source_edge.second, source_edge.third) ||
869 !source_triangulation.tds().is_vertex(source_top) ||
870 !source_triangulation.tds().is_vertex(source_bottom) ||
871 source_triangulation.is_infinite(source_top) ||
872 source_triangulation.is_infinite(source_bottom))
873 {
874 return std::nullopt;
875 }
876
877 // CGAL copy construction duplicates its vertices and cells, invalidating
878 // all handles from the source triangulation. Capture stable point values,
879 // then resolve the corresponding handles in the private copy before
880 // attempting the mutation.
881 auto const pivot_from_1_point =
882 source_edge.first->vertex(source_edge.second)->point();
883 auto const pivot_from_2_point =
884 source_edge.first->vertex(source_edge.third)->point();
885 auto const top_point = source_top->point();
886 auto const bottom_point = source_bottom->point();
887
888 Delaunay triangulation{source_triangulation};
889 auto const pivot_from_1_handle = foliated_triangulations::find_vertex<3>(
890 triangulation, pivot_from_1_point);
891 auto const pivot_from_2_handle = foliated_triangulations::find_vertex<3>(
892 triangulation, pivot_from_2_point);
893 auto const top_handle =
894 foliated_triangulations::find_vertex<3>(triangulation, top_point);
895 auto const bottom_handle =
896 foliated_triangulations::find_vertex<3>(triangulation, bottom_point);
897 if (!pivot_from_1_handle || !pivot_from_2_handle || !top_handle ||
898 !bottom_handle)
899 {
900 return std::nullopt;
901 }
902
903 auto const pivot_from_1 = *pivot_from_1_handle;
904 auto const pivot_from_2 = *pivot_from_2_handle;
905 auto const top = *top_handle;
906 auto const bottom = *bottom_handle;
907
908 Cell_handle copied_edge_cell = nullptr;
909 int copied_edge_first_index{};
910 int copied_edge_second_index{};
911 if (!triangulation.is_edge(pivot_from_1, pivot_from_2, copied_edge_cell,
912 copied_edge_first_index,
913 copied_edge_second_index))
914 {
915 return std::nullopt;
916 }
917 Edge_handle const edge{copied_edge_cell, copied_edge_first_index,
918 copied_edge_second_index};
919
920 auto const incident_cells = get_incident_cells(triangulation, edge);
921 if (!incident_cells || incident_cells->size() != 4 || top == bottom ||
922 top == pivot_from_1 || top == pivot_from_2 || bottom == pivot_from_1 ||
923 bottom == pivot_from_2 ||
924 std::ranges::any_of(*incident_cells, [&](auto const& cell) {
925 return triangulation.is_infinite(cell) || !cell->is_valid();
926 }))
927 {
928 return std::nullopt;
929 }
930
931 auto const incident_count = [&](Vertex_handle const vertex) {
932 return std::ranges::count_if(
933 *incident_cells,
934 [&](Cell_handle const cell) { return cell->has_vertex(vertex); });
935 };
936 if (incident_count(top) != 2 || incident_count(bottom) != 2)
937 {
938 return std::nullopt;
939 }
940
941 // A 3D 4-to-4 bistellar move is the composition of CGAL's checked TDS
942 // 2-to-3 facet flip and checked 3-to-2 edge flip. The TDS operations are
943 // intentional: either geometrically checked Triangulation_3 facet flip
944 // can reject the transient, degenerate intermediate even when the final
945 // 4-to-4 configuration is valid. Flipping a facet that contains the old
946 // edge and either boundary vertex creates the new diagonal; the old edge
947 // then has degree three and can be removed by the second flip.
948 Cell_handle boundary_facet_cell = nullptr;
949 int pivot_from_1_index{};
950 int pivot_from_2_index{};
951 int boundary_index{};
952 if (!triangulation.is_facet(pivot_from_1, pivot_from_2, bottom,
953 boundary_facet_cell, pivot_from_1_index,
954 pivot_from_2_index, boundary_index))
955 {
956 return std::nullopt;
957 }
958 auto constexpr cell_index_sum = 0 + 1 + 2 + 3;
959 auto const boundary_facet_index = cell_index_sum - pivot_from_1_index -
960 pivot_from_2_index - boundary_index;
961 if (!triangulation.tds().flip(
962 Delaunay::Facet{boundary_facet_cell, boundary_facet_index}))
963 {
964 return std::nullopt;
965 }
966
967 Cell_handle old_edge_cell = nullptr;
968 int old_edge_first_index{};
969 int old_edge_second_index{};
970 if (!triangulation.is_edge(pivot_from_1, pivot_from_2, old_edge_cell,
971 old_edge_first_index, old_edge_second_index) ||
972 !triangulation.tds().flip(Delaunay::Edge{
973 old_edge_cell, old_edge_first_index, old_edge_second_index}))
974 {
975 return std::nullopt;
976 }
977
978 if (!post_mutation_validator(static_cast<Delaunay const&>(triangulation)) ||
979 !triangulation.tds().is_valid())
980 {
981 return std::nullopt;
982 }
983
984 for (auto const cell : triangulation.finite_cell_handles())
985 {
986 cell->info() = static_cast<Int_precision>(
987 foliated_triangulations::expected_cell_type<3>(cell));
988 }
989
990 return triangulation;
991 } // bistellar_flip_impl()

1066 template <std::uniform_random_bit_generator Generator>
1067 [[nodiscard]] inline auto do_44_move(Manifold const& t_manifold,
1068 Generator& generator) -> Expected
1069 {
1070 auto triangulation = t_manifold.delaunay_snapshot();
1071 auto spacelike_edges = foliated_triangulations::filter_edges<3>(
1072 foliated_triangulations::collect_edges<3>(triangulation), false);
1073 // Shuffle the container to pick a random sequence of edges to try
1074 std::ranges::shuffle(spacelike_edges, generator);
1075 for (auto const& edge : spacelike_edges)
1076 {
1077 // Obtain all incident cells
1078 if (auto const incident_cells =
1079 find_bistellar_flip_location(triangulation, edge);
1080 incident_cells)
1081 {
1082 // Get edge vertices
1083 auto const& v1 = edge.first->vertex(edge.second);
1084 auto const& v2 = edge.first->vertex(edge.third);
1085
1086 // Find top and bottom vertices
1087 Vertex_handle top = nullptr;
1088 Vertex_handle bottom = nullptr;
1089
1090 // Analyze cells to find the top and bottom vertices
1091 for (auto const& cell : *incident_cells)
1092 {
1093 for (int i = 0; i < 4; ++i)
1094 {
1095 auto vertex = cell->vertex(i);
1096 if (vertex != v1 && vertex != v2)
1097 {
1098 // Use timevalue to determine if it's a top or bottom vertex
1099 if (top == nullptr || vertex->info() > top->info())
1100 {
1101 top = vertex;
1102 }
1103 if (bottom == nullptr || vertex->info() < bottom->info())
1104 {
1105 bottom = vertex;
1106 }
1107 }
1108 }
1109 }
1110
1111 // Try the bistellar flip
1112 if (auto flipped_triangulation =
1113 bistellar_flip(triangulation, edge, top, bottom);
1114 flipped_triangulation.has_value())
1115 {
1116 return detail::make_manifold(std::move(*flipped_triangulation),
1117 t_manifold);
1118 }
1119
1120 // The flip failed; continue trying other edges.
1121 }
1122 // Try next edge
1123 }
1124 // We've run out of edges to try
1125 std::string const msg = "No (4,4) move possible.\n";
1126 return std::unexpected(msg);
1127 } // do_44_move()

1132 template <std::uniform_random_bit_generator Generator>
1133 [[nodiscard]] inline auto propose_44_move(Manifold const& t_manifold,
1134 Generator& generator) -> Expected
1135 {
1136 auto triangulation = t_manifold.delaunay_snapshot();
1137 auto spacelike_edges = foliated_triangulations::filter_edges<3>(
1138 foliated_triangulations::collect_edges<3>(triangulation), false);
1139 auto const candidate = detail::random_element(spacelike_edges, generator);
1140 if (!candidate)
1141 {
1142 return std::unexpected(
1143 "No spacelike edge is available for a (4,4) proposal.\n");
1144 }
1145
1146 auto const incident_cells =
1147 find_bistellar_flip_location(triangulation, *candidate);
1148 if (!incident_cells)
1149 {
1150 return std::unexpected("Selected (4,4) proposal site is not movable.\n");
1151 }
1152
1153 auto const& v1 = candidate->first->vertex(candidate->second);
1154 auto const& v2 = candidate->first->vertex(candidate->third);
1155 Vertex_handle top = nullptr;
1156 Vertex_handle bottom = nullptr;
1157 for (auto const& cell : *incident_cells)
1158 {
1159 for (int index = 0; index < 4; ++index)
1160 {
1161 auto const vertex = cell->vertex(index);
1162 if (vertex == v1 || vertex == v2) { continue; }
1163 if (top == nullptr || vertex->info() > top->info()) { top = vertex; }
1164 if (bottom == nullptr || vertex->info() < bottom->info())
1165 {
1166 bottom = vertex;
1167 }
1168 }
1169 }
1170
1171 if (auto flipped = bistellar_flip(triangulation, *candidate, top, bottom))
1172 {
1173 return detail::make_manifold(std::move(*flipped), t_manifold);
1174 }
1175 return std::unexpected(
1176 "Selected (4,4) proposal site could not be flipped.\n");
1177 }
