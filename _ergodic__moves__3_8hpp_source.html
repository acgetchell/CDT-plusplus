188  [[nodiscard]] inline auto do_26_move(Manifold& t_manifold) -> Expected

192 #endif
193  static auto constinit const INCIDENT_CELLS_FOR_6_2_MOVE = 6;
194  auto one_three = t_manifold.get_triangulation().get_one_three();
195  // Shuffle the container to pick a random sequence of (1,3) cells to try
196  std::shuffle(one_three.begin(), one_three.end(),
197  utilities::make_random_generator());
198  for (auto const& bottom : one_three)
199  {
200  if (auto neighboring_31_index = find_adjacent_31_cell(bottom);

202  {
203 #ifndef NDEBUG
204  spdlog::trace("neighboring_31_index is {}.\n", *neighboring_31_index);
205 #endif
206  Cell_handle const top = bottom->neighbor(neighboring_31_index.value());
207  // Calculate the common face with respect to the bottom cell
208  auto common_face_index = std::numeric_limits<int>::max();
209  if (!bottom->has_neighbor(top, common_face_index))
210  {
211  std::string const msg = "Bottom cell does not have a neighbor.\n";
212 #ifndef NDEBUG
213  spdlog::trace(msg);
214 #endif
215  return tl::make_unexpected(msg);
216  }
217 
218  // Get indices of vertices of common face with respect to bottom cell
219  // A face is denoted by the index of the opposite vertex
220  // Thus, the indices of the vertices of the face are all other indices
221  // except the common_face_index
222  // CGAL uses bitwise operations like (common_face_index +1) & 3
223  // We use % 4 which is equivalent and doesn't trigger clang-tidy
224  auto const i_1 = (common_face_index + 1) % 4;
225  auto const i_2 = (common_face_index + 2) % 4;
226  auto const i_3 = (common_face_index + 3) % 4;

228  // Get vertices of common face from indices
229  auto const v_1 = bottom->vertex(i_1);
230  auto const v_2 = bottom->vertex(i_2);
231  auto const v_3 = bottom->vertex(i_3);
232 
233  // Timeslice of vertices should be same
234  if (v_1->info() != v_2->info() || v_2->info() != v_3->info())
235  {
236  std::string const msg = "Vertices have different timeslices.\n";
237 #ifndef NDEBUG
238  spdlog::trace(msg);
239 #endif
240  return tl::make_unexpected(msg);
241  }
242 
243  // Do the (2,6) move
244  // Insert new vertex
245  Vertex_handle const v_center =
246  t_manifold.triangulation().delaunay().tds().insert_in_facet(
247  bottom, *neighboring_31_index);
248 
249  // Checks
250  Cell_container incident_cells;
251  t_manifold.triangulation().delaunay().tds().incident_cells(
252  v_center, std::back_inserter(incident_cells));
253  // the (2,6) center vertex should be bounded by 6 simplices
254  if (incident_cells.size() != INCIDENT_CELLS_FOR_6_2_MOVE)
255  {
256  std::string const msg =
257  "Center vertex is not bounded by 6 simplices.\n";
258 #ifndef NDEBUG
259  spdlog::trace(msg);
260 #endif
261  return tl::make_unexpected(msg);
262  }
263 
264  // Each incident cell should be combinatorially and geometrically valid
265  if (auto check_cells =
266  std::all_of(incident_cells.begin(), incident_cells.end(),
267  [&t_manifold](auto const& cell) {
268  return t_manifold.get_triangulation()
269  .get_delaunay()
270  .tds()
271  .is_cell(cell);
272  });
273  !check_cells)
274  {
275  std::string const msg = "A cell is invalid.\n";
276 #ifndef NDEBUG
277  spdlog::trace(msg);
278 #endif
279  return tl::make_unexpected(msg);
280  }
281 
282  // Now assign a geometric point to the center vertex
283  auto center_point =
284  CGAL::centroid(v_1->point(), v_2->point(), v_3->point());
285 #ifndef NDEBUG
286  spdlog::trace("Center point is: ({}).\n",
287  utilities::point_to_str(center_point));
288 #endif
289  v_center->set_point(center_point);
290 
291  // Assign a timevalue to the new vertex
292  auto timevalue = v_1->info();
293  v_center->info() = timevalue;
294 
295 #ifndef NDEBUG
296  if (t_manifold.is_vertex(v_center))
297  {
298  spdlog::trace("It's a vertex in the TDS.\n");
299  }
300  else { spdlog::trace("It's not a vertex in the TDS.\n"); }
301  spdlog::trace("Spacelike face timevalue is {}.\n", timevalue);
302  spdlog::trace("Inserted vertex ({}) with timevalue {}.\n",
303  utilities::point_to_str(v_center->point()),
304  v_center->info());
305 #endif
306 
307  // Final checks
308  // is_valid() checks for combinatorial and geometric validity
309  if (!t_manifold.get_triangulation().get_delaunay().tds().is_valid(
310  v_center, true, 1))
311  {
312  std::string const msg = "v_center is invalid.\n";
313 #ifndef NDEBUG
314  spdlog::trace(msg);
315 #endif
316  return tl::make_unexpected(msg);
317  }
318 
319  return t_manifold;
320  }
321  // Try next cell
322 #ifndef NDEBUG
323  spdlog::debug("Cell not insertable.\n");
324 #endif
325  }
326  // We've run out of (1,3) simplices to try
327  std::string const msg = "No (2,6) move possible.\n";
328  spdlog::warn(msg);
329  return tl::make_unexpected(msg);
330  } // do_26_move()

340  [[nodiscard]] inline auto is_62_movable(Manifold const& manifold,
341  Vertex_handle const& candidate)
342  -> bool
343  {
344  if (manifold.dimensionality() != 3)
345  {
346 #ifndef NDEBUG
347  spdlog::trace("Manifold is not 3-dimensional.\n");
348 #endif
349  return false;
350  }
351 
352  if (!manifold.is_vertex(candidate))
353  {
354 #ifndef NDEBUG
355  spdlog::trace("Candidate is not a vertex.\n");
356 #endif
357  return false;
358  }
359 
360  // We must have 5 incident edges to have 6 incident cells
361  if (auto incident_edges = manifold.degree(candidate);
362  incident_edges != 5) // NOLINT
363  {
364 #ifndef NDEBUG
365  spdlog::trace("Vertex has {} incident edges.\n", incident_edges);
366 #endif
367  return false;
368  }
369 
370  // Obtain all incident cells
371  auto const incident_cells = manifold.incident_cells(candidate);
372 
373  // We must have 6 cells incident to the vertex to make a (6,2) move
374  if (incident_cells.size() != 6) // NOLINT
375  {
376 #ifndef NDEBUG
377  spdlog::trace("Vertex has {} incident cells.\n", incident_cells.size());
378 #endif
379  return false;
380  }
381 
382  // Check that none of the incident cells are infinite
383  for (auto const& cell : incident_cells)
384  {
385  if (manifold.get_triangulation().is_infinite(cell))
386  {
387 #ifndef NDEBUG
388  spdlog::trace("Cell is infinite.\n");
389 #endif
390  return false;
391  }
392  }
393 
394  // Run until all vertices are fixed
395  while (foliated_triangulations::fix_vertices<3>(
396  manifold.get_triangulation().get_delaunay(), manifold.initial_radius(),
397  manifold.foliation_spacing()))
398  {
399  spdlog::warn("Fixing vertices found by is_62_movable().\n");
400  }
401  // Run until all cells fixed or 10 passes
402  for (auto passes = 1; passes < 11; ++passes) // NOLINT
403  {
404  if (foliated_triangulations::fix_cells<3>(
405  manifold.get_triangulation().get_delaunay()))
406  {
407  spdlog::warn("Fixing cells found by is_62_movable() pass {}.\n",
408  passes);
409  }
410  }
411 
412  auto const incident_31 = foliated_triangulations::filter_cells<3>(
413  incident_cells, Cell_type::THREE_ONE);
414  auto const incident_22 = foliated_triangulations::filter_cells<3>(
415  incident_cells, Cell_type::TWO_TWO);
416  auto const incident_13 = foliated_triangulations::filter_cells<3>(
417  incident_cells, Cell_type::ONE_THREE);
418 
419  // All cells should be classified
420  if ((incident_13.size() + incident_22.size() + incident_31.size()) !=
421  6) // NOLINT
422  {
423  spdlog::warn("Some incident cells on this vertex need to be fixed.\n");
424  }
425 
426 #ifndef NDEBUG
427  spdlog::trace(
428  "Vertex has {} incident cells with {} incident (3,1) simplices and {} "
429  "incident (2,2) simplices and {} incident (1,3) simplices.\n",
430  incident_cells.size(), incident_31.size(), incident_22.size(),
431  incident_13.size());
432  foliated_triangulations::debug_print_cells<3>(std::span{incident_cells});
433 #endif
434  return ((incident_31.size() == 3) && (incident_22.empty()) &&
435  (incident_13.size() == 3));
436 
437  } // find_62_moves()

489  [[nodiscard]] inline auto incident_cells_from_edge(
490  Delaunay_t<3> const& triangulation, Edge_handle const& edge)
491  -> std::optional<Cell_container>
492  {
493  if (!triangulation.tds().is_edge(edge.first, edge.second, edge.third))
494  {
495  return std::nullopt;
496  }
497  // Create the circulator of cells around the edge, starting with the cell
498  // containing the edge
499  auto circulator = triangulation.incident_cells(edge, edge.first);
500  Cell_container incident_cells;
501  // Add cells to the container until we get back to the first one in the
502  // circulator
503  do {
504  // Ignore cells containing the infinite vertex
505  if (!triangulation.is_infinite(circulator))
506  {
507  incident_cells.emplace_back(circulator);
508  }
509  }
510  while (++circulator != edge.first);
511 #ifndef NDEBUG
512  spdlog::trace("Found {} incident cells on edge.\n", incident_cells.size());
513 #endif
514  return incident_cells;
515  } // incident_cells_from_edge()

570  [[nodiscard]] inline auto bistellar_flip(Delaunay& triangulation,
571  Edge_handle const& edge,
572  Vertex_handle const& top,
573  Vertex_handle const& bottom)
574  -> std::optional<Delaunay>
575  {
576  // Get the cells incident to the edge
577  auto incident_cells = get_incident_cells(triangulation, edge);
578 
579  // Check that there are exactly 4 incident cells
580  if (!incident_cells || incident_cells->size() != 4) { return std::nullopt; }
581 
582  // Check incident cells are valid
583  if (std::any_of(incident_cells->begin(), incident_cells->end(),
584  [](auto const& cell) { return !cell->is_valid(); }))
585  {
586  return std::nullopt;
587  }
588 
589  // Get vertices from pivot edge
590  auto const& pivot_from_1 = edge.first->vertex(edge.second);
591  auto const& pivot_from_2 = edge.first->vertex(edge.third);
592 
593  // Get vertices from cells
594  auto vertices = foliated_triangulations::get_vertices_from_cells<3>(
595  incident_cells.value());
596 
597  // Get vertices for new pivot edge
598  Vertex_container new_pivot_vertices;
599  std::copy_if(vertices.begin(), vertices.end(),
600  std::back_inserter(new_pivot_vertices),
601  [&](auto const& vertex) {
602  return vertex != pivot_from_1 && vertex != pivot_from_2 &&
603  vertex != top && vertex != bottom;
604  });
605 
606  // Check that there are exactly 2 new pivot vertices
607  if (new_pivot_vertices.size() != 2) { return std::nullopt; }
608 
609  // Label the vertices in the new pivot edge
610  auto const& pivot_to_1 = new_pivot_vertices[0];
611  auto const& pivot_to_2 = new_pivot_vertices[1];
612 
613  // Now we need to classify the cells by the vertices they contain
614  Cell_handle before_1; // top, pivot_from_1, pivot_from_2, pivot_to_1
615  Cell_handle before_2; // top, pivot_from_1, pivot_from_2, pivot_to_2
616  Cell_handle before_3; // bottom, pivot_from_1, pivot_from_2, pivot_to_1
617  Cell_handle before_4; // bottom, pivot_from_1, pivot_from_2, pivot_to_2
618  for (auto const& cell : incident_cells.value())
619  {
620  if (cell->has_vertex(top))
621  {
622  if (cell->has_vertex(pivot_to_1)) { before_1 = cell; }
623  else { before_2 = cell; }
624  }
625  else
626  {
627  if (cell->has_vertex(pivot_to_1)) { before_3 = cell; }
628  else { before_4 = cell; }
629  }
630  }
631 
632  // Verify these cells are valid
633  if (!before_1->is_valid() || !before_2->is_valid() ||
634  !before_3->is_valid() || !before_4->is_valid())
635  {
636  return std::nullopt;
637  }
638 
639 #ifndef NDEBUG
640  fmt::print("Cells in the triangulation before deleting old cells: {}\n",
641  triangulation.number_of_cells());
642 #endif
643 
644  // Now find the exterior neighbors of the cells
645  Cell_handle const n_1 = before_1->neighbor(before_1->index(pivot_from_2));
646  Cell_handle const n_2 = before_1->neighbor(before_1->index(pivot_from_1));
647  Cell_handle const n_3 = before_2->neighbor(before_2->index(pivot_from_1));
648  Cell_handle const n_4 = before_2->neighbor(before_2->index(pivot_from_2));
649  Cell_handle const n_5 = before_3->neighbor(before_3->index(pivot_from_2));
650  Cell_handle const n_6 = before_3->neighbor(before_3->index(pivot_from_1));
651  Cell_handle const n_7 = before_4->neighbor(before_4->index(pivot_from_1));
652  Cell_handle const n_8 = before_4->neighbor(before_4->index(pivot_from_2));
653 
654  // Next, delete the old cells
655  triangulation.tds().delete_cell(before_1);
656  triangulation.tds().delete_cell(before_2);
657  triangulation.tds().delete_cell(before_3);
658  triangulation.tds().delete_cell(before_4);
659 
660 #ifndef NDEBUG
661  fmt::print("Cells in the triangulation after deleting old cells: {}\n",
662  triangulation.number_of_cells());
663 #endif
664 
665  // Now create the new cells
666  Cell_handle const after_1 = triangulation.tds().create_cell(
667  top, pivot_from_1, pivot_to_1, pivot_to_2);
668  Cell_handle const after_2 = triangulation.tds().create_cell(
669  top, pivot_from_2, pivot_to_1, pivot_to_2);
670  Cell_handle const after_3 = triangulation.tds().create_cell(
671  bottom, pivot_from_1, pivot_to_1, pivot_to_2);
672  Cell_handle const after_4 = triangulation.tds().create_cell(
673  bottom, pivot_from_2, pivot_to_1, pivot_to_2);
674 
675  // Now set the neighbors of the new cells
676  after_1->set_neighbors(n_1, n_4, after_2, after_3);
677  after_2->set_neighbors(n_2, n_3, after_1, after_4);
678  after_3->set_neighbors(n_5, n_8, after_4, after_1);
679  after_4->set_neighbors(n_6, n_7, after_2, after_3);
680

682  n_1->set_neighbor(n_1->index(triangulation.tds().mirror_vertex(
683  after_1, after_1->index(pivot_to_2))),
684  after_1);
685  n_2->set_neighbor(n_2->index(triangulation.tds().mirror_vertex(
686  after_2, after_2->index(pivot_to_2))),
687  after_2);
688  n_3->set_neighbor(n_3->index(triangulation.tds().mirror_vertex(
689  after_2, after_2->index(pivot_to_1))),
690  after_2);
691  n_4->set_neighbor(n_4->index(triangulation.tds().mirror_vertex(
692  after_1, after_1->index(pivot_to_1))),
693  after_1);

695  after_3, after_3->index(pivot_to_2))),

698  after_4, after_4->index(pivot_to_2))),

701  after_4, after_4->index(pivot_to_1))),
