92 [[nodiscard]] inline auto finite_incident_cells(
93 Delaunay const& triangulation, Edge_handle const& edge)
94 -> std::optional<Cell_container>
95 {
96 if (!is_well_formed_edge(edge) || triangulation.dimension() != 3 ||
97 !triangulation.tds().is_edge(edge.first, edge.second, edge.third))
98 {
99 return std::nullopt;
100 }
101
102 auto circulator = triangulation.incident_cells(edge, edge.first);
103 Cell_container incident_cells;
104 do
105 { // NOLINT(cppcoreguidelines-avoid-do-while)
106 if (triangulation.is_infinite(circulator)) { return std::nullopt; }
107 incident_cells.emplace_back(circulator);
108 }
109 while (++circulator != edge.first);
110 return incident_cells;
111 }

182 Generator& generator)
183 -> std::optional<typename Container::value_type>
184 {
185 if (candidates.empty()) { return std::nullopt; }

193 template <typename Container, std::uniform_random_bit_generator Generator,
194 typename Comparator>
195 [[nodiscard]] inline auto canonical_random_element(Container& candidates,
196 Generator& generator,
197 Comparator comparator)
198 -> std::optional<typename Container::value_type>
199 {
200 if (candidates.empty()) { return std::nullopt; }
201 std::uniform_int_distribution<std::size_t> distribution{
202 0, candidates.size() - 1};
203 auto const index = distribution(generator);
204 auto const nth = candidates.begin() +
205 static_cast<typename Container::difference_type>(index);
206 std::ranges::nth_element(candidates, nth, comparator);
207 return candidates[index];
208 }

249 [[nodiscard]] inline auto try_23_move(Delaunay& triangulation,
250 Cell_handle const& to_be_moved) -> bool
251 {
252 if (to_be_moved == nullptr || triangulation.dimension() != 3 ||
253 !triangulation.tds().is_cell(to_be_moved) ||
254 !foliated_triangulations::is_cell_type_correct<3>(to_be_moved) ||
255 foliated_triangulations::expected_cell_type<3>(to_be_moved) !=
256 Cell_type::TWO_TWO)
257 {
258 return false;
259 }
260 std::array facet_indices{0, 1, 2, 3};
261 std::ranges::sort(facet_indices, [&](auto const left, auto const right) {
262 return detail::point_less(to_be_moved->vertex(left)->point(),
263 to_be_moved->vertex(right)->point());
264 });
265 auto flipped = false;
266 // Try every facet of the (2,2) cell
267 for (auto const i : facet_indices)
268 {
269 auto const neighbor = to_be_moved->neighbor(i);
270 if (triangulation.is_infinite(neighbor)) { continue; }
271
272 auto const neighbor_type =
273 foliated_triangulations::expected_cell_type<3>(neighbor);
274 if (!foliated_triangulations::is_cell_type_correct<3>(neighbor) ||
275 (neighbor_type != Cell_type::THREE_ONE &&
276 neighbor_type != Cell_type::ONE_THREE))
277 {
278 continue;
279 }
280
281 // A causal (2,3) move must replace the facet with a timelike edge.
282 // CGAL also permits topological flips that create a spacelike edge.
283 auto const mirror_index = neighbor->index(to_be_moved);
284 auto const first_time =
285 static_cast<long long>(to_be_moved->vertex(i)->info());
286 auto const second_time =
287 static_cast<long long>(neighbor->vertex(mirror_index)->info());
288 auto const time_difference = first_time > second_time
289 ? first_time - second_time
290 : second_time - first_time;
291 if (time_difference != 1) { continue; }
292
293 if (triangulation.flip(to_be_moved, i))
294 {
295 flipped = true;
296 break;
297 }
298 }
299 return flipped;
300 } // try_23_move

319 template <std::uniform_random_bit_generator Generator>
320 [[nodiscard]] inline auto do_23_move(Manifold const& t_manifold,
321 Generator& generator) -> Expected
322 {
323 Delaunay triangulation{t_manifold.delaunay_snapshot()};
324 auto two_two = foliated_triangulations::filter_cells<3>(
325 foliated_triangulations::collect_cells<3>(triangulation),
326 Cell_type::TWO_TWO);
327 detail::canonicalize(two_two);
328 // Shuffle the container to create a random sequence of (2,2) cells
329 std::ranges::shuffle(two_two, generator);
330 // Try a (2,3) move on successive cells in the sequence
331 if (std::ranges::any_of(two_two, [&](auto& cell) {
332 return try_23_move(triangulation, cell);
333 }))
334 {
335 return detail::make_manifold(std::move(triangulation), t_manifold);
336 }
337 // We've run out of (2,2) cells
338 std::string const msg = "No (2,3) move possible.\n";
339 return std::unexpected(msg);
340 }

365 [[nodiscard]] inline auto is_32_movable(Delaunay const& triangulation,
366 Edge_handle const& candidate)
367 -> bool
368 {
369 auto const incident_cells =
370 finite_incident_cells(triangulation, candidate);
371 if (!incident_cells || incident_cells->size() != 3) { return false; }
372
373 auto const first_time = static_cast<long long>(
374 candidate.first->vertex(candidate.second)->info());
375 auto const second_time = static_cast<long long>(
376 candidate.first->vertex(candidate.third)->info());
377 auto const time_difference = first_time > second_time
378 ? first_time - second_time
379 : second_time - first_time;
380 if (time_difference != 1) { return false; }
381
382 if (!std::ranges::all_of(*incident_cells, [](auto const& cell) {
383 return foliated_triangulations::is_cell_type_correct<3>(cell);
384 }))
385 {
386 return false;
387 }
388
389 auto const cell_type_count = [&](Cell_type const type) {
390 return std::ranges::count_if(*incident_cells, [&](auto const& cell) {
391 return foliated_triangulations::expected_cell_type<3>(cell) == type;
392 });
393 };
394 auto const incident_31 = cell_type_count(Cell_type::THREE_ONE);
395 auto const incident_22 = cell_type_count(Cell_type::TWO_TWO);
396 auto const incident_13 = cell_type_count(Cell_type::ONE_THREE);
397 return incident_22 == 2 && ((incident_31 == 1 && incident_13 == 0) ||
398 (incident_31 == 0 && incident_13 == 1));
399 }

477 [[nodiscard]] inline auto find_adjacent_31_cell(Cell_handle const& t_cell)
478 -> std::optional<int>
479 {
480 if (t_cell == nullptr ||
481 !foliated_triangulations::is_cell_type_correct<3>(t_cell) ||
482 foliated_triangulations::expected_cell_type<3>(t_cell) !=
483 Cell_type::ONE_THREE)
484 {
485 return std::nullopt;
486 }
487 std::vector<int> candidates;
488 for (auto i = 0; i < 4; ++i)
489 {
490 auto const neighbor = t_cell->neighbor(i);
491 if (foliated_triangulations::is_cell_type_correct<3>(neighbor) &&
492 foliated_triangulations::expected_cell_type<3>(neighbor) ==
493 Cell_type::THREE_ONE)
494 {
495 candidates.emplace_back(i);
496 }
497 }
498 if (candidates.empty()) { return std::nullopt; }
499 std::ranges::sort(candidates, [&](auto const left, auto const right) {
500 auto const left_points =
501 detail::canonical_cell_points(t_cell->neighbor(left));
502 auto const right_points =
503 detail::canonical_cell_points(t_cell->neighbor(right));
504 return std::lexicographical_compare(
505 left_points.begin(), left_points.end(), right_points.begin(),
506 right_points.end(), detail::point_less);
507 });
508 return candidates.front();
509 } // find_26_move()

522 // Internal validation seam used to test rejection after mutation.
523 template <std::uniform_random_bit_generator Generator,
524 typename Post_mutation_validator>
525 requires std::predicate<Post_mutation_validator&, Delaunay const&>
526 [[nodiscard]] inline auto detail::do_26_move_impl(
527 Manifold const& t_manifold, Generator& generator,
528 bool const only_first_site,
529 Post_mutation_validator post_mutation_validator) -> Expected
530 {
531 static auto constexpr INCIDENT_CELLS_FOR_6_2_MOVE = 6;
532 Delaunay triangulation{t_manifold.delaunay_snapshot()};
533 auto one_three = foliated_triangulations::filter_cells<3>(
534 foliated_triangulations::collect_cells<3>(triangulation),
535 Cell_type::ONE_THREE);
536 if (only_first_site)
537 {
538 auto const candidate =
539 detail::canonical_random_element(one_three, generator);
540 if (!candidate)
541 {
542 return std::unexpected("No (2,6) proposal site is available.\n");
543 }
544 one_three = {*candidate};
545 }
546 else
547 {
548 detail::canonicalize(one_three);
549 // Shuffle the container to pick a random sequence of (1,3) cells to
550 // try.
551 std::ranges::shuffle(one_three, generator);
552 }
553 for (auto const& bottom : one_three)
554 {
555 if (auto neighboring_31_index = find_adjacent_31_cell(bottom);
556 neighboring_31_index)
557 {
558 Cell_handle const top = bottom->neighbor(neighboring_31_index.value());
559 // Calculate the common face with respect to the bottom cell
560 auto common_face_index = std::numeric_limits<int>::max();
561 if (!bottom->has_neighbor(top, common_face_index))
562 {
563 std::string const msg = "Bottom cell does not have a neighbor.\n";
564 return std::unexpected(msg);
565 }
566
567 // Get indices of vertices of common face with respect to bottom cell
568 // A face is denoted by the index of the opposite vertex
569 // Thus, the indices of the vertices of the face are all other indices
570 // except the common_face_index
571 // CGAL uses bitwise operations like (common_face_index +1) & 3
572 // We use % 4 which is equivalent and doesn't trigger clang-tidy
573 auto const i_1 = (common_face_index + 1) % 4;
574 auto const i_2 = (common_face_index + 2) % 4;
575 auto const i_3 = (common_face_index + 3) % 4;
576
577 // Get vertices of common face from indices
578 auto const v_1 = bottom->vertex(i_1);
579 auto const v_2 = bottom->vertex(i_2);
580 auto const v_3 = bottom->vertex(i_3);
581
582 // Timeslice of vertices should be same
583 if (v_1->info() != v_2->info() || v_2->info() != v_3->info())
584 {
585 std::string const msg = "Vertices have different timeslices.\n";
586 return std::unexpected(msg);
587 }
588
589 // Do the (2,6) move
590 // Insert new vertex
591 Vertex_handle const v_center =
592 triangulation.tds().insert_in_facet(bottom, *neighboring_31_index);
593
594 // Checks
595 Cell_container incident_cells;
596 triangulation.tds().incident_cells(v_center,
597 std::back_inserter(incident_cells));
598 // the (2,6) center vertex should be bounded by 6 simplices
599 if (incident_cells.size() != INCIDENT_CELLS_FOR_6_2_MOVE)
600 {
601 std::string const msg =
602 "Center vertex is not bounded by 6 simplices.\n";
603 return std::unexpected(msg);
604 }
605
606 // Each incident cell should be combinatorially and geometrically
607 // valid
608 if (auto check_cells =
609 std::ranges::all_of(incident_cells,
610 [&triangulation](auto const& cell) {
611 return triangulation.tds().is_cell(cell);
612 });
613 !check_cells)
614 {
615 std::string const msg = "A cell is invalid.\n";
616 return std::unexpected(msg);
617 }
618
619 // Now assign a geometric point to the center vertex
620 std::array face_points{v_1->point(), v_2->point(), v_3->point()};
621 std::ranges::sort(face_points, detail::point_less);
622 auto const center_point =
623 CGAL::centroid(face_points[0], face_points[1], face_points[2]);
624 v_center->set_point(center_point);
625
626 // Assign a timevalue to the new vertex
627 auto timevalue = v_1->info();
628 v_center->info() = timevalue;
629
630 // Final checks
631 // is_valid() checks for combinatorial and geometric validity
632 if (!post_mutation_validator(
633 static_cast<Delaunay const&>(triangulation)) ||
634 !triangulation.tds().is_valid(v_center, true, 1))
635 {
636 std::string const msg = "v_center is invalid.\n";
637 return std::unexpected(msg);
638 }
639
640 return make_manifold(std::move(triangulation), t_manifold);
641 }
642 }
643 // We've run out of (1,3) simplices to try
644 std::string const msg = "No (2,6) move possible.\n";
645 return std::unexpected(msg);
646 } // do_26_move_impl()

702 [[nodiscard]] inline auto is_62_movable(Delaunay const& triangulation,
703 Vertex_handle const& candidate)
704 -> bool
705 {
706 if (triangulation.dimension() != 3) { return false; }
707
708 if (!triangulation.tds().is_vertex(candidate)) { return false; }
709
710 // We must have 5 incident edges to have 6 incident cells
711 if (auto incident_edges = triangulation.degree(candidate);
712 incident_edges != 5) // NOLINT
713 {
714 return false;
715 }
716
717 // Obtain all incident cells
718 Cell_container incident_cells;
719 triangulation.tds().incident_cells(candidate,
720 std::back_inserter(incident_cells));
721
722 // We must have 6 cells incident to the vertex to make a (6,2) move
723 if (incident_cells.size() != 6) // NOLINT
724 {
725 return false;
726 }
727
728 // Check that none of the incident cells are infinite
729 for (auto const& cell : incident_cells)
730 {
731 if (triangulation.is_infinite(cell)) { return false; }
732 }
733
734 auto const cell_type_count = [&](Cell_type const type) {
735 return std::ranges::count_if(incident_cells, [&](auto const& cell) {
736 return foliated_triangulations::expected_cell_type<3>(cell) == type;
737 });
738 };
739 auto const incident_31 = cell_type_count(Cell_type::THREE_ONE);
740 auto const incident_22 = cell_type_count(Cell_type::TWO_TWO);
741 auto const incident_13 = cell_type_count(Cell_type::ONE_THREE);
742
743 // All cells should be causally classified and carry matching metadata.
744 if (incident_13 + incident_22 + incident_31 != 6 || // NOLINT
745 !std::ranges::all_of(incident_cells, [](auto const& cell) {
746 return foliated_triangulations::is_cell_type_correct<3>(cell);
747 }))
748 {
749 return false;
750 }
751
752 return incident_31 == 3 && incident_22 == 0 && incident_13 == 3;
753
754 } // find_62_moves()

768 // Internal validation seam used to test rejection after mutation.
769 template <std::uniform_random_bit_generator Generator,
770 typename Post_mutation_validator>
771 requires std::predicate<Post_mutation_validator&, Delaunay const&>
772 [[nodiscard]] inline auto detail::try_62_move_impl(
773 Delaunay const& source_triangulation,
774 Vertex_handle const source_candidate, Generator& generator,
775 Post_mutation_validator post_mutation_validator)
776 -> std::optional<Delaunay>
777 {
778 if (!source_triangulation.tds().is_vertex(source_candidate) ||
779 source_triangulation.is_infinite(source_candidate))
780 {
781 return std::nullopt;
782 }
783
784 auto const candidate_point = source_candidate->point();
785 Delaunay triangulation{source_triangulation};
786 auto const copied_candidate =
787 foliated_triangulations::find_vertex<3>(triangulation, candidate_point);
788 if (!copied_candidate) { return std::nullopt; }
789
790 auto const candidate = *copied_candidate;
791 auto& tds = triangulation.tds();
792 auto const old_cells = triangulation.number_of_finite_cells();
793 auto const old_vertices = triangulation.number_of_vertices();
794
795 Edge_container incident_edges;
796 triangulation.finite_incident_edges(candidate,
797 std::back_inserter(incident_edges));
798 detail::canonicalize(incident_edges);
799 std::ranges::shuffle(incident_edges, generator);
800
801 auto const is_timelike = [](Edge_handle const& edge) {
802 auto const first_time = edge.first->vertex(edge.second)->info();
803 auto const second_time = edge.first->vertex(edge.third)->info();
804 return first_time != second_time;
805 };
806 auto const flipped = std::ranges::any_of(
807 incident_edges,
808 [&](auto const& edge) { return is_timelike(edge) && tds.flip(edge); });
809 if (!flipped || tds.degree(candidate) != 4 || !tds.is_valid())
810 {
811 return std::nullopt;
812 }
813
814 tds.remove_from_maximal_dimension_simplex(candidate);
815 if (!post_mutation_validator(static_cast<Delaunay const&>(triangulation)) ||
816 !tds.is_valid() ||
817 triangulation.number_of_finite_cells() + 4 != old_cells ||
818 triangulation.number_of_vertices() + 1 != old_vertices)
819 {
820 return std::nullopt;
821 }
822
823 for (auto const cell : triangulation.finite_cell_handles())
824 {
825 auto const type = foliated_triangulations::expected_cell_type<3>(cell);
826 if (type == Cell_type::ACAUSAL || type == Cell_type::UNCLASSIFIED)
827 {
828 return std::nullopt;
829 }
830 cell->info() = static_cast<Int_precision>(type);
831 }
832
833 return triangulation;
834 } // try_62_move_impl()

992 // Internal validation seam used to test rejection after mutation.
993 template <typename Post_mutation_validator>
994 requires std::predicate<Post_mutation_validator&, Delaunay const&>
995 [[nodiscard]] inline auto detail::bistellar_flip_impl(
996 Delaunay const& source_triangulation, Edge_handle const source_edge,
997 Vertex_handle const source_top, Vertex_handle const source_bottom,
998 Post_mutation_validator post_mutation_validator)
999 -> std::optional<Delaunay>
1000 {
1001 if (!is_well_formed_edge(source_edge) || source_top == nullptr ||
1002 source_bottom == nullptr ||
1003 !source_triangulation.tds().is_edge(
1004 source_edge.first, source_edge.second, source_edge.third) ||
1005 !source_triangulation.tds().is_vertex(source_top) ||
1006 !source_triangulation.tds().is_vertex(source_bottom) ||
1007 source_triangulation.is_infinite(source_top) ||
1008 source_triangulation.is_infinite(source_bottom))
1009 {
1010 return std::nullopt;
1011 }
1012
1013 // CGAL copy construction duplicates its vertices and cells, invalidating
1014 // all handles from the source triangulation. Capture stable point values,
1015 // then resolve the corresponding handles in the private copy before
1016 // attempting the mutation.
1017 auto const pivot_from_1_point =
1018 source_edge.first->vertex(source_edge.second)->point();
1019 auto const pivot_from_2_point =
1020 source_edge.first->vertex(source_edge.third)->point();
1021 auto const top_point = source_top->point();
1022 auto const bottom_point = source_bottom->point();
1023
1024 Delaunay triangulation{source_triangulation};
1025 auto const pivot_from_1_handle = foliated_triangulations::find_vertex<3>(
1026 triangulation, pivot_from_1_point);
1027 auto const pivot_from_2_handle = foliated_triangulations::find_vertex<3>(
1028 triangulation, pivot_from_2_point);
1029 auto const top_handle =
1030 foliated_triangulations::find_vertex<3>(triangulation, top_point);
1031 auto const bottom_handle =
1032 foliated_triangulations::find_vertex<3>(triangulation, bottom_point);
1033 if (!pivot_from_1_handle || !pivot_from_2_handle || !top_handle ||
1034 !bottom_handle)
1035 {
1036 return std::nullopt;
1037 }
1038
1039 auto const pivot_from_1 = *pivot_from_1_handle;
1040 auto const pivot_from_2 = *pivot_from_2_handle;
1041 auto const top = *top_handle;
1042 auto const bottom = *bottom_handle;
1043
1044 Cell_handle copied_edge_cell = nullptr;
1045 int copied_edge_first_index{};
1046 int copied_edge_second_index{};
1047 if (!triangulation.is_edge(pivot_from_1, pivot_from_2, copied_edge_cell,
1048 copied_edge_first_index,
1049 copied_edge_second_index))
1050 {
1051 return std::nullopt;
1052 }
1053 Edge_handle const edge{copied_edge_cell, copied_edge_first_index,
1054 copied_edge_second_index};
1055
1056 auto const incident_cells = get_incident_cells(triangulation, edge);
1057 if (!incident_cells || incident_cells->size() != 4 || top == bottom ||
1058 top == pivot_from_1 || top == pivot_from_2 || bottom == pivot_from_1 ||
1059 bottom == pivot_from_2 ||
1060 std::ranges::any_of(*incident_cells, [&](auto const& cell) {
1061 return triangulation.is_infinite(cell) || !cell->is_valid();
1062 }))
1063 {
1064 return std::nullopt;
1065 }
1066
1067 auto const incident_count = [&](Vertex_handle const vertex) {
1068 return std::ranges::count_if(
1069 *incident_cells,
1070 [&](Cell_handle const cell) { return cell->has_vertex(vertex); });
1071 };
1072 if (incident_count(top) != 2 || incident_count(bottom) != 2)
1073 {
1074 return std::nullopt;
1075 }
1076
1077 // A 3D 4-to-4 bistellar move is the composition of CGAL's checked TDS
1078 // 2-to-3 facet flip and checked 3-to-2 edge flip. The TDS operations are
1079 // intentional: either geometrically checked Triangulation_3 facet flip
1080 // can reject the transient, degenerate intermediate even when the final
1081 // 4-to-4 configuration is valid. Flipping a facet that contains the old
1082 // edge and either boundary vertex creates the new diagonal; the old edge
1083 // then has degree three and can be removed by the second flip.
1084 Cell_handle boundary_facet_cell = nullptr;
1085 int pivot_from_1_index{};
1086 int pivot_from_2_index{};
1087 int boundary_index{};
1088 if (!triangulation.is_facet(pivot_from_1, pivot_from_2, bottom,
1089 boundary_facet_cell, pivot_from_1_index,
1090 pivot_from_2_index, boundary_index))
1091 {
1092 return std::nullopt;
1093 }
1094 auto constexpr cell_index_sum = 0 + 1 + 2 + 3;
1095 auto const boundary_facet_index = cell_index_sum - pivot_from_1_index -
1096 pivot_from_2_index - boundary_index;
1097 if (!triangulation.tds().flip(
1098 Delaunay::Facet{boundary_facet_cell, boundary_facet_index}))
1099 {
1100 return std::nullopt;
1101 }
1102
1103 Cell_handle old_edge_cell = nullptr;
1104 int old_edge_first_index{};
1105 int old_edge_second_index{};
1106 if (!triangulation.is_edge(pivot_from_1, pivot_from_2, old_edge_cell,
1107 old_edge_first_index, old_edge_second_index) ||
1108 !triangulation.tds().flip(Delaunay::Edge{
1109 old_edge_cell, old_edge_first_index, old_edge_second_index}))
1110 {
1111 return std::nullopt;
1112 }
1113
1114 if (!post_mutation_validator(static_cast<Delaunay const&>(triangulation)) ||
1115 !triangulation.tds().is_valid())
1116 {
1117 return std::nullopt;
1118 }
1119
1120 for (auto const cell : triangulation.finite_cell_handles())
1121 {
1122 cell->info() = static_cast<Int_precision>(
1123 foliated_triangulations::expected_cell_type<3>(cell));
1124 }
1125
1126 return triangulation;
1127 } // bistellar_flip_impl()

1203 [[nodiscard]] inline auto do_44_move(Manifold const& t_manifold,
1204 Generator& generator) -> Expected
1205 {
1206 auto triangulation = t_manifold.delaunay_snapshot();
1207 auto spacelike_edges = foliated_triangulations::filter_edges<3>(
1208 foliated_triangulations::collect_edges<3>(triangulation), false);
1209 detail::canonicalize(spacelike_edges);
1210 // Shuffle the container to pick a random sequence of edges to try
1211 std::ranges::shuffle(spacelike_edges, generator);
1212 for (auto const& edge : spacelike_edges)
1213 {
1214 // Obtain all incident cells
1215 if (auto const incident_cells =
1216 find_bistellar_flip_location(triangulation, edge);
1217 incident_cells)
1218 {
1219 // Get edge vertices
1220 auto const& v1 = edge.first->vertex(edge.second);
1221 auto const& v2 = edge.first->vertex(edge.third);
1222
1223 // Find top and bottom vertices
1224 Vertex_handle top = nullptr;
1225 Vertex_handle bottom = nullptr;
1226
1227 // Analyze cells to find the top and bottom vertices
1228 for (auto const& cell : *incident_cells)
1229 {
1230 for (int i = 0; i < 4; ++i)
1231 {
1232 auto vertex = cell->vertex(i);
1233 if (vertex != v1 && vertex != v2)
1234 {
1235 // Use timevalue to determine if it's a top or bottom vertex
1236 if (top == nullptr || detail::vertex_precedes(top, vertex))
1237 {
1238 top = vertex;
1239 }
1240 if (bottom == nullptr || detail::vertex_precedes(vertex, bottom))
1241 {
1242 bottom = vertex;
1243 }
1244 }
1245 }
1246 }
1247
1248 // Try the bistellar flip
1249 if (auto flipped_triangulation =
1250 bistellar_flip(triangulation, edge, top, bottom);
1251 flipped_triangulation.has_value())
1252 {
1253 return detail::make_manifold(std::move(*flipped_triangulation),
1254 t_manifold);
1255 }
1256
1257 // The flip failed; continue trying other edges.
1258 }
1259 // Try next edge
1260 }
1261 // We've run out of edges to try
1262 std::string const msg = "No (4,4) move possible.\n";
1263 return std::unexpected(msg);
1264 } // do_44_move()
